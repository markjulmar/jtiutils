00059 /****************************************************************************/
00060 // RefCountedObject
00061 //
00062 // This class defines a reference counted base class which can be derived
00063 // from in order to create a reference counted object.  This is a more
00064 // efficient approach than the below UsageCountedObject class as the count is
00065 // included within the object being counted.  This saves the allocation
00066 // of the reference count off the heap.
00067 //
00068 // This object assumes the same interface as COM.
00069 //
00070 /****************************************************************************/
00071 template <class _LockType = JTI_REFCOUNT_LOCK_TYPE>
00072 class RefCountedObject
00073 {
00074 // Class data
00075 protected:
00076         mutable long _RefCount;
00077 
00078 // Constructor
00079 public:
00080         RefCountedObject() : _RefCount(1) {/* */}
00081 
00082 // Private constructor - can only be deleted by Release() call or
00083 // called by derived types.
00084 protected:
00085         virtual ~RefCountedObject() {/* */}
00086 
00087 // Access methods
00088 public:
00089         // This method isused to increment the reference counter.
00090         void AddRef() const 
00091         { 
00092                 _LockType::Increment(&_RefCount); 
00093         }
00094 
00095         // This method is used to decrement the reference count.  
00096         // If it hits zero, the held object is deleted.
00097         bool Release() const 
00098         {
00099                 if (_LockType::Decrement(&_RefCount) == 0) {
00100                         delete this;
00101                         return true;
00102                 }
00103                 return false;
00104         }
00105         
00106         // This swaps the reference count for an object with another object
00107         void Swap(RefCountedObject& rhs) { std::swap(_RefCount, rhs._RefCount); }
00108 
00109 // Unavailable methods; violates ref counting.
00110 private:
00111         RefCountedObject(const RefCountedObject&);
00112         RefCountedObject& operator=(const RefCountedObject&);
00113 };

00115 /****************************************************************************/
00116 // UsageCountedObject
00117 //
00118 // This class can be used to implement a copy-on-write usage counted object.
00119 // This seperates the refCount away from the data itself (which is in contrast
00120 // to the above class which combines them).
00121 //
00122 // This class is used to manually implement handle classes when the interface
00123 // is going to be changed, or the full "value-type" experience is desired.
00124 //
00125 /****************************************************************************/
00126 template <class _Ty, class _LockType = JTI_REFCOUNT_LOCK_TYPE>
00127 class UsageCountedObject
00128 {
00129 // Class data
00130 protected:
00131         typedef _Ty _PtrType;
00132 private:
00133         _PtrType* _p;
00134         long* _pnCount;
00135 
00136 // Constructor
00137 public:
00138         UsageCountedObject() : _pnCount(JTI_NEW long(1)), _p(0) {/* */}
00139         UsageCountedObject(_PtrType* p) : _pnCount(JTI_NEW long(1)), _p(p) {/* */}
00140         explicit UsageCountedObject(const UsageCountedObject& cnt) : _p(cnt._p), _pnCount(cnt._pnCount) { _LockType::Increment(_pnCount); }
00141         ~UsageCountedObject() { 
00142                 if (_LockType::Decrement(_pnCount) == 0) {
00143                         delete _pnCount;
00144                         delete _p;
00145                 }
00146         }
00147 
00148 // Friend function which allows underlying ptr access when absolutely necessary; note this is dangerous!
00149 public:
00150         friend _PtrType* get_UnderlyingPointer(UsageCountedObject<_Ty,_LockType>& sp) { return sp._p; }
00151 
00152 // Access methods
00153 protected:
00154         _PtrType* ptr() const { return _p; }
00155 
00156         // Use this method to change the underlying pointer
00157         void Assign(_PtrType* p)
00158         {
00159                 if (_LockType::Decrement(_pnCount) == 0)
00160                 {
00161                         *_pnCount = 1;
00162                         delete _p;
00163                 }
00164                 else
00165                         _pnCount = JTI_NEW long(1);
00166                 _p = p;
00167         }
00168 
00169         // Use this method to assign this use count to a different use count in an assignment.
00170         void Assign(const UsageCountedObject& cnt)
00171         {
00172                 // Increment the new use count so it doesn't go away.
00173                 _LockType::Increment(cnt._pnCount);
00174                 
00175                 // Decrement the old use count and delete if necessary.  Then reattach to the new use count.
00176                 if (_LockType::Decrement(_pnCount) == 0) 
00177                 {
00178                         delete _pnCount;
00179                         delete _p;
00180                 }
00181                 _pnCount =  cnt._pnCount;
00182                 _p = cnt._p;
00183         }
00184 
00185 // Unavailable methods
00186 private:
00187         UsageCountedObject& operator=( const UsageCountedObject& cnt );
00188 };

00190 /****************************************************************************/
00191 // RefCountedPolicy
00192 //
00193 // This class defines a reference counted policy class based on the above
00194 // base class which is used by the smart pointer wrapper class (CRefPtr).
00195 // This is a required parameter to the CRefPtr template.
00196 //
00197 // The interface and concept was (to my knowledge) introduced by 
00198 // Andrei Alexandrescu and presented in a C++ Report article and then later
00199 // expanded in his book "Modern C++ Design".
00200 //
00201 /****************************************************************************/
00202 template <class _Ty>
00203 class RefCountedPolicy
00204 {
00205 // Accessors
00206 public:
00207         // Copies the object
00208         static _Ty* Clone(const _Ty* p) {
00209                 if (p) p->AddRef();
00210                 return const_cast<_Ty*>(p);
00211         }
00212         // Releases the object; return true if the object should
00213         // be deleted by the wrapper class.
00214         static bool Release(const _Ty* p) {
00215                 if (p)p->Release();
00216                 return false;
00217         }
00218         // Swaps the refCount with another object
00219         static void Swap(RefCountedPolicy&) {/* */}
00220 };

00240 /****************************************************************************/
00241 // UsageCountedPolicy
00242 //
00243 // This class defines a non-intrusive usage counted policy class which
00244 // is used by the smart pointer wrapper class (CRefPtr).
00245 // This is a required parameter to the CRefPtr template.
00246 //
00247 // The interface and concept was (to my knowledge) introduced by 
00248 // Andrei Alexandrescu and presented in a C++ Report article and then later
00249 // expanded in his book "Modern C++ Design".
00250 //
00251 /****************************************************************************/
00252 template <class _Ty>
00253 class UsageCountedPolicy
00254 {
00255 // Class data
00256 private:
00257         long* _pRefCount;
00258 
00259 // Constructor
00260 public:
00261         UsageCountedPolicy() { _pRefCount = JTI_NEW long(1); }
00262         UsageCountedPolicy(const UsageCountedPolicy& rhs) : _pRefCount(rhs._pRefCount) {/* */}
00263 
00264 // Access methods
00265 public:
00266         // This method is used to copy the object
00267         _Ty* Clone(const _Ty* p) {
00268                 ::InterlockedIncrement(_pRefCount);
00269                 return const_cast<_Ty*>(p);
00270         }
00271         // Releases the object; return true if the object should
00272         // be deleted by the wrapper class.
00273         bool Release(const _Ty*) { 
00274                 if (!::InterlockedDecrement(_pRefCount)) {
00275                         delete _pRefCount;
00276                         return true;
00277                 }
00278                 return false;
00279         }
00280         // Swaps the refCount with another object
00281         void Swap(UsageCountedPolicy& rhs) { std::swap(_pRefCount, rhs._pRefCount); }
00282 };

00284 /****************************************************************************/
00285 // DefaultStoragePolicy
00286 //
00287 // The storage policy is used to manage the raw pointer type which is held
00288 // by the CRefPtr class.  This is a required parameter to the CRefPtr template.
00289 //
00290 // The interface and concept was (to my knowledge) introduced by 
00291 // Andrei Alexandrescu and presented in a C++ Report article and then later
00292 // expanded in his book "Modern C++ Design".
00293 //
00294 /****************************************************************************/
00295 template <class T>
00296 class DefaultStoragePolicy
00297 {
00298 // Typedefs
00299 protected:
00300         typedef T* _PtrType;            // The type returned by operator->
00301         typedef T& _RefType;            // The type returned by operator*
00302         typedef T* _StoredType;         // The raw pointer type
00303 // Class data
00304 private:
00305         _StoredType _p;
00306 
00307 // Constructor
00308 public:
00309         DefaultStoragePolicy() : _p(0) {/* */}
00310         DefaultStoragePolicy(const _StoredType& p) : _p(p) {/* */}
00311 
00312 // Operators
00313 public:
00314         _PtrType operator->() const { return _p; }
00315         _RefType operator*() const { return *_p; }
00316 
00317 // Access methods
00318 protected:
00319         _StoredType ptr() const { return _p; }
00320         _StoredType& ref() const { return _p; }
00321         bool IsValid() const { return (_p != 0); }
00322         void Destroy() { delete _p; _p = 0; }
00323         void Swap(DefaultStoragePolicy& lp) { std::swap(_p, lp._p);     }
00324 };

00368 /****************************************************************************/
00369 // CRefPtr
00370 //
00371 // This smart pointer class uses the supplied policy classes for determining
00372 // how to reference count the object and how to store the pointer to the object
00373 // and then implements a full wrapper class which allows the object to be
00374 // passed around as a handle.
00375 //
00376 // This class doesn't perform the creation of the underlying type.  It requires
00377 // that the user know the underlying type and how to create it.  It is, therefore
00378 // only usable as a lifetime management class and not a true handle class.
00379 //
00380 // This concept was (to my knowledge) introduced by Andrei Alexandrescu and 
00381 // presented in a C++ Report article and then later expanded in his book 
00382 // "Modern C++ Design".
00383 //
00384 // The names are changed because this class replaced an existing class which
00385 // did essentially the same thing but was not as extensible as the model presented
00386 // by Alexandrescu.
00387 //
00388 /****************************************************************************/
00389 template<class T, 
00390                  template <class> class _UsagePolicy = RefCountedPolicy,
00391                  template <class> class _StoragePolicy = DefaultStoragePolicy>
00392 
00393 class CRefPtr : 
00394         public _UsagePolicy<T>,
00395         public _StoragePolicy<T>
00396 {
00397 // Class data
00398 private:
00399         typename typedef _UsagePolicy<T> UP;
00400         typename typedef _StoragePolicy<T> SP;
00401         typename typedef SP::_PtrType _PtrType;
00402         typename typedef SP::_RefType _RefType;
00403         typename typedef SP::_StoredType _StoredType;
00404         typename typedef CRefPtr<T,_UsagePolicy,_StoragePolicy> _PtrInst;
00405 
00406 // Constructors
00407 public:
00408         CRefPtr(int null=0) { if (null != 0) throw std::runtime_error(std::string("Not a valid pointer")); }
00409         CRefPtr(_StoredType lp) : SP(lp) {/* */}
00410         CRefPtr(_StoredType lp, bool bumpRefCount) : SP((bumpRefCount) ? UP::Clone(lp) : lp) {/* */}
00411         CRefPtr(const _PtrInst& lp) : SP(UP::Clone(lp.ptr())), UP(lp) {/* */}
00412         ~CRefPtr() { 
00413                 if (SP::IsValid() && UP::Release(SP::ptr()))
00414                         SP::Destroy();
00415         }
00416 
00417 // Operators
00418 public:
00419         // Assignment from a direct pointer
00420         _PtrInst& operator=(_StoredType rhs) { 
00421                 CRefPtr temp(rhs);
00422                 UP::Swap(temp);
00423                 SP::Swap(temp);
00424                 return *this;
00425         }
00426 
00427         // Assignment from another CRefPtr instance
00428         _PtrInst& operator=(const _PtrInst& rhs) {
00429                 CRefPtr temp(rhs);
00430                 UP::Swap(temp);
00431                 SP::Swap(temp);
00432                 return *this;
00433         }
00434 
00435         // De-referencing
00436         _RefType operator*() { 
00437                 if (!SP::IsValid()) throw std::runtime_error(std::string("De-referenced null pointer"));
00438                 return SP::operator*(); 
00439         }
00440         _RefType operator*() const { 
00441                 if (!SP::IsValid()) throw std::runtime_error(std::string("De-referenced null pointer"));
00442                 return SP::operator*(); 
00443         }
00444 
00445         // Auto conversion; allows this to be used directly as a type; this is required if the
00446         // smart-pointer is to be used with STL functor wrappers (i.e. std::mem_fun).  Unfortunately
00447         // this operator also allows for delete() to be called against the underlying pointer.
00448         // I cannot find any way to stop this behavior while still allowing for direct implicit
00449         // conversions to the native type.
00450         operator _PtrType() const { 
00451                 // Do not throw an exception; this will also be called when "if (p)" syntax is used.
00452                 return SP::operator->();
00453         }
00454 
00455         // Pointer de-reference
00456         _PtrType operator->() { 
00457                 if (!SP::IsValid()) throw std::runtime_error(std::string("De-referenced null pointer"));
00458                 return SP::operator->();
00459         }
00460         _PtrType operator->() const { 
00461                 if (!SP::IsValid()) throw std::runtime_error(std::string("De-referenced null pointer"));
00462                 return SP::operator->();
00463         }
00464 
00465         // Allow tests for null "if (!p)"
00466         bool operator!() const { return (!SP::IsValid()); }
00467 
00468         // Allow for direct tests with pointers
00469         inline friend bool operator==(const CRefPtr& lhs, const _PtrType rhs) { return lhs.ptr() == rhs; }
00470         inline friend bool operator==(const _PtrType lhs, const CRefPtr& rhs) { return rhs.ptr() == lhs; }
00471         inline friend bool operator!=(const CRefPtr& lhs, const _PtrType rhs) { return lhs.ptr() != rhs; }
00472         inline friend bool operator!=(const _PtrType lhs, const CRefPtr& rhs) { return rhs.ptr() != lhs; }
00473 
00474         // Allow for tests against derived pointer types.
00475         template <class U>
00476         inline friend bool operator==(const CRefPtr& lhs, const U* rhs) { return lhs.ptr() == rhs; }
00477         template <class U>
00478         inline friend bool operator==(const U* lhs, const CRefPtr& rhs) { return rhs.ptr() == lhs; }
00479         template <class U>
00480         inline friend bool operator!=(const CRefPtr& lhs, const U* rhs) { return lhs.ptr() != rhs; }
00481         template <class U>
00482         inline friend bool operator!=(const U* lhs, const CRefPtr& rhs) { return rhs.ptr() != lhs; }
00483 
00484         // Finally, an ambiguity resolver when the pointer types are not compatible but are both
00485         // wrapped by a CRefPtr<> instance.  In this case, both operator== operators could be used.
00486         template <class U>
00487         bool operator==(const CRefPtr<U>& rhs) const { return ptr() == rhs.ptr(); }
00488         template <class U>
00489         bool operator!=(const CRefPtr<U>& rhs) const { return ptr() != rhs.ptr(); }
00490 
00491         // Allow sorting within the STL containers; sort by pointer value (i.e. behave identically to a raw pointer).
00492         inline friend bool operator<(const _PtrInst& lhs, const _StoredType rhs) { return lhs.ptr() < rhs; }
00493         inline friend bool operator<(const _StoredType lhs, const _PtrInst& rhs) { return rhs.ptr() < lhs; }
00494 
00495 // Accessor Methods
00496 public:
00497         _StoredType get() const { return SP::ptr(); }
00498         bool IsValid() const { return SP::IsValid(); }
00499 };
