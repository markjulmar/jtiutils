00066 // EventSynch
00067 //
00068 // Win32 Event object class wrapper
00069 //
00070 /****************************************************************************/
00071 class EventSynch
00072 {
00073 // Class data
00074 private:
00075         HANDLE hEvent_;
00076         bool wasCreatedByMe_;
00077 
00078 // Constructor
00079 public:
00080         EventSynch(bool bInitiallyOwn = false, bool bManualReset = false, LPCTSTR lpszName = NULL, LPSECURITY_ATTRIBUTES lpsaAttribute = NULL) : 
00081                 hEvent_(NULL), wasCreatedByMe_(false)
00082         {
00083                 hEvent_ = ::CreateEvent(lpsaAttribute, bManualReset ? TRUE : FALSE, bInitiallyOwn ? TRUE : FALSE, lpszName);
00084                 wasCreatedByMe_ = (GetLastError()!=ERROR_ALREADY_EXISTS);
00085         }
00086 
00087         EventSynch(const EventSynch& rhs) : hEvent_(0), wasCreatedByMe_(rhs.wasCreatedByMe_) {
00088                 if (rhs.hEvent_)
00089                         ::DuplicateHandle(GetCurrentProcess(), rhs.hEvent_, GetCurrentProcess(), &hEvent_, 0, FALSE, DUPLICATE_SAME_ACCESS); //lint !e534
00090         }
00091         EventSynch& operator=(const EventSynch& rhs) {
00092                 if (this != &rhs)
00093                 {
00094                         wasCreatedByMe_ = rhs.wasCreatedByMe_;
00095                         if (rhs.hEvent_)
00096                                 ::DuplicateHandle(GetCurrentProcess(), rhs.hEvent_, GetCurrentProcess(), &hEvent_, 0, FALSE, DUPLICATE_SAME_ACCESS);  //lint !e534
00097                         else
00098                                 hEvent_ = NULL;
00099                 }
00100                 return *this;
00101         }
00102         
00103         ~EventSynch() 
00104         {       
00105                 if (hEvent_)
00106                 {
00107                         ::CloseHandle(hEvent_);  //lint !e534
00108                         hEvent_ = NULL; 
00109                 }
00110         }
00111 
00112 // Operators
00113 public:
00114         operator HANDLE () const { return get(); }
00115 
00116 // Properties
00117 public:
00118         __declspec(property(get=get_WasCreated)) bool Created;
00119         __declspec(property(get=get_IsValid)) bool IsValid;
00120 
00121 // Operations
00122 public:
00123         bool get_IsValid() const throw() { return (hEvent_!=NULL); }
00124         bool get_WasCreated() const throw() { return wasCreatedByMe_; }
00125 
00126 // Operations
00127 public:
00128         bool SetEvent() throw() { return (::SetEvent(hEvent_) == TRUE); }
00129         bool PulseEvent() throw() { return (::PulseEvent(hEvent_) == TRUE); }
00130         bool ResetEvent() throw() { return (::ResetEvent(hEvent_) == TRUE); }
00131         DWORD Wait(DWORD dwMsecs = INFINITE) const throw() {    return ::WaitForSingleObject(hEvent_,dwMsecs); }
00132 
00133 // Operators
00134 public:
00135         HANDLE get() const throw() { return hEvent_; }
00136 };

00171                         else
00172                                 hSemaphore_ = NULL;
00173                 }
00174                 return *this;
00175         }
00176         
00177         ~SemaphoreSynch()

00179                 if (hSemaphore_)

00184         }
00185

00189 
00190 // Properties

00193         __declspec(property(get=get_IsValid)) bool IsValid;
00194 
00195 // Operations
00196 public:
00197         bool get_IsValid() const throw() { return (hSemaphore_ != NULL); }
00198         bool get_WasCreated() const throw() { return wasCreatedByMe_; }
00199         DWORD Lock(DWORD dwMsecs = INFINITE) throw() {  return WaitForSingleObject(hSemaphore_,dwMsecs); }
00200         long Unlock(long Count=1) throw() { 
00201                 if (Count <= 0) Count = 1;
00202                 long OldCount = 0;
00203                 if (::ReleaseSemaphore(hSemaphore_, Count, &OldCount) == FALSE)
00204                         OldCount = 0;
00205                 return OldCount;
00206         }

00208 // Operators

00210         HANDLE get() const throw() { return hSemaphore_; }
