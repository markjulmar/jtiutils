00143         class WorkThreadPool : public IOCPThreadPool
00144         {
00145         private:
00146                 mutable volatile long running_;
00147                 mutable volatile long inQueue_;
00148                 mutable volatile long inWork_;
00149                 long minThreads_;
00150 
00151                 virtual void WorkerThreadStart() {_TNotify::WorkerThreadPool_StartThread();}
00152                 virtual void WorkerThreadEnd() {_TNotify::WorkerThreadPool_EndThread();}
00153                 virtual bool ProcessWork(LPOVERLAPPED pio, DWORD dwBytesTransferred, 
00154                         ULONG_PTR CompletionKey, BOOL rc, DWORD dwLastError)
00155                 {
00156                         UNREFERENCED_PARAMETER(pio); UNREFERENCED_PARAMETER(dwBytesTransferred);
00157 
00158                         // Timeout? Allow the thread to exit.
00159                         if (rc == FALSE && dwLastError == WAIT_TIMEOUT)
00160                                 return (NumThreads > minThreads_);
00161 
00162                         // If we have a wait event, signal it.
00163                         if (pio && pio->hEvent)
00164                                 SetEvent(pio->hEvent);
00165 
00166                         // Check for our work packet
00167                         if (CompletionKey != 0)
00168                         {
00169                                 InterlockedDecrement(&inQueue_);
00170                                 IncDecHolder incItem(&inWork_);
00171                                 WorkItemDelegate* pDelegate = reinterpret_cast<WorkItemDelegate*>(CompletionKey);
00172                                 std::auto_ptr<WorkItemDelegate> acItem(pDelegate);
00173 
00174                                 // Call the function; we will automatically cleanup if an exception occurs
00175                                 // because of the above class holders.
00176                                 if (pDelegate->hasArgs)
00177                                         (*pDelegate->ThisDelegate.pfun1)(pDelegate->arg);
00178                                 else
00179                                         (*pDelegate->ThisDelegate.pfun)();
00180                         }
00181                         return false;
00182                 }
00183 
00184                 public:
00185                         WorkThreadPool() : IOCPThreadPool(), 
00186                                 inQueue_(0), inWork_(0), minThreads_(0), running_(0) { ThreadTimeout = _TTimers::THREAD_TIMEOUT; }
00187                         virtual ~WorkThreadPool() {/* */}
00188 
00189                         bool Start(int nMinThreads, int nSimulThreads)
00190                         {
00191                                 minThreads_ = nMinThreads;
00192                                 return IOCPThreadPool::Start(nSimulThreads, nMinThreads);
00193                         }
00194 
00195                         bool PostQueuedCompletionStatus(_Arg pItem, OVERLAPPED* pio) {
00196                                 InterlockedIncrement(&inQueue_);
00197                                 return IOCPThreadPool::PostQueuedCompletionStatus((TP_COMPLETION_KEY)pItem, 0, pio);
00198                         }
00199 
00200                         long get_Queued() const { return InterlockedCompareExchange(&inQueue_, 0, 0); }
00201                         long get_InWork() const { return InterlockedCompareExchange(&inWork_, 0, 0); }
00202         };

00210                 OVERLAPPED oio;
00211                 int maxThreads_;
00212         public:
00213                 DispatchThreadPool(WorkThreadPool& workPool, EventSynch& evtStop) : IOCPThreadPool(), 
00214                         maxThreads_(0), workPool_(workPool), evtWait_(false, true), evtStop_(evtStop) { oio.hEvent = evtWait_.get(); }
00215                 ~DispatchThreadPool() {/* */}
00216

00221                 }
00222 
00223                 virtual bool ProcessWork(LPOVERLAPPED pio, DWORD dwBytesTransferred, 
00224                         ULONG_PTR CompletionKey, BOOL rc, DWORD dwLastError)
00225                 {
00226                         UNREFERENCED_PARAMETER(pio); UNREFERENCED_PARAMETER(dwBytesTransferred);
00227                         UNREFERENCED_PARAMETER(rc); UNREFERENCED_PARAMETER(dwLastError);
00228 
00229                         // Reset our wait event
00230                         evtWait_.ResetEvent();
00231 
00232                         // Move the request to the real thread pool; it may
00233                         // get queued up on the worker pool, but that's ok.
00234                         if (CompletionKey && workPool_.PostQueuedCompletionStatus((_Arg)CompletionKey, &oio))
00235                         {
00236                                 // Check to see how busy the pool is.  We may need to create
00237                                 // some additional worker threads if we cannot start this request
00238                                 // right away.
00239                                 if (evtWait_.Wait(_TTimers::MAX_WAIT_THRESHOLD) == WAIT_TIMEOUT)
00240                                 {
00241                                         long numQueued = workPool_.get_Queued();
00242                                         long numThreads = workPool_.NumThreads;
00243                                         long numActive = workPool_.get_InWork();
00244                                         long numNeeded = min(numActive + numQueued, maxThreads_);
00245                                         if (numThreads < numNeeded) {
00246                                                 numThreads = min(numNeeded+_TTimers::THREAD_INCREMENT, maxThreads_);
00247                                                 workPool_.set_NumThreads(numThreads);
00248                                         }
00249                                 }
00250                         }
00251 
00252                         // Never ask the thread to terminate
00253                         return false;
00254                 }
00255         };

00408         static unsigned __stdcall RegisterWaitCallback(void* p)
00409         {
00410                 WorkItemDelegateWaiter* pItem = reinterpret_cast<WorkItemDelegateWaiter*>(p);
00411                 HANDLE arrHandles[3] = { pItem->hStop, pItem->hKill, pItem->hWait };
00412                 int nCount = (pItem->hWait == NULL) ? 2 : 3;
00413 
00414                 _TNotify::WorkerThreadPool_StartThread();
00415 
00416                 for (;;)
00417                 {
00418                         DWORD rc = WaitForMultipleObjects(nCount, arrHandles, FALSE, pItem->dwTimeout);
00419                         if (rc == WAIT_FAILED ||
00420                                 rc == WAIT_OBJECT_0 || rc == WAIT_ABANDONED_0 ||
00421                                 rc == WAIT_OBJECT_0 + 1 || rc == WAIT_ABANDONED_0 + 1)
00422                                 break;
00423 
00424                         if (pItem->hasArgs)
00425                                 (*pItem->ThisDelegate.pfun1)(pItem->arg, (rc == WAIT_TIMEOUT));
00426                         else
00427                                 (*pItem->ThisDelegate.pfun)((rc==WAIT_TIMEOUT));
00428 
00429                         if (pItem->executeOnce)
00430                                 break;
00431                 }
00432                 delete pItem;
00433 
00434                 _TNotify::WorkerThreadPool_EndThread();
00435 
00436                 return 0;
00437         }
