00243 /****************************************************************************/
00244 // ServiceInstance
00245 //
00246 // This class models the typed service instance which is derived from
00247 // in order to create a concrete class.
00248 //
00249 /****************************************************************************/
00250 //lint --e{1704} Private constructor ok
00251 template <class _Ty, class LockType = JTI_SERVICE_LOCK_TYPE>
00252 class ServiceInstance : 
00253         public ServiceBase,
00254         public LockableObject<LockType>
00255 {
00256 // Constructor
00257 public:
00258         ServiceInstance(const char* pszServiceName, const char* pszDisplayName, DWORD dwStartTime=0) :
00259                   ServiceBase(pszServiceName, pszDisplayName, dwStartTime) {JTI_ASSERT(g_pThis == NULL); g_pThis = (_Ty*)this;}
00260         ServiceInstance(const wchar_t* pszServiceName, const wchar_t* pszDisplayName, DWORD dwStartTime=0) : 
00261                   ServiceBase(pszServiceName, pszDisplayName, dwStartTime) {JTI_ASSERT(g_pThis == NULL); g_pThis = (_Ty*)this;}
00262         virtual ~ServiceInstance() {g_pThis = NULL;}
00263 
00264 // Singleton accessor
00265 public:
00266         static _Ty& Instance() { 
00267                 JTI_ASSERT(g_pThis != NULL);
00268                 return *g_pThis; 
00269         }
00270 
00271 // Static methods for callback
00272 protected:
00273         static void WINAPI _Handler(DWORD dwOpcode) 
00274         {
00275                 _Ty& This = ServiceInstance<_Ty,LockType>::Instance();
00276                 switch (dwOpcode)
00277                 {
00278                         case SERVICE_CONTROL_STOP:      This.Stop(); break;
00279                         case SERVICE_CONTROL_PAUSE:     This.Pause(); break;
00280                         case SERVICE_CONTROL_CONTINUE: This.Continue(); break;
00281                         case SERVICE_CONTROL_INTERROGATE: This.Interrogate(); break;
00282                         case SERVICE_CONTROL_SHUTDOWN: This.Shutdown(); break;
00283                         default: break;
00284                 }
00285         }
00286         static BOOL WINAPI _ConsoleHandler(DWORD dwCtrlType) 
00287         {
00288                 // If the user shuts down, logs off, closes the window or hits CTRL+C then exit.
00289                 if (dwCtrlType == CTRL_SHUTDOWN_EVENT || dwCtrlType == CTRL_LOGOFF_EVENT || dwCtrlType == CTRL_CLOSE_EVENT ||
00290                         dwCtrlType == CTRL_BREAK_EVENT || dwCtrlType == CTRL_C_EVENT) {
00291                         ServiceInstance<_Ty,LockType>::Instance().Stop();
00292                         return TRUE;
00293                 }
00294                 return FALSE;
00295         }
00296         static void WINAPI _ServiceMain(DWORD dwArgc, LPTSTR* lpszArgv) 
00297         { 
00298                 ServiceInstance<_Ty,LockType>::Instance().ServiceMain(dwArgc, lpszArgv); 
00299         }
00300 
00301 // Required overrides
00302 protected:
00303         virtual SERVICE_TABLE_ENTRY* GetServiceTable() const
00304         {
00305         static SERVICE_TABLE_ENTRY st[] = { 
00306                         { const_cast<LPTSTR>(ServiceName_.c_str()), _ServiceMain }, 
00307                         { NULL, NULL } 
00308                 };
00309                 return st;
00310         }
00311 
00312         virtual LPHANDLER_FUNCTION GetFunctionHandler() const
00313         {
00314                 return _Handler; 
00315         }
00316                 
00317         virtual void SetupConsoleHandler()
00318         {
00319                 SetConsoleCtrlHandler(_ConsoleHandler, TRUE);
00320         }
00321 
00322 // Locked up methods
00323 private:
00324         ServiceInstance(const ServiceInstance&);
00325         ServiceInstance& operator=(const ServiceInstance&);
00326 // Class data
00327 private:
00328         static _Ty* g_pThis;
00329 };
