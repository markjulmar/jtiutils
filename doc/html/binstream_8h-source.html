00049 // binstream
00050 //
00051 // Base class for the binary input stream used to retrieve persistent object
00052 // data from some location. This location could be the registry, a file, an OLE
00053 // object, etc.
00054 //
00055 // INTERNAL DATA STRUCTURE
00056 //
00057 /******************************************************************************/
00058 class binstream
00059 {
00060 // Class data
00061 public:
00062         enum { 
00063                 eofbit  = 0x1,
00064                 failbit = 0x2
00065         };
00066 private:
00067         unsigned char bitFlags_;
00068 
00069 // Destructor
00070 public:
00071         binstream() : bitFlags_(0) {/* */}
00072         virtual ~binstream() {/* */}
00073 
00074 // Overridable operations required in the derived classes
00075 public:
00076         virtual bool open() { return true;}
00077         virtual void close() {/* */}
00078         virtual bool skip(int) { return false; }
00079         virtual bool peek(void*, unsigned int) const = 0;
00080         virtual bool read(void*, unsigned int) = 0;
00081         virtual bool write(const void*, unsigned int) = 0;
00082 
00083 // Other methods
00084 public:
00085         binstream& ignore(int nCount=1) { skip(nCount); return *this; }
00086         bool good() const { return (bitFlags_ & failbit) == 0; }
00087         bool eof() const { return (bitFlags_ & (eofbit|failbit)) != 0; }
00088         bool fail() const { return !good(); }
00089         
00090         unsigned char peek() const { char ch = 0; peek(&ch,1); return ch; }
00091 
00092         operator void*() { return (!fail()) ? (void*)1 : (void*)0; }
00093         bool operator !() { return fail(); }
00094 
00095 // Internal methods
00096 protected:
00097         void setbit(int bit) { bitFlags_ |= (bit&0xff); }
00098         void clrbit(int bit=0xff) { bitFlags_ &= ~(bit&0xff); }
00099 };

00354 /******************************************************************************/
00355 // binstream_input_iterator
00356 //
00357 // Iterator class for the binary stream type.  This allows the stream to be
00358 // used in STL algorithms.
00359 //
00360 /******************************************************************************/
00361 template <typename _Ty = unsigned char>
00362 class binstream_input_iterator : std::iterator<std::input_iterator_tag, _Ty>
00363 {
00364 // Class data
00365 public:
00366         typedef binstream_input_iterator<_Ty> _Myt;
00367 private:
00368         binstream* m_stm;
00369 
00370 // Constructor
00371 public:
00372         binstream_input_iterator() : m_stm(0) {/* */}
00373         binstream_input_iterator(binstream& stm) : m_stm(&stm) { if (m_stm && m_stm->eof()) m_stm = 0; }
00374 
00375         binstream_input_iterator(const binstream_input_iterator& b) : m_stm(b.m_stm) {/* */}
00376         _Myt& operator=(const _Myt& b) {
00377                 if (this != &b)
00378                         m_stm = b.m_stm;
00379                 return *this;
00380         }
00381 
00382 // Accessors
00383 public:
00384         // Dereferencing returns the value
00385         _Ty operator*() const { return const_cast<_Myt*>(this)->GetNextVal(); }
00386 
00387         // Pre/Post increment return the stream
00388         _Myt& operator++() { return (*this); }
00389         _Myt operator++(int) { return (*this); }
00390 
00391         // Equality operators
00392         inline bool operator==(_Myt& _Right) { return m_stm == _Right.m_stm; }
00393         inline bool operator!=(_Myt& _Right) { return !operator==(_Right); }
00394 
00395 // Internal methods
00396 private:
00397         _Ty GetNextVal() {
00398                 _Ty _Val = 0;
00399                 if (m_stm && !m_stm->eof())
00400                         *m_stm >> _Val;
00401                 else
00402                         m_stm = 0;
00403                 return _Val;
00404         }
00405 };

00455 // VersionInfo
00456 //
00457 // Stores and manages schema versions within a object stream source
00458 //
00459 // INTERNAL DATA STRUCTURE
00460 //
00461 /******************************************************************************/
00462 class VersionInfo
00463 {
00464 // Class data
00465 private:
00466         static const DWORD VER_MARK = 0x52455600;
00467         int m_nVerRead;
00468 
00469 // Constructor
00470 public:
00471         VersionInfo(binstream& ar) : m_nVerRead(1) {
00472                 // If the stream doesn't support backups then exit.
00473                 if (ar.skip(0))
00474                 {
00475                         // Read the version marker and backup if it doesn't exist.
00476                         DWORD dwID; ar >> dwID;
00477                         if ((dwID & 0xffffff00) == VER_MARK)
00478                                 m_nVerRead = (int) (dwID & 0x000000ff);
00479                         else
00480                                 ar.skip(-1 * static_cast<int>(sizeof(DWORD)));
00481                 }
00482         }
00483 
00484         VersionInfo(binstream& ar, int nVer) : m_nVerRead(1) {
00485                 // If the stream doesn't support backups then exit.
00486                 if (ar.skip(0))
00487                 {
00488                         // Otherwise, store the version marker
00489                         DWORD dwID = (VER_MARK | (DWORD)nVer);  
00490                         ar << dwID;
00491                 }
00492         }
00493 
00494 // Properties
00495 public:
00496         __declspec(property(get=get_Version)) int Version;
00497 
00498 // Accessors
00499 public:
00500         int get_Version() const { return m_nVerRead; }
00501 };
