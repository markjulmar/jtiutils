00121 /******************************************************************************/
00122 // inauto_ptr
00123 //
00124 // auto_ptr class for intrinsic data types (i.e. non UDT)
00125 //
00126 // Usage:
00127 //
00128 // stdx::inauto_ptr<char> = new char[256];
00129 //
00130 /******************************************************************************/
00131 template<class _Ty>
00132 class inauto_ptr 
00133 {
00134 public:
00135         typedef _Ty element_type;
00136         explicit inauto_ptr(_Ty *_P = 0) throw() : m_fOwner(_P != 0), m_pData(_P) {}
00137         inauto_ptr(const inauto_ptr<_Ty>& src) throw() : m_fOwner(src.m_fOwner), m_pData(src.release()) {}
00138         inauto_ptr<_Ty>& operator=(const inauto_ptr<_Ty>& src) throw()
00139         {
00140                 if (this != &src) {
00141                         if (m_pData != src.get()) {
00142                                 if (m_fOwner) 
00143                                         delete [] m_pData;
00144                                 m_fOwner = src.m_fOwner; 
00145                         }
00146                         else if (src.m_fOwner)
00147                                 m_fOwner = true, src.release(); //lint !e534
00148                         m_pData = src.release(); 
00149                 }
00150                 return (*this); 
00151         }
00152 
00153         //lint --e{1540}
00154         ~inauto_ptr() throw()
00155         {
00156                 if (m_fOwner) {
00157                         delete [] m_pData; 
00158                         m_pData = 0;
00159                 }
00160         }
00161 
00162         _Ty *get() const throw() { return m_pData; }    //lint !e1763
00163         _Ty& operator*() const throw() { return (*get()); }
00164         _Ty *release() const throw()
00165         {
00166                 const_cast<inauto_ptr<_Ty>*>(this)->m_fOwner = false;
00167                 return (m_pData); 
00168         }
00169         void reset(_Ty* pData = 0)
00170         {
00171                 if (pData != m_pData)
00172                         delete [] m_pData;
00173                 m_fOwner = (pData != static_cast<_Ty*>(NULL));
00174                 m_pData = pData; 
00175         }
00176 
00177 // Class data
00178 private:
00179         bool m_fOwner;
00180         _Ty *m_pData;
00181 };

00210 /******************************************************************************/
00211 // compose_f_gx
00212 //
00213 // Composition adapter f(g(elem))
00214 //
00215 // This is the general form unary compose function.  It allows nested calls of
00216 // unary predicates such that the result of calling predicate g() for elem is
00217 // used as the input for predicate f().  The whole expression then operates as
00218 // a unary predicate.
00219 // 
00220 /******************************************************************************/
00221 template <class OP1, class OP2>
00222 class compose_f_gx_t : public std::unary_function<typename OP2::argument_type, typename OP1::result_type>
00223 {
00224 private:
00225         OP1 op1; OP2 op2;
00226 public:
00227         compose_f_gx_t(const OP1& o1, const OP2& o2) : op1(o1), op2(o2) {}
00228         typename OP1::result_type 
00229                 operator()(const typename OP2::argument_type& x) const {
00230                         return op1(op2(x));
00231                 }
00232 };
00233 
00234 template <class OP1, class OP2>
00235 inline compose_f_gx_t<OP1,OP2>
00236 compose_f_gx(const OP1& o1, const OP2& o2) {
00237         return compose_f_gx_t<OP1,OP2>(o1,o2);
00238 }

00240 /******************************************************************************/
00241 // compose_f_gx_hx
00242 //
00243 // Composition adapter f(g(elem), h(elem))
00244 //
00245 // This is compose function where the elem is passed to two different unary
00246 // predicates g() and h() and the result of both is then processed by the binary
00247 // predicate f().  In a way, this "injects" a single argument into a composed
00248 // function.
00249 // 
00250 /******************************************************************************/
00251 template <class OP1, class OP2, class OP3>
00252 class compose_f_gx_hx_t : public std::unary_function<typename OP2::argument_type, typename OP1::result_type>
00253 {
00254 private:
00255         OP1 op1; OP2 op2; OP3 op3;
00256 public:
00257         compose_f_gx_hx_t (const OP1& o1, const OP2& o2, const OP3& o3) : op1(o1), op2(o2), op3(o3) {}
00258         typename OP1::result_type
00259                 operator()(const typename OP2::argument_type& x) const {
00260                         return op1(op2(x),op3(x));
00261                 }
00262 };
00263 
00264 template <class OP1, class OP2, class OP3>
00265 inline compose_f_gx_hx_t<OP1,OP2,OP3>
00266 compose_f_gx_hx(const OP1& o1, const OP2& o2, const OP3& o3) {
00267         return compose_f_gx_hx_t<OP1,OP2,OP3>(o1,o2,o3);

00325 //
00326 // Splits a string into multiple pieces based on a separator
00327 //
00328 /******************************************************************************/
00329 template<class OutIt>
00330 void split(const std::string& s, const std::string& sep, OutIt dest) {
00331         std::string::size_type left = s.find_first_not_of(sep);
00332         std::string::size_type right = s.find_first_of(sep, left);
00333         while (left < right) {
00334                 *dest = s.substr(left, right-left);
00335                 dest++;
00336                 left = s.find_first_not_of(sep, right);
00337                 right = s.find_first_of(sep, left);
00338         }
00339 }

00353 /******************************************************************************/
00354 // split_preserving_quotes
00355 //
00356 // Splits a string into multiple pieces based on a separator; preserving quotes
00357 // as a single literal
00358 //
00359 /******************************************************************************/
00360 template<class OutIt>
00361 void split_preserving_quotes(const std::string& s, const std::string& sep, OutIt dest) {
00362         const char quoteChar = '\"';
00363         if (s.find_first_of(quoteChar) == std::string::npos) {
00364                 split(s,sep,dest);
00365         }
00366         else {
00367                 std::string::size_type left = s.find_first_not_of(sep);
00368                 std::string::size_type quote = s.find_first_of(quoteChar);
00369                 std::string::size_type right = s.find_first_of(sep, left);
00370 
00371                 while (left < right) {
00372                         if (quote < right) // find the end of the quote or EOS.
00373                                 right = s.find_first_of(sep, s.find_first_of(quoteChar, quote+1));
00374                         *dest++ = s.substr(left, right-left);
00375                         left = s.find_first_not_of(sep, right);
00376                         quote = s.find_first_of(quoteChar,left);
00377                         right = s.find_first_of(sep, left);
00378                 }
00379         }
00380 }

00436 //
00437 // Quick functions to strip starting/trailing blanks. Supply both unicode/ansi versions.
00438 //
00439 /******************************************************************************/
00440 #pragma warning(disable : 4996)
00441 inline wchar_t* trim(wchar_t* psz)
00442 {
00443         if (!psz || !*psz) return psz;
00444         wchar_t* pc = psz, *pe = (psz+wcslen(psz));
00445         while (*pc == L' ') ++pc;
00446         if (pc != psz) memcpy(psz,pc,static_cast<unsigned int>((pe+1-pc)*sizeof(wchar_t)));
00447         pe = (psz+wcslen(psz)-1);
00448         while (*pe == L' ') --pe; *(pe+1) = 0;
00449         return psz;
00450 }

00452 {
00453         if (!psz || !*psz) return psz;
00454         char* pc = psz, *pe = (psz+strlen(psz));
00455         while (*pc == ' ') ++pc;
00456         if (pc != psz) memcpy(psz,pc,static_cast<unsigned int>(pe+1-pc));
00457         pe = (psz+strlen(psz)-1); 
00458         while (*pe == ' ') --pe; *(pe+1) = 0;
00459         return psz;

00501 //
00502 /******************************************************************************/

00504 inline void string_replace(std::basic_string<CH>& s, const std::basic_string<CH>& sFind, const std::basic_string<CH>& sReplace)
00505 {
00506     std::basic_string<CH>::size_type pos = 0, szf = sFind.size(), szr = sReplace.size();;
00507     while((pos = s.find(sFind, pos)) != std::basic_string<CH>::npos) {
00508         s.replace(pos, szf, sReplace);
00509                 pos += szr;
00510         }
