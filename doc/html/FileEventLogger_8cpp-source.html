00164 /*****************************************************************************
00165 ** Procedure:  FileEventLogger::CleanQueue
00166 ** 
00167 ** Arguments:  void
00168 ** 
00169 ** Returns: void
00170 ** 
00171 ** Description: This removes all elements from the queue and writes them
00172 **              to our log.
00173 **
00174 /****************************************************************************/
00175 bool FileEventLogger::CleanQueue()
00176 {
00177         // Reopen our file if necessary
00178         CheckFile();
00179 
00180         // Write to the file if it's open
00181         CCSLock<FileEventLogger> lockGuard(this);
00182         if (logFile_ == INVALID_HANDLE_VALUE) {
00183                 qData_.clear();
00184                 return false;
00185         }
00186 
00187     // Copy off the current data into a private queue and unlock.
00188     std::deque<std::string> queue;
00189         queue.swap(qData_);
00190         lockGuard.Unlock();
00191 
00192     // Get the file size of the current logfile.  If the hi file size is set then
00193     // assume max size reached.
00194     DWORD dwFileSize = 0;
00195     if (maxSize_ > 0)
00196     {
00197         DWORD dwHiSize;
00198         dwFileSize = GetFileSize(logFile_, &dwHiSize); 
00199         if (dwHiSize > 0)
00200             dwFileSize = std::numeric_limits<DWORD>::max();
00201 
00202         // Reopen the file immediately if necessary.
00203                 if (dwFileSize >= maxSize_)
00204                 {
00205                         CheckFile();
00206                         if (logFile_ == INVALID_HANDLE_VALUE)
00207                                 return false;
00208             dwFileSize = 0;
00209                 }
00210     }
00211 
00212     // Write out our entries.
00213         while (!queue.empty())
00214         {
00215                 const std::string& s = queue.front();
00216                 DWORD dwLength = static_cast<DWORD>(s.length()), dwWritten = 0;
00217                 JTI_VERIFY(WriteFile(logFile_, s.c_str(), dwLength, &dwWritten, NULL) != 0);
00218                 JTI_ASSERT(dwWritten == dwLength);
00219 
00220                 if (maxSize_ > 0)
00221         {
00222             if ((dwFileSize+dwWritten) < dwFileSize || (dwFileSize+dwWritten) >= maxSize_)
00223                     {
00224                         CheckFile();
00225                         if (logFile_ == INVALID_HANDLE_VALUE)
00226                                 return false;
00227                 dwFileSize = 0;
00228                     }
00229             else
00230                     dwFileSize += dwWritten;
00231         }
00232                 queue.pop_front();
00233         }
00234         return true;
00235 
00236 }// FileEventLogger::CleanQueue

00238 /*****************************************************************************
00239 ** Procedure:  FileEventLogger::CheckFile
00240 ** 
00241 ** Arguments:  void
00242 ** 
00243 ** Returns: void
00244 ** 
00245 ** Description: This checks the file and reopens it if necessary.
00246 **
00247 /****************************************************************************/
00248 void FileEventLogger::CheckFile()
00249 {
00250         // Lock access to the file handle
00251         CCSLock<FileEventLogger> lockGuard(this);
00252 
00253         bool fReopen = (logFile_ == INVALID_HANDLE_VALUE);
00254                 
00255         // See if our date has changed; if so, open a new log.
00256         SYSTEMTIME stNow; ::GetLocalTime(&stNow);
00257         if (dayOfWeek_ != stNow.wDay)
00258         {
00259                 fReopen = true;
00260                 fileIndex_ = 0;
00261         }
00262         else if (HitMaxSize())
00263         {
00264                 fReopen = true;
00265         }
00266 
00267         // Need to re-open the file?
00268         if (fReopen)
00269         {
00270                 // Reset
00271                 fReopen = false;
00272 
00273                 // If we already have a log file, close it here.
00274                 if (logFile_ != INVALID_HANDLE_VALUE) {
00275                         JTI_VERIFY(::CloseHandle(logFile_) != 0);
00276                         logFile_ = INVALID_HANDLE_VALUE;
00277                 }
00278 
00279                 // Get the directory to work with
00280                 std::string strDir = dirName_;
00281                 if (!strDir.empty())
00282                 {
00283                         if (_access(strDir.c_str(),0) == -1) {
00284                                 if (_mkdir(strDir.c_str())== -1)
00285                                         strDir = "";
00286                         }
00287                 }
00288 
00289                 // Check to see if we are going to rename the file to a new name.
00290                 if (!currName_.empty() && !renName_.empty())
00291                 {
00292                         MoveFileExA(currName_.c_str(), GetNextFileName(strDir, renName_, stNow).c_str(), 
00293                                 MOVEFILE_REPLACE_EXISTING | MOVEFILE_COPY_ALLOWED);
00294                 }
00295 
00296                 // Set the new day of week
00297                 dayOfWeek_ = stNow.wDay;
00298 
00299                 // Get the next filename
00300                 currName_ = GetNextFileName(strDir, baseName_, stNow);
00301 
00302                 // Open the file.
00303                 logFile_ = ::CreateFileA(currName_.c_str(), GENERIC_WRITE, FILE_SHARE_READ, NULL, OPEN_ALWAYS, FILE_ATTRIBUTE_NORMAL, NULL);
00304                 if (logFile_ != INVALID_HANDLE_VALUE)
00305                 {
00306                         if (truncateExisting_)
00307                         {
00308                                 SetFilePointer(logFile_, 0L, NULL, FILE_BEGIN);
00309                                 SetEndOfFile(logFile_);
00310                         }
00311                         else
00312                                 SetFilePointer(logFile_, 0L, NULL, FILE_END);
00313                 }
00314         }
00315 }//

00330         std::ostringstream stm;

00340

00342         if (fileIndex_ < 0)     {
00343                 fileIndex_ = 0;
00344                 throw jtiexception(-1, "Ran out of file indexes for file logger.");

00369 /*****************************************************************************
00370 ** Procedure:  FileEventLogger::BuildFilename

00372 ** Arguments:  stNow - Current Time

00378 /****************************************************************************/
00379 //lint -e{661}  //lint --e(661) Possible access outside bounds of pointer array
00380 std::string FileEventLogger::BuildFilename(const std::string& strName, const SYSTEMTIME& stNow) const
00381 {
00382         std::ostringstream ostm;
00383 
00384         if (strName.empty())
00385         {
00386                 ostm << std::dec << std::setfill('0') << std::setw(4) << std::setiosflags(std::ios::right) << stNow.wYear
00387                          << std::dec << std::setfill('0') << std::setw(2) << std::setiosflags(std::ios::right) << stNow.wMonth
00388                          << std::dec << std::setfill('0') << std::setw(2) << std::setiosflags(std::ios::right) << stNow.wDay
00389                          << ".log";
00390         }
00391         else
00392         {
00393                 std::string::const_iterator itEnd = strName.end();
00394                 for (std::string::const_iterator it = strName.begin(); it != itEnd; ++it)
00395                 {
00396                         if ((*it) == '%')
00397                         {
00398                                 if (++it == itEnd)
00399                                         break;
00400                                 switch ((*it))
00401                                 {
00402                                         case 'd': // Day of month as decimal number (01  31) 
00403                                                 ostm << std::dec << std::setfill('0') << std::setw(2) << std::setiosflags(std::ios::right) << stNow.wDay;
00404                                                 break;
00405                                         case 'm': // Month as a decimal number
00406                                                 ostm << std::dec << std::setfill('0') << std::setw(2) << std::setiosflags(std::ios::right) << stNow.wMonth;
00407                                                 break;
00408                                         case 'y': // Year without century, as decimal number (00  99)
00409                                                 ostm << std::dec << std::setfill('0') << std::setw(2) << std::setiosflags(std::ios::right) << stNow.wYear - 2000;
00410                                                 break;
00411                                         case 'Y': // Year with century
00412                                                 ostm << std::dec << std::setfill('0') << std::setw(4) << std::setiosflags(std::ios::right) << stNow.wYear;
00413                                                 break;
00414                                         case 'c': // Counter
00415                                                 if (fileIndex_ < 999)
00416                                                         ostm << std::dec << std::setfill('0') << std::setw(3) << std::setiosflags(std::ios::right) << fileIndex_;
00417                                                 else
00418                                                         ostm << std::dec << fileIndex_;
00419                                                 break;
00420                                         case '%': // Percent sign
00421                                                 ostm << '%';
00422                                                 break;
00423                                         default:
00424                                                 ostm << (*it);
00425                                                 break;
00426                                 }
00427                         }
00428                         else
00429                                 ostm << (*it);
00430                 }
00431         }
00432         return ostm.str();
00433 
00434 }// FileEventLogger::BuildFilename
