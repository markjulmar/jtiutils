00032 /****************************************************************************/
00033 // CreateDynamicObject
00034 //
00035 // Create a new object dynamically. Search the list of dynamic
00036 // objects and call each create_object() method until one of them 
00037 // is successful in making the required object. If we get to the end of the
00038 // list, then the object wasn't registered.
00039 //
00040 // Usage:
00041 //
00042 // class A {...};
00043 // DECLARE_JTI_DYNCREATE(A)
00044 //      ...
00045 // A* pa = JTI_Util::CreateDynamicObject<A>();
00046 //
00047 /****************************************************************************/
00048 template <class T>
00049 T* CreateDynamicObject()
00050 {
00051         return reinterpret_cast<T*>(IRttiFactory::CreateDynamicObject(typeid(T).name()));
00052 
00053 }// CreateDynamicObject

00055 /****************************************************************************/
00056 // IRttiFactory
00057 //
00058 // This interface describes the abstract RTTI factory which is derived from
00059 // using a type-specific class from TFactory.
00060 //
00061 /****************************************************************************/
00062 class IRttiFactory
00063 {    
00064 // Constructor/Destructor
00065 public:
00066         IRttiFactory() 
00067         { 
00068                 nextPtr_ = headPtr_; 
00069                 headPtr_ = this; 
00070         }
00071         
00072         virtual ~IRttiFactory() 
00073         {
00074                 IRttiFactory** ppNext = &headPtr_;
00075                 for (; *ppNext != NULL; ppNext = &(*ppNext)->nextPtr_) {
00076                         if (*ppNext == this) {
00077                                 *ppNext = (*ppNext)->nextPtr_;
00078                                 break;
00079                         }
00080                 }
00081         }
00082         
00083 // Methods
00084 public:
00085         // This method walks all the registered classes and invokes the CreateObject
00086         // function of the specific type to create.
00087         static void* CreateDynamicObject(const char* pszClass) 
00088         {
00089                 void* pObject = NULL;
00090                 const IRttiFactory* currFactory = headPtr_;
00091                 for (; currFactory != NULL && pObject == NULL; currFactory = currFactory->nextPtr_)
00092                         pObject = currFactory->CreateObject(pszClass);
00093                 return pObject;
00094         }
00095 
00096 // Required overrides
00097 private:
00098         virtual void* CreateObject(const char* pszClass) const = 0;
00099 
00100 // Class data
00101 private:
00102         static IRttiFactory* headPtr_;
00103         IRttiFactory* nextPtr_;
00104 };
00105 
00106 /****************************************************************************/
00107 // TFactory
00108 //
00109 // This template class implements the specific typed factory for a given
00110 // C++ RTTI-capable object.
00111 //
00112 /****************************************************************************/
00113 template <class T> 
00114 class TFactory : public IRttiFactory
00115 {
00116         virtual void *CreateObject(const char* pszClass) const 
00117         {
00118                 return !stricmp(typeid(T).name(), pszClass)
00119                                 ? (void*)( new T ) : (void*)0;
00120         }
00121 };
