00096 /*****************************************************************************
00097 ** Procedure:  IOCPThreadPool::Start
00098 ** 
00099 ** Arguments: 'nThreads' - Number of threads to use
00100 **            'hIOCP' - Handle to the I/O completion port
00101 ** 
00102 ** Returns: true/false success code
00103 ** 
00104 ** Description: This starts the IOCP manager
00105 **
00106 /****************************************************************************/
00107 bool IOCPThreadPool::Start(int nConcurrentThreads, int nStartThreads, HANDLE hIOCP) throw()
00108 {
00109         CCSLock<IOCPThreadPool> lockGuard(this);
00110 
00111         if (iocp_ == NULL) 
00112         {
00113                 // If we have worker threads in IOCP right now, then don't allow
00114                 // the server to be restarted.
00115                 if (!threads_.empty()) //lint !e1550 Exception not thrown
00116                         return false;
00117 
00118                 // Mark we are starting up.
00119                 shutdown_ = false;
00120 
00121                 // Set our defaults
00122                 SYSTEM_INFO sysInfo; ::GetSystemInfo(&sysInfo); 
00123                 if (nConcurrentThreads == 0)
00124                         nConcurrentThreads = static_cast<int>(sysInfo.dwNumberOfProcessors);
00125                 if (nStartThreads == 0)
00126                         nStartThreads = nConcurrentThreads + max((nConcurrentThreads/4),1);
00127 
00128                 // Create the IO completion port this manager will work with
00129                 iocp_ = CreateIoCompletionPort(hIOCP, NULL, 0, static_cast<DWORD>(nConcurrentThreads));
00130                 if (iocp_ == NULL)
00131                         return false;
00132 
00133                 // Set the number of threads -- this will create/destroy threads
00134                 if (!set_NumThreads(nStartThreads))
00135                 {
00136                         CloseHandle(iocp_);
00137                         iocp_ = NULL;
00138                         return false;
00139                 }
00140         }
00141         return true;
00142 
00143 }// IOCPThreadPool::Start

00145 /*****************************************************************************
00146 ** Procedure:  IOCPThreadPool::set_NumThreads
00147 ** 
00148 ** Arguments: 'nThreads' - Number of threads to use
00149 ** 
00150 ** Returns: true/false success code
00151 ** 
00152 ** Description: This changes the number of threads involved in the pool
00153 **
00154 /****************************************************************************/
00155 bool IOCPThreadPool::set_NumThreads(int nThreads) throw()
00156 {
00157         CCSLock<IOCPThreadPool> lockGuard(this);
00158 
00159         if (iocp_ == NULL)
00160                 return false;
00161 
00162         // Save off the current thread count
00163         int numThreads = get_NumThreads();
00164 
00165         // Already have the given number of threads?
00166         if (numThreads == nThreads)
00167                 return true;
00168 
00169         // Or if we need to add threads to our list
00170         if (nThreads > numThreads)
00171         {
00172                 unsigned int nThreadID;
00173                 for (int i = numThreads; i < nThreads; ++i)
00174                 {
00175                         HANDLE hThread = reinterpret_cast<HANDLE>(_beginthreadex(NULL, 0, &InternalEventIOEntry, static_cast<void*>(this), 0, &nThreadID));
00176                         if (hThread) {
00177                                 InterlockedIncrement(&numThreads_);
00178                                 threads_.push_back(std::make_pair(nThreadID, hThread));
00179                         }
00180                 }
00181         }
00182 
00183         // Or if we need to remove threads from our list
00184         else // (nThreads < numThreads)
00185         {
00186                 for (int i = numThreads; i > nThreads; --i)
00187                         ::PostQueuedCompletionStatus(iocp_, 0, (TP_COMPLETION_KEY)-1, NULL);
00188         }
00189 
00190         return true;
00191 
00192 }// IOCPThreadPool::set_NumThreads

00195 /*****************************************************************************
00196 ** Procedure:  IOCPThreadPool::Shutdown
00197 ** 
00198 ** Arguments: void
00199 ** 
00200 ** Returns: true/false success code
00201 ** 
00202 ** Description: This stops the server
00203 **
00204 /****************************************************************************/
00205 bool IOCPThreadPool::Shutdown(DWORD dwWaitTime) throw()
00206 {
00207         bool allEnded = false;
00208         CCSLock<IOCPThreadPool> lockGuard(this);
00209 
00210         if (iocp_ != NULL && !shutdown_)
00211         {
00212                 // Mark that we are shutting down
00213                 shutdown_ = true;
00214 
00215                 try
00216                 {
00217                         // Copy the list of handles into a flat array
00218                         std::vector<HANDLE> arrHandles; arrHandles.reserve(threads_.size());
00219                         for (ThreadList::iterator it = threads_.begin(); it != threads_.end(); ++it) 
00220                         {
00221                                 // Do not wait on ourselves.
00222                                 if (it->first != GetCurrentThreadId())
00223                                 {
00224                                         HANDLE h;
00225                                         if (::DuplicateHandle(GetCurrentProcess(), it->second, GetCurrentProcess(), &h, 0, FALSE, DUPLICATE_SAME_ACCESS))
00226                                                 arrHandles.push_back(h);
00227                                 }
00228                         }
00229 
00230                         // Unlock the guard; this allows the threads to exit.
00231                         lockGuard.Unlock();
00232 
00233                         // Stop all the worker threads
00234                         set_NumThreads(0);
00235 
00236                         // Wait for all the threads to terminate and then close the handles.
00237                         if (!arrHandles.empty())
00238                         {
00239                                 allEnded = (WaitForMultipleObjects(static_cast<DWORD>(arrHandles.size()), &arrHandles[0], TRUE, dwWaitTime) != WAIT_TIMEOUT);
00240                                 std::for_each(arrHandles.begin(), arrHandles.end(), stdx::sc_ptr_fun(&CloseHandle));
00241                         }
00242                 }
00243                 catch(...)
00244                 {
00245                 }
00246 
00247                 // Close the IOCP port
00248                 ::CloseHandle(iocp_);
00249                 iocp_ = NULL;
00250                 
00251                 return allEnded;
00252         }
00253 
00254         return true;
00255 
00256 }// IOCPThreadPool::Shutdown

00361 /*****************************************************************************
00362 ** Procedure:  IOCPThreadPool::Run
00363 ** 
00364 ** Arguments: void
00365 ** 
00366 ** Returns: void
00367 ** 
00368 ** Description: This is the function that processes overlapped I/O events.
00369 **              It can have several threads processing it to handle socket
00370 **              traffic.
00371 **
00372 /****************************************************************************/
00373 void IOCPThreadPool::Run()
00374 {
00375         LPOVERLAPPED pio; DWORD dwBytesTransferred; TP_COMPLETION_KEY clientKey;
00376 
00377         // Note the worker thread starting ..
00378         WorkerThreadStart();
00379 
00380         // Wait for new events on any socket
00381         for (;;)
00382         {
00383                 // Clear out the input data
00384                 pio = NULL; dwBytesTransferred = 0; clientKey = 0;
00385 
00386                 // Get the next completion I/O block.
00387                 BOOL rc = GetQueuedCompletionStatus(iocp_, &dwBytesTransferred, &clientKey, &pio, timeout_);
00388                 if (clientKey == (TP_COMPLETION_KEY)-1)
00389                         break;
00390 
00391                 // Check the result code and make sure we save off the correct error.
00392                 DWORD dwLastError = (rc == FALSE) ? GetLastError() : 0;
00393 
00394                 // Invalid IOCP handle or timeout? exit the thread.
00395                 if (rc == FALSE && dwLastError == ERROR_INVALID_HANDLE)
00396                         break;
00397 
00398                 // Verify the OVERLAPPED structure; if it fails then loop around and get the next one.
00399                 if (!(pio == NULL || (!IsBadReadPtr(pio, sizeof(OVERLAPPED)))))
00400                 {
00401                         JTI_ASSERT(!"Invalid OVERLAPPED I/O packet dequeued from completion port");
00402                         continue;
00403                 }
00404 
00405                 // Call virtual worker function; do not allow exceptions outside this function.
00406                 try
00407                 {
00408                         //lint -e{1933} call to unqualified virtual function
00409                         if (ProcessWork(pio, dwBytesTransferred, clientKey, rc, dwLastError))
00410                                 break;
00411                 }
00412                 catch(...)
00413                 {
00414                         // Let the derived class cleanup before termination
00415                         WorkerThreadEnd();
00416 
00417                         // Cleanup the thread internally
00418                         OnThreadClosing();
00419 
00420                         // Rethrow
00421                         throw;
00422                 }
00423         }
00424 
00425         // Note the worker thread ending
00426         WorkerThreadEnd();
00427 
00428         // Cleanup the thread
00429         OnThreadClosing();
00430 
00431 }// IOCPThreadPool::Run

00434 /*****************************************************************************
00435 ** Procedure:  IOCPThreadPool::OnThreadClosing
00436 ** 
00437 ** Arguments: void
00438 ** 
00439 ** Returns: void
00440 ** 
00441 ** Description: This function closes out a thread.
00442 **
00443 /****************************************************************************/
00444 void IOCPThreadPool::OnThreadClosing() throw()
00445 {
00446         // Remove the thread count
00447         InterlockedDecrement(&numThreads_);
00448 
00449         // Remove the handle from the array and close it.
00450         CCSLock<IOCPThreadPool> lockGuard(this);
00451         ThreadList::iterator it = std::find_if(threads_.begin(), threads_.end(), 
00452                 search_bytid(GetCurrentThreadId()));
00453         if (it != threads_.end()) {
00454                 CloseHandle(it->second);
00455                 threads_.erase(it);
00456         }
00457 
00458 }// IOCPThreadPool::OnThreadClosing
