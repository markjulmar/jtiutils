00449         HRESULT OpenConnection(bool& bConnOpened, const char* pszConnStr = NULL)
00450         {
00451                 USES_CONVERSION;
00452 
00453                 lastHresult_ = S_OK;
00454                 lastErr_.clear();
00455                 bConnOpened = false;
00456 
00457                 try
00458                 {
00459                         // Reset the connection data if necessary
00460                         if (pszConnStr != NULL && lstrcmpi(connstr_.c_str(), pszConnStr)) {
00461                                 CloseConnection();
00462                                 connstr_ = pszConnStr;
00463                         }
00464 
00465                         // If our connection has been reset, then create a new one
00466                         if (connPtr_ == 0)
00467                         {
00468                                 HR(connPtr_.CreateInstance(__uuidof(ADODB::Connection)));
00469                         }
00470 
00471                         // Connect to the database.
00472                         if (connPtr_->GetState() == ADODB::adStateClosed)
00473                         {
00474                                 HR(connPtr_->Open(A2W(connstr_.c_str()), _bstr_t(), _bstr_t(), ADODB::adConnectUnspecified));

00479                 {
00480                         lastHresult_ = e.Error();

00487                         lastHresult_ = E_UNEXPECTED;

00506         HRESULT ExecQuery(const char* pszQueryFormat, ...)
00507         {
00508                 stdx::inauto_ptr<char> chBuff;
00509 
00510                 // Look for a format signature.
00511                 if (strchr(pszQueryFormat, '%') != NULL)
00512                 {
00513                         va_list args;
00514                         va_start(args, pszQueryFormat);
00515 
00516                         // Allocate a buffer for our query
00517                         chBuff.reset(new char[MAX_QUERY_STRING]);
00518                         LPSTR lpBuff = chBuff.get();
00519 
00520                         // Format the string
00521                         int nCount = _vsnprintf(lpBuff, MAX_QUERY_STRING, pszQueryFormat, args);
00522                         if (nCount == -1)
00523                                 *(lpBuff + MAX_QUERY_STRING-1) = '\0';
00524                         va_end(args);
00525                         pszQueryFormat = lpBuff;
00526                 }
00527 
00528                 // Run the query
00529                 try
00530                 {
00531                         // Attempt to open the connection if necessary.
00532                         bool bConnOpened = false;
00533                         HR(OpenConnection(bConnOpened));
00534 
00535                         // Execute the passed SQL query; assume it is a text command
00536                         connPtr_->CommandTimeout = timeout_;
00537                         connPtr_->Execute(_bstr_t(pszQueryFormat), NULL, ADODB::adCmdText | ADODB::adExecuteNoRecords);
00538                 }
00539                 catch(const _com_error &e)
00540                 {
00541                         lastHresult_ = e.Error();
00542                         lastErr_ = comutils::_format_com_error(e);
00543                         CloseConnection();
00544                 }
00545                 catch(...)
00546                 {
00547                         CloseConnection();
00548                         lastHresult_ = E_UNEXPECTED;
00549                         lastErr_ = comutils::_format_com_error(lastHresult_, NULL, NULL, "An unexpected exception has occurred.");
00550                 }
00551                 return lastHresult_;
00552         }

00554         HRESULT GetRecordset(ADODB::_Recordset** ppRecordset, const char* pszQueryFormat, ...)
00555         {
00556                 // Validate incoming parameters
00557                 COINITPARAM(ppRecordset);
00558                 stdx::inauto_ptr<char> chBuff;
00559 
00560                 // If this is a var-arg command, then format it.
00561                 if (strchr(pszQueryFormat, '%') != NULL)
00562                 {
00563                         va_list args;
00564                         va_start(args, pszQueryFormat);
00565 
00566                         // Allocate a buffer for our query
00567                         chBuff.reset(new char[MAX_QUERY_STRING]);
00568                         LPSTR lpBuff = chBuff.get();
00569 
00570                         // Format the string
00571                         int nCount = _vsnprintf(lpBuff, MAX_QUERY_STRING, pszQueryFormat, args);
00572                         if (nCount == -1)
00573                                 *(lpBuff + MAX_QUERY_STRING-1) = '\0';
00574                         va_end(args);
00575                         pszQueryFormat = lpBuff;
00576                 }
00577 
00578                 try
00579                 {
00580                         // Open the connection if necessary
00581                         bool bConnOpened = false;
00582                         HR(OpenConnection(bConnOpened));
00583 
00584                         // Create an ADO recordset object
00585                         ADODB::_RecordsetPtr spRecordset(__uuidof(ADODB::Recordset), NULL, CLSCTX_INPROC);
00586                         if (spRecordset == NULL)
00587                         {
00588                                 HR(E_NOINTERFACE);
00589                         }
00590                         else
00591                         {
00592                                 // Set the command timeout
00593                                 connPtr_->CommandTimeout = timeout_;
00594 
00595                                 // Initialize the client-side cursor.
00596                                 spRecordset->CursorLocation = ADODB::adUseClient;
00597 
00598                                 // Connect the recordset to the selected connection.
00599                                 spRecordset->PutRefActiveConnection(connPtr_);
00600 
00601                                 // Populate our recordset by executing the provided query. Note, we ask for a returned
00602                                 // static recordset since we are about to go disconnected, we cannot use a dynamic recordset.
00603                                 spRecordset->Open(_bstr_t(pszQueryFormat), _variant_t(DISP_E_PARAMNOTFOUND, VT_ERROR), ADODB::adOpenStatic, ADODB::adLockBatchOptimistic, ADODB::adCmdText);
00604 
00605                                 // If the recordset is closed, then it returned no records.
00606                                 // Otherwise, if there is returned data, then cause the provider to 
00607                                 // fetch it by moving to the first row.
00608                                 if (spRecordset->State != ADODB::adStateClosed && spRecordset->RecordCount > 0)
00609                                         spRecordset->MoveFirst();
00610 
00611                                 // Now disconnect the recordset - this leaves all the data fetched in place, but we no longer
00612                                 // have a database connection attached to it (i.e. future updates/deletes, etc. don't work).
00613                                 // This allows the connection to be re-pooled when it is no longer in use.
00614                                 spRecordset->PutRefActiveConnection(NULL);
00615 
00616                                 // Now detach the actual ADO object from our wrapper and assign it to the passed recordset object.
00617                                 spRecordset->QueryInterface(__uuidof(*ppRecordset), reinterpret_cast<void**>(ppRecordset));
00618                         }
00619                 }
00620                 catch(const _com_error &e)
00621                 {
00622                         lastHresult_ = e.Error();
00623                         lastErr_ = comutils::_format_com_error(e);
00624                         CloseConnection();
00625                 }
00626                 catch(...)
00627                 {
00628                         CloseConnection();
00629                         lastHresult_ = E_UNEXPECTED;
00630                         lastErr_ = comutils::_format_com_error(lastHresult_, NULL, NULL, "An unexpected exception has occurred.");
