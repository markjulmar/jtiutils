00040 /****************************************************************************/
00041 // Singleton
00042 //
00043 // This template class wraps a class and gives it singleton characteristics.
00044 //
00045 /****************************************************************************/
00046 template <class T, 
00047                   int Longevity = 0,
00048                   class LockType = SimpleMultiThreadModel>
00049 class Singleton
00050 {
00051 // Class data
00052 private:
00053         // This implements our global class data; The original release of VS.NET did 
00054         // not allow for UDT class template parameters outside the template definition 
00055         // itself.  This construct allows a similar functionality which will
00056         // compile with VS2002 and beyond.
00057         static bool* _Destroyed() {     static bool _gDestroyed = false; return &_gDestroyed; }
00058         static T** _Instance() { static T* _gInstance = 0; return &_gInstance; }
00059         static typename LockType::CriticalSection& _Lock() { static typename LockType::CriticalSection _gLock; return _gLock; }
00060 
00061 // Implementation functions
00062 public:
00063         // This method returns the singleton instance
00064         static T& Instance()
00065         {
00066                 // Use the double-check lock pattern to test the pointer.
00067                 T** pi = _Instance();
00068                 if (*pi == NULL)
00069                 {
00070                         CCSLock<LockType::CriticalSection> guard(&_Lock());
00071                         if (*pi == NULL)
00072                         {
00073                                 if (*_Destroyed() == true)
00074                                         throw std::logic_error("Singleton Dead Reference Detected");
00075                                 *pi = new T();
00076                                 LifetimeTracker<T,CallbackDeleter<T> >::SetLongevity(*pi, Longevity, 
00077                                                 CallbackDeleter<T>(&DestroySingleton));
00078                         }
00079                 }
00080                 return **pi;
00081         }
00082 
00083 // Internal functions
00084 private:
00085     static void DestroySingleton(T*)
00086         {
00087                 CCSLock<LockType::CriticalSection> guard(&_Lock());
00088                 T** pi = _Instance();
00089         
00090                 delete (*pi);
00091         *pi = NULL;
00092         *_Destroyed() = true;
00093         }
00094 
00095 // Unavailable functions
00096 private:
00097         Singleton();
00098 };
00099
