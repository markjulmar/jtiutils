00493         bool Populate(const TString& logName, HANDLE eventLog, DWORD dwPos=0)
00494         {
00495                 // First try to read from our indicated position.
00496                 if (!ReadEventLogEntries(logName, eventLog, EVENTLOG_FORWARDS_READ | EVENTLOG_SEEK_READ, dwPos))
00497                 {
00498                         DWORD rc = GetLastError();
00499                         if (rc == ERROR_INVALID_PARAMETER)
00500                         {
00501                                 // Per KB
00502                                 // The ReadEventLog() Win32 API function might fail and GetLastError() returns 87 
00503                                 // (ERROR_INVALID_PARAMETERS) while having all valid parameters passed to ReadEventLog(). 
00504                                 // Cause: The Event Logging Service fails to process the read operation when an application 
00505                                 // uses the ReadEventLog() function with the EVENTLOG_SEEK_READ flag to read large event log file. 
00506                                 if (!ReadEventLogEntries(logName, eventLog, EVENTLOG_FORWARDS_READ | EVENTLOG_SEQUENTIAL_READ, dwPos))
00507                                         throw EventLogException("ReadEventLog failed", GetLastError());
00508                         }
00509                         else if (rc != ERROR_HANDLE_EOF)
00510                                 throw EventLogException("ReadEventLog failed", rc);
00511                 }
00512                 return !arrEntries_.empty();
00513         }

00515         bool ReadEventLogEntries(const TString& logName, HANDLE eventLog, DWORD dwType, DWORD dwPos)
00516         {
00517                 BYTE bBuffer[1024]; 
00518                 EVENTLOGRECORD *pevlr = (EVENTLOGRECORD*) bBuffer; 
00519                 DWORD dwRead, dwNeeded; 
00520 
00521                 // Read the 1st record
00522                 if (ReadEventLog(eventLog, 
00523                         dwType, dwPos, pevlr, sizeof(bBuffer), &dwRead, &dwNeeded))
00524                 {
00525                         do 
00526                         {
00527                                 while (dwRead > 0) 
00528                                 {
00529                                         arrEntries_.push_back(EventLogEntry(logName, pevlr));
00530                                         dwRead -= pevlr->Length; 
00531                                         pevlr = (EVENTLOGRECORD*)((LPBYTE)pevlr + pevlr->Length); 
00532                                 } 
00533                                 pevlr = (EVENTLOGRECORD*) bBuffer; 
00534                         } while (ReadEventLog(eventLog,
00535                                 EVENTLOG_FORWARDS_READ | EVENTLOG_SEQUENTIAL_READ,
00536                                 0,            // ignored for sequential reads 
00537                                 pevlr,        // pointer to buffer 
00538                                 sizeof(bBuffer),  // size of buffer 
00539                                 &dwRead,      // number of bytes read 
00540                                 &dwNeeded));   // bytes in next record 
00541                 }
00542                 return !arrEntries_.empty();
00543         }

00903         void EventLogWatcher()
00904         {
00905                 // Open a new handle to the event log
00906                 HANDLE hEventLog = ::OpenEventLog(machineName_.c_str(), logName_.c_str());
00907 
00908                 HANDLE arrHandles[2];
00909                 arrHandles[0] = evtStop_;
00910                 arrHandles[1] = CreateEvent(NULL, TRUE, FALSE, NULL);
00911                 NotifyChangeEventLog(hEventLog, arrHandles[1]);
00912                 
00913                 while (raisingEvents_)
00914                 {
00915                         // First record how many records are in the event log currently.
00916                         DWORD dwNumRecs = 0, dwOldestRecord = 0;
00917                         GetOldestEventLogRecord(hEventLog, &dwOldestRecord);
00918                         GetNumberOfEventLogRecords(hEventLog, &dwNumRecs);
00919 
00920                         // Wait for a change
00921                         DWORD rc = WaitForMultipleObjects(2, arrHandles, FALSE, INFINITE);
00922                         if (rc == WAIT_OBJECT_0)
00923                                 break;
00924 
00925                         // Build an EventLogWrittenEvent event and fire the event
00926                         EventLogWrittenEventArgs evArgs(logName_, eventLog_, (dwNumRecs + dwOldestRecord));
00927                         writeEvent_(evArgs);
00928                 }
00929 
00930                 // Close our handle to the event log.
00931                 CloseEventLog(hEventLog);
00932         }
