00261         {
00262                 if (psz == NULL) {
00263                         pszBuff_ = NULL;
00264                         return;

00266                 int nLength = MultiByteToWideChar(nCodePage, 0, psz, -1, NULL, 0)+1;

00270                         if (pszBuff_ == NULL)
00271                                 throw std::bad_alloc("CA2W failed to allocate buffer");

00273                 MultiByteToWideChar(nCodePage, 0, psz, -1, pszBuff_, nLength);
00274         }

00333 /*****************************************************************************
00334 ** Procedure:  patchk
00335 ** 
00336 ** Arguments:  szPattern - Pattern to match
00337 **             szString - Test string
00338 ** 
00339 ** Returns: TRUE if the string matches the given pattern
00340 ** 
00341 ** Description: This function pattern matches two unicode strings.
00342 **
00343 /****************************************************************************/
00344 inline bool patchk(const wchar_t *szPattern, const wchar_t *szString)
00345 {
00346         if (*szString == L'\0')
00347                 return (*szPattern == L'\0') ? true : false;
00348         if (*szPattern == L'*') {
00349                 if (*(szPattern+1) == L'\0')
00350                         return true;
00351                 int nLen = lstrlenW(szString);
00352                 for (int i = 0; i <= nLen; i++) {
00353                         if (patchk(szPattern+1, szString+i))
00354                                 return true;
00355                 }
00356         }
00357         if ((*szString == *szPattern) || (*szPattern == L'?')) {
00358                 if (patchk(szPattern+1, szString+1))
00359                         return true;
00360         }
00361         return false;
00362 }

00409 // _SupportsConversion
00410 //
00411 // This internal template structure determines whether a class has 
00412 // virtual functions defined for it.

00414 // Use the below SupportsConversion instead

00416 /******************************************************************************/

00424                 static Yes test(_Type2);

00433 /******************************************************************************/
00434 // SupportsConversion
00435 //
00436 // This template structure determines whether a class has virtual functions
00437 // defined for it.
00438 //
00439 // class A
00440 // {
00441 // };
00442 //
00443 // SupportsConversion<int,long>::check          == true
00444 // SupportsConversion<int,double>::check        == true
00445 // SupportsConversion<int,A>::check                     == false
00446 //
00447 /******************************************************************************/
00448 template <class _Type1, class _Type2>
00449 struct SupportsConversion
00450 {
00451         enum { check = JTI_Internal::_SupportsConversion<_Type1,_Type2>::check };
00452         enum { check_both = check && JTI_Internal::_SupportsConversion<_Type2, _Type1>::check };
00453 };

00455 /******************************************************************************/
00456 // IsDerivedFrom
00457 //
00458 // This template class determines whether a class (_Derived) derives from a
00459 // another class (_Base).
00460 //
00461 // class A 
00462 // {
00463 // };
00464 //
00465 // class B : public A
00466 // {
00467 // };
00468 //
00469 // Then you can test in the code:
00470 //
00471 //      IsDerivedFrom<A,B>::check == false
00472 //      IsDerivedFrom<B,A>::check == true
00473 //
00474 // To enforce a base interface at compile time:
00475 //
00476 // class C
00477 // {
00478 // };
00479 //
00480 // template <class T>
00481 // class D : public IsDerivedFrom<T,A>
00482 // {
00483 // };
00484 //
00485 // D<A> d1;  <-- this is ok
00486 // D<C> d2;  <-- this will fail @ compile time with an error.
00487 //
00488 /******************************************************************************/
00489 template <typename _Derived, typename _Base>
00490 class IsDerivedFrom
00491 {
00492         static void constraints(_Derived* p) {_Base* pb = p; pb = p; }
00493 public:
00494         enum { check = SupportsConversion<_Derived,_Base>::check && !SupportsConversion<_Base,_Derived>::check };
00495         IsDerivedFrom() { void(*p)(_Derived*) = constraints; p; }
00496 };

00530 /******************************************************************************/
00531 // Int2Type/Bool2Type
00532 //
00533 // This template structure allows an integer to become a valid type which can
00534 // then be used to change a function signature:
00535 //
00536 //      void DoTest(Int2Type<true>)
00537 //      {
00538 //              DoTestImpl();
00539 //      }
00540 //      void DoTest(Int2Type<false>)
00541 //      {
00542 //              cout << "DoTest not supported." << endl;
00543 //      }
00544 //
00545 // This can then be called as:
00546 //
00547 // DoTest(true) or DoTest(false) which will invoke either function.
00548 //
00549 /******************************************************************************/

00552 {
00553    enum { value = val };
00554 };

00556 /******************************************************************************/
00557 // Type2Type
00558 //
00559 // This template structure allows a type to be treated as a distinct type
00560 // for the purposes of template expansion.  This allows template specialization
00561 // to be treated differently by return type (which currently isn't enough
00562 // to differentiate a function in C++).
00563 //
00564 // template <class T, class U>
00565 // T* Create(const U& arg, Type2Type<T>)
00566 // {
00567 //         return new T(arg);
00568 // }
00569 // template <class U>
00570 // Widget* Create(const U& arg, Type2Type<Widget>)
00571 // {
00572 //     return new Widget(arg, -1);
00573 // }
00574 //
00575 // String* pStr = Create("Hello", Type2Type<String>());
00576 // Widget* pW = Create(100, Type2Type<Widget>());
00577 //
00578 /******************************************************************************/
00579 template<typename _Ty>
00580 struct Type2Type
00581 {
00582     typedef _Ty OriginalType;
00583 };
