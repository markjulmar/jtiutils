00035 /****************************************************************************/
00036 // MemoryMappedFile
00037 //
00038 // This class wraps a memory mapped file.
00039 //
00040 /****************************************************************************/
00041 class MemoryMappedFile
00042 {
00043 // Class data
00044 private:
00045         void* lpv_;
00046         __int64 size_;
00047 
00048 // Constructor
00049 public:
00050         MemoryMappedFile(const char* pszFile, 
00051                         DWORD dwDesiredAccess = GENERIC_READ, 
00052                         DWORD dwShareMode = FILE_SHARE_READ, 
00053                         LPSECURITY_ATTRIBUTES lpSecurityAttributes = NULL,
00054                         DWORD dwCreationDisposition = OPEN_EXISTING,
00055                         DWORD dwFlagsAndAttributes = FILE_ATTRIBUTE_NORMAL) : lpv_(0)
00056         {
00057                 USES_CONVERSION;
00058                 HANDLE hFile = CreateFile(A2T(const_cast<char*>(pszFile)), dwDesiredAccess, dwShareMode, lpSecurityAttributes, dwCreationDisposition, dwFlagsAndAttributes, NULL);
00059                 if (hFile != INVALID_HANDLE_VALUE)
00060                 {
00061                         // Get the total file size to map
00062                         DWORD dwSizeHigh = 0, dwSizeLow = GetFileSize(hFile, &dwSizeHigh);
00063 
00064                         // Store off the file size
00065                         size_ = ((static_cast<__int64>(dwSizeHigh)) << 32) + dwSizeLow;
00066 
00067                         // Create a file mapping object; it increments the ref count on the original file so close
00068                         // it now to avoid resource leaks.
00069                         DWORD protectionType = ((dwDesiredAccess & (FILE_WRITE_ATTRIBUTES|FILE_WRITE_DATA|FILE_WRITE_EA)) > 0) ? PAGE_READWRITE : PAGE_READONLY;
00070                         HANDLE hFileMap = CreateFileMapping(hFile, lpSecurityAttributes, protectionType, dwSizeHigh, dwSizeLow, NULL);
00071                         CloseHandle(hFile);
00072 
00073                         // Map a view of the file; again close the handle we used.
00074                         if (hFileMap != NULL) {
00075                                 protectionType = ((dwDesiredAccess & (FILE_WRITE_ATTRIBUTES|FILE_WRITE_DATA|FILE_WRITE_EA)) > 0) ? FILE_MAP_WRITE : FILE_MAP_READ;
00076                                 lpv_ = MapViewOfFile(hFileMap, protectionType, 0, 0, 0);
00077                                 CloseHandle(hFileMap);
00078                         }
00079                 }
00080         }
00081 
00082         ~MemoryMappedFile()
00083         {
00084                 if (lpv_ != NULL)
00085                         UnmapViewOfFile(lpv_);
00086         }
00087 
00088 // Properties
00089 public:
00090         __declspec(property(get=get_Pointer)) void* Buffer;
00091         __declspec(property(get=get_IsValid)) bool IsValid;
00092         __declspec(property(get=get_Size)) __int64 Size;
00093 
00094 // Property helpers
00095 public:
00096         bool get_IsValid() { return (lpv_ != NULL); }
00097         void* get_Pointer() { return lpv_; }
00098         __int64 get_Size() { return size_; }
00099 };
