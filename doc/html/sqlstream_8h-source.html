00041 /****************************************************************************/
00042 // CDlParam
00043 //
00044 // Basic parameter class - defines base behavior for all derived parameter
00045 // "breaker" classes.
00046 //
00047 /****************************************************************************/
00048 class CDlParam  
00049 {
00050 // Class data
00051 private:
00052         bool isDefined_;                                // true/false if the parameter is defined
00053         const std::wstring fieldName_;  // Field name
00054 
00055 // Constructor
00056 protected:
00057         explicit CDlParam(const wchar_t *szName, bool bIsDefined)
00058                 : isDefined_(bIsDefined), fieldName_(szName) {/* */};
00059 // Destructor
00060 public:
00061         virtual ~CDlParam() {}
00062 
00063 // Access methods
00064 public:
00065         // Returns the validated value combined with the field name
00066         std::pair<bool,std::pair<std::wstring,std::wstring> > AsValidatedPair() const {
00067                 std::pair<bool,std::wstring> pairValue = ValidatedValue(); //lint !e1933
00068                 std::pair<std::wstring,std::wstring> pairParam(fieldName_,pairValue.second);
00069                 return std::make_pair(pairValue.first, pairParam);
00070         }
00071 
00072         // Required implementation by derived classes
00073         virtual std::pair<bool,std::wstring> ValidatedValue() const = 0;
00074 
00075 // Methods
00076 protected:
00077         bool get_IsDefined() const { return isDefined_; }
00078         void set_IsDefined(bool bIsDefined) { isDefined_= bIsDefined;}
00079 
00080 // Non-implemented methods
00081 private:
00082         CDlParam(const CDlParam&);
00083         const CDlParam& operator=(const CDlParam&);
00084 };

00320 /****************************************************************************/
00321 // CDlCmdSQL
00322 //
00323 // Command broker which executes an SQL statement with optional parameters
00324 //
00325 /****************************************************************************/
00326 class CDlCmdSQL : public CDlCmd  
00327 {
00328 // Class data
00329 protected:
00330         enum { NOMINAL_ARG_COUNT = 5 };
00331         std::vector<std::wstring> argsArray_;
00332         const wchar_t *sqlText_;
00333         long argLength_;
00334 
00335 // Constructor/Destructor
00336 public:
00337         explicit CDlCmdSQL(const wchar_t* szName) : 
00338                 CDlCmd(), sqlText_(szName), argLength_(0) { argsArray_.reserve(NOMINAL_ARG_COUNT); }
00339 
00340 // Insertion operators
00341 public:
00342         CDlCmdSQL& operator <<(const CDlParam &param)
00343         {
00344                 std::pair<bool, std::pair<std::wstring,std::wstring> > pairValidated = param.AsValidatedPair();
00345                 if (pairValidated.first) {
00346                         const std::wstring& strValue = pairValidated.second.second;
00347                         argsArray_.push_back(strValue);
00348                         argLength_ += static_cast<long>(strValue.length());
00349                 }
00350                 return *this;
00351         }
00352 
00353 // Methods
00354 public:
00355         virtual std::wstring BuildCommand() const;
00356         void DropParams() 
00357         {
00358                 argsArray_.clear();
00359                 argLength_ = 0;
00360                 argsArray_.reserve(NOMINAL_ARG_COUNT);
00361         }
00362         virtual operator _bstr_t() const        //lint !e1930 Implicit conversion to another type
00363         {
00364                 return _bstr_t(BuildCommand().c_str());
00365         }
00366 
00367 // Non-implemented methods
00368 private:
00369         CDlCmdSQL(const CDlCmdSQL&);
00370         const CDlCmdSQL& operator=(const CDlCmdSQL&);
00371 };
