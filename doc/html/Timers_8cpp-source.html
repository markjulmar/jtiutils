00065 /*****************************************************************************
00066 ** Procedure:  TimerManager::KillTimer
00067 ** 
00068 ** Arguments:  'nTimer' - Timer ID to remove
00069 ** 
00070 ** Returns: Success indicator
00071 ** 
00072 ** Description: This removes a specific timer by id.
00073 **
00074 /****************************************************************************/
00075 bool TimerManager::KillTimer(int nTimer)
00076 {
00077         CCSLock<TimerManager> lockGuard(this);
00078         std::vector<TimerEntry*>::iterator it = std::find_if(qTimers_.begin(), qTimers_.end(), 
00079                 stdx::compose_f_gx(std::bind2nd(std::equal_to<int>(),nTimer), 
00080                 std::mem_fun(&TimerEntry::get_ID)));
00081         bool foundTimer = (it != qTimers_.end());
00082         if (foundTimer) {
00083                 TimerEntry* pEntry = *it;
00084                 qTimers_.erase(it);
00085                 delete pEntry;
00086         }
00087         // We always reset the event so that the timer thread "resets" itself
00088         // as to the next timeout event.
00089         evtNewTimer_.SetEvent();
00090         return foundTimer;
00091 
00092 }// TimerManager::KillTimer

00094 /*****************************************************************************
00095 ** Procedure:  TimerManager::AddTimer
00096 ** 
00097 ** Arguments:  'nTimer' - Unique timer id
00098 **             'msecTimeout' - Timeout
00099 **             'func' - Delegate to fire
00100 ** 
00101 ** Returns: void
00102 ** 
00103 ** Description: This adds a new timer
00104 **
00105 /****************************************************************************/
00106 void TimerManager::AddTimer(int nTimer, DWORD msecTimeout, DelegateInvoke_1<int>* Func)
00107 {
00108         CCSLock<TimerManager> lockGuard(this);
00109 
00110         // Verify this entry doesn't exist already
00111         std::vector<TimerEntry*>::iterator it = std::find_if(qTimers_.begin(), qTimers_.end(), 
00112                 stdx::compose_f_gx(std::bind2nd(std::equal_to<int>(),nTimer), 
00113                 std::mem_fun(&TimerEntry::get_ID)));
00114         if (it != qTimers_.end()) {
00115                 TimerEntry* pEntry = *it;
00116                 qTimers_.erase(it);
00117                 delete pEntry;
00118         }
00119 
00120         // Add the element to the list
00121         qTimers_.push_back(JTI_NEW TimerEntry(nTimer, msecTimeout, Func));
00122 
00123         // If the thread isn't running yet, start it up.
00124         if (thread_ == 0)
00125         {
00126                 unsigned nThreadId;
00127                 thread_ = reinterpret_cast<HANDLE>(_beginthreadex(NULL, 0, &TimerWorkerThread, 
00128                                                 reinterpret_cast<void*>(this), 0, &nThreadId));
00129                 if (thread_ == 0)
00130                         throw std::runtime_error("AddTimer unable to create worker thread.");
00131         }
00132 
00133         // Wake the thread up.
00134         evtNewTimer_.SetEvent();
00135 
00136 }// TimerManager::AddTimer

00138 /*****************************************************************************
00139 ** Procedure:  TimerManager::TimerWorker
00140 ** 
00141 ** Arguments:  void
00142 ** 
00143 ** Returns: void
00144 ** 
00145 ** Description: This is the worker thread which fires timers
00146 **
00147 /****************************************************************************/
00148 void TimerManager::TimerWorker()
00149 {
00150         CCSLock<TimerManager> lockGuard(this, false);
00151         HANDLE arrHandles[] = { evtStop_.get(), evtNewTimer_.get() };
00152         for (;;)
00153         {
00154                 // Lock for exclusive access to the array
00155                 lockGuard.Lock();
00156 
00157                 // No more timers? exit
00158                 if (qTimers_.empty())
00159                         break;
00160 
00161                 // Re-sort our vector
00162                 if (qTimers_.size() > 1)
00163                 {
00164                         std::partial_sort(qTimers_.begin(), qTimers_.begin()+1, 
00165                                 qTimers_.end(), stdx::ptr_less<TimerEntry>());
00166                 }
00167 
00168                 // Get the sleep timeout
00169                 TimerEntry* pEntry = qTimers_.front();
00170                 DWORD dwTickLast = pEntry->get_LastFireTime();
00171                 DWORD dwElapsed = ELAPSED_TIME(dwTickLast);
00172                 DWORD dwTimeout = pEntry->get_Interval();
00173                 
00174                 // Allow other threads to access array again.
00175                 lockGuard.Unlock();
00176 
00177                 // Now determine how long to sleep
00178                 if (dwElapsed < dwTimeout)
00179                         dwTimeout = (dwTimeout - dwElapsed);
00180                 else
00181                         dwTimeout = 0;
00182 
00183                 // Go to sleep
00184                 DWORD rc = WaitForMultipleObjects(2, arrHandles, FALSE, dwTimeout);
00185                 if (rc == WAIT_OBJECT_0)
00186                         break;
00187                 else if (rc == WAIT_OBJECT_0 + 1)
00188                 {
00189                         // New object - loop around and re-determine sleep time.
00190                         evtNewTimer_.ResetEvent();
00191                         continue;
00192                 }
00193                 else if (rc == WAIT_TIMEOUT)
00194                 {
00195                         // Fire a timer
00196                         lockGuard.Lock();
00197 
00198                         // Catch the condition where the timer has been removed _right_ before
00199                         // the timeout occurs and we end up with a timeout before the "change"
00200                         // event.
00201                         if (!qTimers_.empty())
00202                         {
00203                                 TimerEntry* pTimer = qTimers_.front();
00204                                 if (ELAPSED_TIME(pTimer->get_LastFireTime()) >= pTimer->get_Interval())
00205                                         pTimer->FireTimer();
00206                         }
00207 
00208                         lockGuard.Unlock();
00209                 }
00210         }
00211         CloseHandle(thread_);
00212         thread_ = 0;
00213 
00214 }// TimerManager::TimerWorker

00231         if (*invokeFunc_) 
00232         { 
00233                 lastFired_ = GetTickCount(); 
00234                 (*invokeFunc_)(timerID_); 
00235                 return true;
00236         } 
00237         return false;

00239 }// TimerEntry::FireTimer
