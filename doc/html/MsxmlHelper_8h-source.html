00267         void SetNodeValueType(MSXML2::IXMLDOMNodePtr& spNode, _variant_t vt, VARTYPE vType=VT_ERROR)
00268         {
00269                 // Element?
00270                 if (spNode->nodeType == MSXML2::NODE_ELEMENT && spNode->hasChildNodes() == VARIANT_TRUE)
00271                 {
00272                         // Remove for all child "#text" value
00273                         MSXML2::IXMLDOMNodeListPtr piList = spNode->childNodes;
00274                         if (piList != NULL)
00275                         {
00276                                 MSXML2::IXMLDOMNodePtr piNode = piList->nextNode();
00277                                 while (piNode != NULL)
00278                                 {
00279                                         if (piNode->nodeType == MSXML2::NODE_TEXT)
00280                                                 spNode->removeChild(piNode);
00281                                         piNode = piList->nextNode();
00282                                 }
00283                         }
00284                 }
00285 
00286                 // Append a new child text node.
00287                 MSXML2::IXMLDOMNodePtr spNodeText = xmlDOM_->createNode(_variant_t(static_cast<long>(MSXML2::NODE_TEXT)), "","");
00288                 spNode->appendChild(spNodeText);
00289 
00290                 if (!hasSchema_)
00291                 {
00292                         switch(vType)
00293                         {
00294                                 case VT_BSTR: spNode->PutdataType(L"string"); break;
00295                                 case VT_UI4:  spNode->PutdataType(L"ui4"); break;
00296                                 case VT_I4:       spNode->PutdataType(L"i4"); break;
00297                                 case VT_BOOL: spNode->PutdataType(L"boolean"); break;
00298                                 case VT_DATE: spNode->PutdataType(L"dateTime"); break;
00299                                 default: break;
00300                         }
00301                 }
00302 
00303                 // Set the value of the text
00304                 if (vType == VT_BOOL)
00305                         spNodeText->nodeTypedValue = _variant_t( (static_cast<bool>(vt) == true) ? (short)1 : (short)0, VT_I2);
00306                 else
00307                         spNodeText->nodeTypedValue = vt;
00308         }
