00053 /*****************************************************************************
00054 ** Procedure:  FileSystemWatcher::OnFileWriteOccurred
00055 ** 
00056 ** Arguments:  'arrThen' - Array which captures the previous snap point
00057 **             'arrNow' - New snap point
00058 ** 
00059 ** Returns: void
00060 ** 
00061 ** Description: This handles the case where a file is written to
00062 **
00063 /****************************************************************************/
00064 void FileSystemWatcher::OnFileWriteOccurred(std::vector<FILEENTRY>& arrThen, std::vector<FILEENTRY>& arrNow)
00065 {
00066         // Search for changes in file size.
00067         for (std::vector<FILEENTRY>::iterator itThen = arrThen.begin();
00068                         itThen != arrThen.end(); ++itThen)
00069         {
00070                 // Look for the element in the new array.  It should be there.
00071                 std::vector<FILEENTRY>::iterator itNow = findEntry(itThen, arrNow.begin(), arrNow.end());
00072                 if (itNow != arrNow.end())
00073                 {
00074                         if (itNow->fileTime_.dwLowDateTime != itThen->fileTime_.dwLowDateTime ||
00075                                 itNow->fileTime_.dwHighDateTime != itThen->fileTime_.dwHighDateTime)
00076                                 changedEvent_(FileSystemWatcherEvent(FileSystemWatcherEvent::Changed, itNow->dirName_.c_str(), itNow->fileName_.c_str()));
00077                 }
00078         }
00079 
00080 }// FileSystemWatcher::OnFileWriteOccurred

00082 /*****************************************************************************
00083 ** Procedure:  FileSystemWatcher::OnFileChangeOccurred
00084 ** 
00085 ** Arguments:  'arrThen' - Array which captures the previous snap point
00086 **             'arrNow' - New snap point
00087 ** 
00088 ** Returns: void
00089 ** 
00090 ** Description: This handles the case where a file is renamed, created or deleted.
00091 **
00092 /****************************************************************************/
00093 void FileSystemWatcher::OnFileChangeOccurred(std::vector<FILEENTRY>& arrThen, std::vector<FILEENTRY>& arrNow)
00094 {
00095         // Search for renamed/deleted elements.
00096         for (std::vector<FILEENTRY>::iterator itThen = arrThen.begin();
00097                         itThen != arrThen.end(); ++itThen)
00098         {
00099                 // Look for the element in the new array.  If not found, it has either been
00100                 // deleted or renamed.
00101                 if (findEntry(itThen, arrNow.begin(), arrNow.end()) == arrNow.end())
00102                 {
00103                         // If the size of the arrays are different, then this is easy -- it has been deleted.
00104                         if (arrNow.size() != arrThen.size())
00105                                 deletedEvent_(FileSystemWatcherEvent(FileSystemWatcherEvent::Deleted, itThen->dirName_.c_str(), itThen->fileName_.c_str()));
00106 
00107                         // Otherwise it is a bit more difficult.  Look for a new file in the new array
00108                         else
00109                         {
00110                                 std::vector<FILEENTRY>::iterator itNow = arrNow.begin();
00111                                 for (; itNow != arrNow.end(); ++itNow)
00112                                 {
00113                                         if (findEntry(itNow, arrThen.begin(), arrThen.end()) == arrThen.end())
00114                                                 break;
00115                                 }
00116                                 renamedEvent_(FileSystemWatcherEvent(itNow->dirName_.c_str(), itNow->fileName_.c_str(), itThen->dirName_.c_str(), itThen->fileName_.c_str()));
00117                         }
00118                 }
00119         }
00120 
00121         // Search for created elements
00122         if (arrNow.size() > arrThen.size())
00123         {
00124                 for (std::vector<FILEENTRY>::iterator itNow = arrNow.begin();
00125                                 itNow != arrNow.end(); ++itNow)
00126                 {
00127                         // Look for the element in the new array.  If not found, it has either been
00128                         // deleted or renamed.
00129                         if (findEntry(itNow, arrThen.begin(), arrThen.end()) == arrThen.end())
00130                                 createdEvent_(FileSystemWatcherEvent(FileSystemWatcherEvent::Created, itNow->dirName_.c_str(), itNow->fileName_.c_str()));
00131                 }
00132         }
00133 
00134 }// FileSystemWatcher::OnFileChangeOccurred

00140 **             'arrNow' - New snap point
00141 ** 
00142 ** Returns: void
00143 ** 
00144 ** Description: This handles the case where a directory is renamed, created or deleted.
00145 **
00146 /****************************************************************************/
00147 void FileSystemWatcher::OnDirChangeOccurred(std::vector<FILEENTRY>& arrThenIn, std::vector<FILEENTRY>& arrNowIn)
00148 {
00149         // Get a copy of the two directories
00150         std::vector<FILEENTRY> arrNow, arrThen;
00151         std::vector<FILEENTRY>::iterator itNow, itThen;
00152         for (itNow = arrNowIn.begin(); itNow != arrNowIn.end(); ++itNow)
00153         {
00154                 if (itNow->attribs_ & FILE_ATTRIBUTE_DIRECTORY)
00155                         arrNow.push_back(*itNow);
00156         }
00157 
00158         for (itThen = arrThenIn.begin(); itThen != arrThenIn.end(); ++itThen)
00159         {
00160                 if (itThen->attribs_ & FILE_ATTRIBUTE_DIRECTORY)
00161                         arrThen.push_back(*itThen);
00162         }
00163 
00164         // Search for renamed/deleted elements.
00165         for (itThen = arrThen.begin(); itThen != arrThen.end(); ++itThen)
00166         {
00167                 // Look for the element in the new array.  If not found, it has either been
00168                 // deleted or renamed.
00169                 if (findEntry(itThen, arrNow.begin(), arrNow.end()) == arrNow.end())
00170                 {
00171                         // If the size of the arrays are different, then this is easy -- it has been deleted.
00172                         if (arrNow.size() != arrThen.size())
00173                                 deletedEvent_(FileSystemWatcherEvent(FileSystemWatcherEvent::Deleted, itThen->dirName_.c_str(), _T("")));
00174 
00175                         // Otherwise it is a bit more difficult.  Look for a new file in the new array
00176                         else
00177                         {
00178                                 for (itNow = arrNow.begin(); itNow != arrNow.end(); ++itNow)
00179                                 {
00180                                         if (findEntry(itNow, arrThen.begin(), arrThen.end()) == arrThen.end())
00181                                                 break;
00182                                 }
00183                                 renamedEvent_(FileSystemWatcherEvent(itNow->dirName_.c_str(), _T(""), itThen->dirName_.c_str(), _T("")));
00184                         }
00185                 }
00186         }
00187 
00188         // Search for created elements
00189         if (arrNow.size() > arrThen.size())
00190         {
00191                 for (itNow = arrNow.begin(); itNow != arrNow.end(); ++itNow)
00192                 {
00193                         // Look for the element in the new array.  If not found, it has either been
00194                         // deleted or renamed.
00195                         if (findEntry(itNow, arrThen.begin(), arrThen.end()) == arrThen.end())
00196                                 createdEvent_(FileSystemWatcherEvent(FileSystemWatcherEvent::Created, itNow->dirName_.c_str(), _T("")));
00197                 }
00198         }
00199 
00200 }// FileSystemWatcher::OnDirChangeOccurred

00230 /*****************************************************************************
00231 ** Procedure:  FileSystemWatcher::SnapshotDirectory
00232 ** 
00233 ** Arguments:  'arr' - Array to fill
00234 **             'fIncludeSubDirs' - True to include subdirectories in snap point
00235 **             'pszDirName' - Directory name for subdirs
00236 ** 
00237 ** Returns: void
00238 ** 
00239 ** Description: This takes a snapshot of a directory at a point in time (snap point).
00240 **
00241 /****************************************************************************/
00242 void FileSystemWatcher::SnapshotDirectory(std::vector<FILEENTRY>& arr, bool fIncludeSubDirs, const TCHAR* pszDirName)
00243 {
00244         TString sDir = dirPath_;
00245         if (pszDirName)
00246         {
00247                 sDir += _T('\\');
00248                 sDir += pszDirName;
00249         }
00250 
00251         TString sMask = sDir; 
00252         if (!sMask.empty() && *(sMask.rbegin()) != _T('\\'))
00253                 sMask += _T('\\');
00254         if (!nameFilter_.empty())
00255                 sMask += nameFilter_;
00256         else
00257                 sMask += _T("*");
00258 
00259         WIN32_FIND_DATA fd;
00260         HANDLE hFind = FindFirstFile(sMask.c_str(), &fd);
00261         if (hFind != INVALID_HANDLE_VALUE)
00262         {
00263                 do
00264                 {
00265                         if ((fd.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY) == 0)
00266                         {
00267                                 ULARGE_INTEGER ul; ul.HighPart = fd.nFileSizeHigh; ul.LowPart = fd.nFileSizeLow;
00268                                 arr.push_back(FILEENTRY(sDir.c_str(), fd.cFileName, fd.ftLastWriteTime, ul.QuadPart, fd.dwFileAttributes));
00269                         }
00270                         else if (lstrcmpi(fd.cFileName,_T(".")) && lstrcmpi(fd.cFileName,_T("..")))
00271                         {
00272                                 arr.push_back(FILEENTRY(sDir.c_str(), fd.cFileName, fd.ftLastWriteTime, static_cast<__int64>(0), fd.dwFileAttributes));
00273                         }
00274 
00275                 } while (FindNextFile(hFind, &fd));
00276                 FindClose(hFind);       //lint !e534
00277         }
00278 
00279         // Now enumerate through sub-directories if necessary.
00280         if (fIncludeSubDirs)
00281         {
00282                 sMask = sDir; sMask += _T("\\*");
00283                 hFind = FindFirstFile(sMask.c_str(), &fd);
00284                 if (hFind != INVALID_HANDLE_VALUE)
00285                 {
00286                         do
00287                         {
00288                                 if ((fd.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY)>0 &&
00289                                         lstrcmpi(fd.cFileName,_T(".")) && lstrcmpi(fd.cFileName,_T("..")))
00290                                 {
00291                                         TString s;
00292                                         if (pszDirName) {
00293                                                 s = pszDirName;
00294                                                 s += _T("\\");
00295                                         }
00296                                         s += fd.cFileName;
00297                                         SnapshotDirectory(arr,fIncludeSubDirs,s.c_str());
00298                                 }
00299                         } while (FindNextFile(hFind, &fd));
00300                         FindClose(hFind);       //lint !e534
00301                 }
00302         }
00303 
00304         // Sort it by name
00305         std::sort(arr.begin(), arr.end(), FILEENTRY::sortby_name);
00306 
00307 }// FileSystemWatcher::SnapshotDirectory

00310 ** Procedure:  FileSystemWatcher::Restart
00311 ** 
00312 ** Arguments:  void
00313 ** 
00314 ** Returns: void
00315 ** 
00316 ** Description: This is called to reset the file system watcher
00317 **
00318 /****************************************************************************/
00319 void FileSystemWatcher::Restart()
00320 {
00321         if (fireEvents_)
00322         {
00323                 if (InterlockedCompareExchange(&threadRunning_, 1, 0) == 0)
00324                 {
00325                         JTI_ASSERT(hThread_ == NULL);
00326 
00327                         if (dirPath_.empty())
00328                                 throw std::runtime_error("Cannot start FileSystemWatcher without the Path property being set.");
00329 
00330                         // Start the thread to monitor the directory.
00331                         unsigned nThreadId;
00332                         hThread_ = reinterpret_cast<HANDLE>(_beginthreadex(NULL, 0, &WorkerThread, reinterpret_cast<void*>(this), 0, &nThreadId));
00333                         if (!hThread_)
00334                                 throw std::runtime_error("Failed to create FileSystemWatcher monitor thread.");
00335                 }
00336         }
00337         else
00338         {
00339                 HANDLE hThread = hThread_;
00340                 hThread_ = NULL;
00341                 evtStop_.SetEvent();    //lint !e534
00342                 if (hThread != NULL) {
00343             WaitForSingleObject(hThread, 3000);
00344                         CloseHandle(hThread);
00345         }
00346         }
00347 
00348 }// FileSystemWatcher::Restart

00350 /*****************************************************************************
00351 ** Procedure:  FileSystemWatcher::RunWorker
00352 ** 
00353 ** Arguments:  void
00354 ** 
00355 ** Returns: void
00356 ** 
00357 ** Description: This is the main monitor thread.  It sets up the file notification
00358 **              and then calls each delegate when an event occurs.
00359 **
00360 /****************************************************************************/
00361 void FileSystemWatcher::RunWorker()
00362 {
00363         std::vector<FILEENTRY> fileEntries; // File entries
00364         HANDLE hevtFileChange = 0, hevtDirChange = 0;
00365         HANDLE arrHandles[3] = {0};
00366         DWORD cntHandles = 0;
00367 
00368         while (fireEvents_)
00369         {
00370                 // Reset the event
00371                 evtStop_.ResetEvent(); //lint !e534
00372 
00373                 // Close all the file change handles
00374                 if (hevtFileChange)
00375                         FindCloseChangeNotification(hevtFileChange); //lint !e534
00376                 if (hevtDirChange)
00377                         FindCloseChangeNotification(hevtDirChange); //lint !e534
00378                 hevtFileChange = hevtDirChange = 0;
00379                 
00380                 // Now setup the handles
00381                 arrHandles[0] = evtStop_.get(); cntHandles++;
00382                 if (notifyFilter_ & ~DirectoryName)
00383                 {
00384                         hevtFileChange = FindFirstChangeNotification(dirPath_.c_str(), watchSubDirs_, (notifyFilter_&~DirectoryName));
00385                         if (hevtFileChange != INVALID_HANDLE_VALUE)
00386                                 arrHandles[cntHandles++] = hevtFileChange;
00387                         else hevtFileChange = NULL;
00388                 }
00389                                 
00390                 if (notifyFilter_ & DirectoryName)
00391                 {
00392                         hevtDirChange = FindFirstChangeNotification(dirPath_.c_str(), watchSubDirs_, FILE_NOTIFY_CHANGE_DIR_NAME);
00393                         if (hevtDirChange != INVALID_HANDLE_VALUE)
00394                                 arrHandles[cntHandles++] = hevtDirChange;
00395                         else hevtDirChange = NULL;
00396                 }
00397 
00398                 // If we have no directory to monitor, then exit immediately.
00399                 if (cntHandles == 1)
00400                 {
00401                         fireEvents_ = false;
00402                         break;
00403                 }
00404                         
00405                 // Take a snapshot of the directory as it is now.
00406                 SnapshotDirectory(fileEntries, watchSubDirs_);
00407 
00408                 // Now wait for one of our handles to signal
00409                 for (;;)
00410                 {
00411                         // Wait on the event
00412                         DWORD rc = WaitForMultipleObjects(cntHandles, arrHandles, FALSE, INFINITE);
00413                         if (rc == WAIT_OBJECT_0 || rc == WAIT_FAILED)
00414                                 break;
00415 
00416                         // Take a new snapshot
00417                         std::vector<FILEENTRY> arrNew;
00418                         SnapshotDirectory(arrNew, watchSubDirs_);
00419 
00420                         // Some file event occurred; determine which handle caused the trigger.
00421                         HANDLE hEvent = arrHandles[rc-WAIT_OBJECT_0];
00422                         if (hEvent == hevtFileChange)
00423                         {
00424                                 OnFileChangeOccurred(fileEntries, arrNew);
00425                                 OnFileWriteOccurred(fileEntries, arrNew);
00426                                 OnFileSizeChangeOccurred(fileEntries, arrNew);
00427                         }
00428                         else if (hEvent == hevtDirChange)
00429                                 OnDirChangeOccurred(fileEntries, arrNew);
00430 
00431                         // Copy over the new fileentry data
00432                         fileEntries = arrNew;
00433 
00434                         // Reset the fcn and loop around.
00435                         FindNextChangeNotification(hEvent); //lint !e534
00436                 }
00437         }
00438         InterlockedCompareExchange(&threadRunning_, 0, 1); //lint !e534
00439 
00440 }// FileSystemWatcher::RunWorker
