00030 /******************************************************************************/
00031 // BuildThunk (0 or 1 parameter version)
00032 //
00033 // Function which builds a thunking piece of code for invoking a method call
00034 //
00035 /******************************************************************************/
00036 template <class _ResultType, class _Ty, class _Param>
00037 _ResultType BuildThunk(_Ty* pType, void (_Ty::*classAddr)(_Param))
00038 {
00039 #pragma pack(1)
00040         struct ThunkTemplate 
00041         {
00042                 BYTE pushInstr_1;
00043                 DWORD_PTR thisPtr;
00044                 BYTE popECX_1;
00045                 BYTE pushInstr_2;
00046                 void (_Ty::*addrPtr)(_Param);
00047                 BYTE retInstr_1;
00048         };
00049 #pragma pack()
00050 
00051         ThunkTemplate* pThunk = reinterpret_cast<ThunkTemplate*>(malloc(sizeof(ThunkTemplate)));
00052         pThunk->pushInstr_1 = 0x68;
00053         pThunk->thisPtr = (DWORD_PTR) pType;
00054         pThunk->popECX_1 = 0x59;
00055         pThunk->pushInstr_2 = 0x68;
00056         pThunk->addrPtr = classAddr;
00057         pThunk->retInstr_1 = 0xC3;
00058 
00059         DWORD dwOldProtect;
00060         VirtualProtect(pThunk, sizeof(ThunkTemplate), PAGE_EXECUTE_WRITECOPY, &dwOldProtect);
00061         return reinterpret_cast<_ResultType>(pThunk);
00062 }

00170 //
00171 /******************************************************************************/
00172 template <class _ResultType>
00173 void FreeThunk(_ResultType pfunc)
00174 {
00175         LPVOID pvFunc = reinterpret_cast<LPVOID>(pfunc);
00176         DWORD dwOldProtect;
00177         VirtualProtect(pvFunc, 20, PAGE_READWRITE, &dwOldProtect);
