00055 // the objects are handling their synchronization in some other fashion, or

00058 /******************************************************************************/
00059 //lint --e{1762} suppress "function could be made const"

00061 {
00062 public:
00063 #if _WIN32_WINNT >= 0x0500
00064         inline bool TryLock() throw() {return true;}
00065 #endif
00066         inline void Lock() throw() {}
00067         inline void Unlock() throw() {}
00068 };

00070 /******************************************************************************/
00071 // PrimitiveLockImpl
00072 //
00073 // This implements a primitive locking object which can be added to the
00074 // data object to provide basic lock/unlock mechanics.  Warning, this lock
00075 // is not capable of being re-entered by the same thread as it doesn't stack
00076 // the unlocks correctly for single-thread re-entry.
00077 //
00078 /******************************************************************************/
00079 class PrimitiveLockImpl
00080 {
00081 // Class data
00082 private:
00083         volatile long lock_;
00084         enum { LOCK_FREE=0 };
00085 
00086 // Constructor
00087 public:
00088         PrimitiveLockImpl() : lock_(LOCK_FREE) {}
00089         PrimitiveLockImpl(const PrimitiveLockImpl&) : lock_(LOCK_FREE) {/* */}
00090         PrimitiveLockImpl& operator=(const PrimitiveLockImpl&) { lock_ = LOCK_FREE; return *this; }
00091         ~PrimitiveLockImpl() {/* */}
00092 
00093         inline bool TryLock() throw() { 
00094                 long lKey = static_cast<long>(GetCurrentThreadId());
00095                 return (::InterlockedCompareExchange(&lock_, lKey, static_cast<long>(LOCK_FREE)) == lKey);
00096         }
00097         inline void Lock() throw() {
00098                 long lKey = static_cast<long>(GetCurrentThreadId());
00099                 while (::InterlockedCompareExchange(&lock_, lKey, static_cast<long>(LOCK_FREE)) != lKey)
00100                         ;
00101         }
00102         inline void Unlock() throw() { ::InterlockedExchange(&lock_, LOCK_FREE); }
00103 };

00245 /******************************************************************************/
00246 // CCSLock
00247 //
00248 // Exception-safe locking class which works with any object exposing 
00249 // Lock/Unlock.  This allows an object to be locked for longer than a single
00250 // function call.
00251 //
00252 /******************************************************************************/
00253 template <class T>
00254 class CCSLock
00255 {
00256 private:
00257         T* p_;
00258         long lockCount_;
00259 public:
00260         CCSLock(const T* pob, bool init_lock=true) throw() : 
00261                 p_(const_cast<T*>(pob)), lockCount_(0) { if (init_lock) Lock(); }
00262 
00263         ~CCSLock() { 
00264                 while (lockCount_ > 0) 
00265                         Unlock(); 
00266                 p_ = NULL; 
00267         }
00268 
00269 #if _WIN32_WINNT >= 0x0500
00270         inline bool TryLock() throw() {
00271                 if (p_) {
00272                         if (p_->TryLock()) {
00273                                 ++lockCount_;
00274                                 return true;
00275                         }
00276                 }
00277                 return false;
00278         }
00279 #endif
00280         inline void Lock() throw() {
00281                 if (p_) {
00282                         ++lockCount_;
00283                         p_->Lock();
00284                 }
00285         }
00286         inline void Unlock() throw() {
00287                 if (p_ && lockCount_ > 0) {
00288                         --lockCount_;
00289                         p_->Unlock();
00290                 }
00291         }
00292 
00293 // Unavailable methods
00294 private:
00295         CCSLock(const CCSLock&);
00296         CCSLock& operator=(const CCSLock&);
00297 };
