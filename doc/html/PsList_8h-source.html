00264                 CloseHandle(hToken);

00372         {

00374                         throw std::runtime_error("Unable to load ToolHelp or PSAPI.DLL");
00375 
00376                 ModuleCollection modColl;
00377 
00378                 // Cannot query pid 0 (system); always returns current process info.
00379                 if (pid == 0)
00380                         return modColl;
00381 
00382                 MODULEENTRY32 me;
00383                 ZeroMemory(&me, sizeof(me));
00384                 me.dwSize = sizeof(MODULEENTRY32);
00385 
00386                 // Enum using Toolhelp32
00387                 if (pfnCreateToolhelp32Snapshot)
00388                 {
00389                         HANDLE htlSnap = (*pfnCreateToolhelp32Snapshot)(TH32CS_SNAPMODULE, pid);
00390                         if (htlSnap == INVALID_HANDLE_VALUE && GetLastError() == ERROR_ACCESS_DENIED)
00391                         {
00392                                 // Adjust our privileges if we can.
00393                                 if (JTI_Internal::AdjustPrivilege(SE_DEBUG_NAME, SE_PRIVILEGE_ENABLED))
00394                                         htlSnap = (*pfnCreateToolhelp32Snapshot)(TH32CS_SNAPMODULE, pid);
00395                         }
00396                         if (htlSnap == INVALID_HANDLE_VALUE ||
00397                                 !(*pfnModule32First)(htlSnap, &me))
00398                                 return modColl;
00399 
00400                         do 
00401                         {
00402                                 TranslateFilename(me.szExePath, me.szExePath);
00403                                 modColl.add(me);
00404 
00405                         } while ((*pfnModule32Next)(htlSnap, &me) == TRUE);
00406                         CloseHandle(htlSnap);

00411                 me.th32ProcessID = pid;
00412                 HANDLE hProcess = ::OpenProcess(PROCESS_QUERY_INFORMATION | PROCESS_VM_READ, FALSE, me.th32ProcessID);

00414                 {

00416                         {
00417                                 hProcess = ::OpenProcess(PROCESS_QUERY_INFORMATION | PROCESS_VM_READ, FALSE, me.th32ProcessID);
00418                                 if (!hProcess)
00419                                         return modColl;
00420                         }
00421                         else

00577                 HMODULE hKernel = GetModuleHandleW(L"KERNEL32.DLL");
00578

00580                 exData.chCommandLine[0] = L'\0';
00581                 exData.chWorkingDir[0] = L'\0';
00582                 exData.pGCWD = (JTI_Internal::PGetCurrentDirectory) GetProcAddress(hKernel, "GetCurrentDirectoryW");

00586                         return false;

00590                 if (!JTI_Internal::AdjustPrivilege(SE_DEBUG_NAME, SE_PRIVILEGE_ENABLED))
00591                         return false;

00594                                 PROCESS_VM_WRITE | PROCESS_VM_READ, FALSE, ProcessID);
00595                 if (hProcess == NULL)
00596                         return false;

00599                 LPVOID lpvFunc = &JTI_Internal::RemoteFunction;
00600                 if (*((LPBYTE)lpvFunc) == 0xE9) // JMP?
00601                 {
00602                         CloseHandle(hProcess);
00603                         return false;
00604                 }
00605 
00606                 bool okToFree = true;
00607 
00608                 // Allocate the memory which will execute our function.
00609                 LPVOID pCode = VirtualAllocEx(hProcess, NULL, JTI_Internal::MAXINJECTSIZE, MEM_COMMIT, PAGE_EXECUTE_READWRITE);

00613                 {

00615                         DWORD dwSizeCode, dwSizeData;
00616                         if (WriteProcessMemory(hProcess, pCode, lpvFunc, JTI_Internal::MAXINJECTSIZE, &dwSizeCode) &&
00617                                 WriteProcessMemory(hProcess, pData, &exData, sizeof(exData), &dwSizeData) &&
00618                                 dwSizeCode == JTI_Internal::MAXINJECTSIZE && dwSizeData == sizeof(exData))
00619                         {
00620                                 // Now execute the code in the process; we create a thread to do the work.
00621                                 DWORD dwThreadId;
00622                                 HANDLE hThread = CreateRemoteThread(hProcess, NULL, 0, 
00623                                                 (LPTHREAD_START_ROUTINE)pCode,pData, 0, &dwThreadId);
00624                                 if (hThread != NULL)
00625                                 {
00626                                         // Wait for the thread to terminate in the process.  If the thread
00627                                         // never returns then don't free the code/data segment as it might cause
00628                                         // the target process to terminate.
00629                                         switch(WaitForSingleObject(hThread, 5000))
00630                                         {
00631                                                 case WAIT_OBJECT_0:
00632                                                         if (ReadProcessMemory(hProcess, pData, &exData, sizeof(exData), 0))
00633                                                         {
00634 #if defined(UNICODE) || defined(_UNICODE)
00635                                                                 commandLine_ = TString(exData.chCommandLine);
00636                                                                 currentWorkingDir_ = TString(exData.chWorkingDir);
00637 #else
00638                                                                 commandLine_ = TString(CW2A<2048>(exData.chCommandLine));
00639                                                                 currentWorkingDir_ = TString(CW2A<MAX_PATH+1>(exData.chWorkingDir));
00640 #endif
00641                                                         }
00642                                                         break;
00643                                                 case WAIT_TIMEOUT:
00644                                                         okToFree = false;
00645                                                         break;
00646                                                 case WAIT_FAILED:
00647                                                 case WAIT_ABANDONED:
00648                                                 default:
00649                                                         break;
00650                                         }
00651                                         CloseHandle(hThread);
00652                                 }
00653                         }
00654                 }
00655                 
00656                 // Cleanup
00657                 if (okToFree)
00658                 {
00659                         if (pCode)
00660                                 VirtualFreeEx(hProcess, pCode, 0, MEM_RELEASE);
00661                         if (pData)
00662                                 VirtualFreeEx(hProcess, pData, 0, MEM_RELEASE);
00663                 }               
00664                 CloseHandle(hProcess);
00665                 FreeLibrary(hKernel);
00666                 return !commandLine_.empty();
00667         }
00668 };

00729 
00730 // Constructor
00731 public:
00732         ProcessEnumerator() : helperInst_(0)
00733         {
00734                 // Look for the TOOLHELP32 functions first; they are more reliable and fully supported.
00735                 helperInst_ = LoadLibrary(_T("KERNEL32.DLL"));
00736                 pfnCreateToolhelp32Snapshot=(PFNCREATETOOLHELP32SNAPSHOT)GetProcAddress(helperInst_,"CreateToolhelp32Snapshot");

00742                 pfnProcess32Next=(PFNPROCESS32NEXT)GetProcAddress(helperInst_,"Process32Next");
00743 #endif
00744         
00745                 // If we couldn't find all the functions we are looking for, drop down to the PSAPI.DLL support.
00746                 if (!pfnCreateToolhelp32Snapshot || !pfnProcess32First || !pfnProcess32Next)
00747                 {
00748                         ::FreeLibrary(helperInst_);

00750                         if (helperInst_ != NULL)
00751                         {
00752                                 pfnEnumProcesses=(PFNENUMPROCESSES)GetProcAddress(helperInst_,"EnumProcesses");
00753                                 pfnEnumProcessModules=(PFNENUMPROCESSMODULES)GetProcAddress(helperInst_,"EnumProcessModules");

00756 #else

00758 #endif

00760                                 {
00761                                         ::FreeLibrary(helperInst_);

00763                                 }
00764                         }
00765                         else
00766                         {
00767                                 pfnEnumProcesses = NULL;
00768                                 pfnEnumProcessModules = NULL;

00770                         }

00780 // Methods
00781 public:
00782         bool CanEnumerateProcesses() const { return helperInst_ != NULL; }
00783 
00784         ProcessCollection Snapshot() const
00785         {
00786                 if (helperInst_ == NULL)
00787                         throw std::runtime_error("Unable to load ToolHelp or PSAPI.DLL");
00788 
00789                 ProcessCollection pColl;
00790                 PROCESSENTRY32 pe;
00791                 ZeroMemory(&pe, sizeof(pe));
00792                 pe.dwSize = sizeof(PROCESSENTRY32);
00793 
00794                 // Enum using Toolhelp32
00795                 if (pfnCreateToolhelp32Snapshot)
00796                 {
00797                         HANDLE htlSnap = (*pfnCreateToolhelp32Snapshot)(TH32CS_SNAPPROCESS, 0);
00798                         if (htlSnap == INVALID_HANDLE_VALUE)
00799                                 return pColl;
00800                         
00801                         // Walk the process list
00802                         if ((*pfnProcess32First)(htlSnap, &pe) == TRUE)
00803                         {
00804                                 do 
00805                                 {
00806                                         pColl.add(pe);
00807                                 } while((*pfnProcess32Next)(htlSnap, &pe) == TRUE);
00808 
00809                         }
00810                         CloseHandle(htlSnap);
00811                         return pColl;
00812                 }
00813 
00814                 // Enum using PSAPI.DLL
00815                 DWORD cbNeeded;
00816                 DWORD arrPids[512];
00817 
00818                 // EnumProcesses returns an array of process IDs
00819                 if(!(*pfnEnumProcesses)(arrPids,sizeofarray(arrPids),&cbNeeded))
00820                         return pColl;
00821 
00822                 int procCount = cbNeeded/sizeof(DWORD);
00823                 for (int i = 0; i < procCount; i++)
00824                 {
00825                         pe.th32ProcessID = arrPids[i];
00826                         InternalGetModuleName(pe.th32ProcessID, pe.szExeFile, sizeofarray(pe.szExeFile));
00827                         pColl.add(pe);
00828                 }
00829                 return pColl;
00830         }

00834         bool InternalGetModuleName(DWORD pid, LPTSTR pszBuff, int nSize) const
00835         {       
00836                 if (pid == 0)
00837                         lstrcpyn(pszBuff,_T("[System Process]"),nSize);
00838                 else
00839                 {
00840                         *pszBuff = _T('\0');
00841 
00842                         // Open the pid and get the first module; this will be the executable.
00843                         HANDLE hProcess = ::OpenProcess(PROCESS_QUERY_INFORMATION | PROCESS_VM_READ, FALSE, pid);
00844                         if (!hProcess)
00845                         {
00846                                 // Adjust our privileges if we can.
00847                                 if (JTI_Internal::AdjustPrivilege(SE_DEBUG_NAME, SE_PRIVILEGE_ENABLED))
00848                                         hProcess = ::OpenProcess(PROCESS_QUERY_INFORMATION | PROCESS_VM_READ, FALSE, pid);
00849                         }
00850 
00851                         if (hProcess)
00852                         {
00853                                 HMODULE hModule;
00854                                 DWORD nModuleCount;
00855                                 bool fRC = false;
00856 
00857                                 (*pfnEnumProcessModules)(hProcess, &hModule, sizeof(hModule), &nModuleCount);
00858                                 if (nModuleCount > 0)

00860                                 CloseHandle(hProcess);
00861                                 if (!fRC)
00862                                 {
00863                                         if (pid < 10)
00864                                                 lstrcpyn(pszBuff, _T("System"), nSize);
00865                                         else
00866                                                 return false;
00867                                 }
00868                         }
00869                         else
00870                                 lstrcpyn(pszBuff, _T("unknown"), nSize);
00871 
00872                         if (*pszBuff)
00873                         {
00874                                 TCHAR* psz = _tcsrchr(pszBuff, _T('\\'));
00875                                 if (psz)
00876                                         lstrcpyn(pszBuff, psz+1, nSize);
00877                         }
00878                 }
00879 
00880                 return true;
00881         }
