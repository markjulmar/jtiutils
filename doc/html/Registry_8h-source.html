00773         // Deletes a tree
00774         void IntDeleteSubKeyTree(HKEY hKeyMain, const TCHAR* pKeyName) const
00775         {
00776                 // Attempt to delete the key directly. Under Win95, this will also delete any branches under it.
00777                 if (RegDeleteKey(hKeyMain, pKeyName) != ERROR_SUCCESS)
00778                 {
00779                         // Open the top-level key.
00780                         HKEY hKey;
00781                         LONG rc = RegOpenKeyEx(hKeyMain, pKeyName, 0, KEY_ENUMERATE_SUB_KEYS | DELETE, &hKey);
00782                         if (rc == ERROR_SUCCESS)
00783                         {
00784                                 TCHAR chBuff[_MAX_PATH]; DWORD dwReqSize;
00785                                 while (rc == ERROR_SUCCESS)
00786                                 {
00787                                         dwReqSize = _MAX_PATH;
00788                                         rc = RegEnumKeyEx(hKey, 0, chBuff, &dwReqSize, NULL, NULL, NULL, NULL);
00789                                         if (rc == ERROR_NO_MORE_ITEMS)
00790                                         {
00791                                                 rc = RegDeleteKey(hKey, pKeyName);
00792                                                 if (rc != ERROR_SUCCESS)
00793                                                         throw RegistryException("RegDeleteKey failed.", rc);
00794                                                 break;
00795                                         }
00796                                         else if (rc == ERROR_SUCCESS)
00797                                                 IntDeleteSubKeyTree(hKey, chBuff);
00798                                         else
00799                                                 throw RegistryException("RegEnumKeyEx failed.", rc);
00800                                 }
00801                                 RegCloseKey(hKey);
00802                         }
00803                         else
00804                                 throw RegistryException("RegOpenKeyEx failed.", rc);
00805                 }
00806         }
