00047 // MRSWLock
00048 //
00049 // This implements a multiple-reader single writer locking object which can be 
00050 // used to protect a resource where read attempts are much more common than 
00051 // write attempts.
00052 //
00053 /******************************************************************************/
00054 class MRSWLock : public LockableObject<MultiThreadModel>

00056 // Internal structures
00057 private:
00058         // Lock State constants
00059         // The lockState_ variable is used to maintain the current status of the
00060         // lock.  It is composed of the following sections:
00061         //
00062         // +----------+-----------+---+----+----+-------+-+
00063         // | WCTR     |  RCTR     | W | WS | RS | CRDR  |R|
00064         // +----------+-----------+---+----+----+-------+-+
00065         // 31         22          12  11   10   9       1 0
00066         //
00067         // WCTR - Write wait count - # of waiting threads for a write.
00068         // RCTR - Read wait count - # of waiting threads for a read.
00069         // W - A write lock is currently in place
00070         // WS - The write event is signaled
00071         // RS - The read event is signaled
00072         // CRDR - Current reader count
00073         // R - A read lock is in place
00074         // 
00075         // The comparable C-structure is:
00076         // 
00077         // struct _LockState
00078         // {
00079         //      unsigned long CurrentReaders : 10;
00080         //      unsigned long ReadSignaled   : 1;
00081         //      unsigned long WriteSignaled  : 1;
00082         //      unsigned long WriteLock      : 1;
00083         //      unsigned long WaitingReaders : 10;
00084         //      unsigned long WaitingWriters : 9;
00085         // };
00086         //
00087         enum { 
00088                 STATE_NONE                = 0x00000000, // No reader/writer lock 
00089                 STATE_READLOCK    = 0x00000001, // Locked for read
00090                 CURRENT_READERS   = 0x000003ff, // Reader count (up to 1023 : 10 bits)
00091                 STATE_READSIGNAL  = 0x00000400, // Reader event is signaled
00092                 STATE_WRITESIGNAL = 0x00000800, // Writer event is signaled
00093                 STATE_WRITELOCK   = 0x00001000, // Has a writer lock
00094                 STATE_READWAIT    = 0x00002000, // Has at least one waiting reader
00095                 WAITREADERS_MASK  = 0x007FE000, // Waiting reader mask (up to 1023 : 10 bits)
00096                 WAITREADERS_SHIFT = 13,                 // Shift mask for readers
00097                 STATE_WRITEWAIT   = 0x00800000, // Has at least one waiting writer
00098                 WAITWRITERS_MASK  = 0xFF800000, // Mask for writers (up to 511 : 9 bits)
00099                 WAITWRITERS_SHIFT = 23                  // Shift mask for writers
00100         };
00101 
00102         enum { 
00103                 MAX_CACHE_ENTRIES = 100,                // Max cached entries
00104                 MAX_WRITER_WAITTIME = 250,              // Maximum time (msec) a writer will wait before readers are blocked
00105         };      
00106 
00107         // This structure is used to cache off the locking type and count for each
00108         // thread interested in this MSRWLock.  It is stored in a container and typed
00109         // by the thread ID.

00112                 int count;                      // reader level lock count
00113                 LockCounter* pNext;     // Next in chain
00114                 LockCounter* pPrev;     // Previous in chain
00115         };

00168         // ReadLock:
00169         //
00170         // This method acquires the read lock - a read lock can be acquired more than
00171         // one time by a single thread - you must release the lock for each time it is 
00172         // acquired.
00173         //
00174     bool ReadLock(DWORD nMillisecondTimeout)
00175         {
00176                 const DWORD nThreadId = GetCurrentThreadId();
00177 
00178                 // If we are already a writer for this lock, then bump our 
00179                 // write counter.
00180                 if (writerID_ == nThreadId)
00181                         return WriteLock(nMillisecondTimeout);
00182 
00183                 // Find or create our lock entry
00184                 LockCounter* pLock = FindLockCounter(nThreadId, true);
00185                 assert(pLock != NULL);
00186 
00187                 // See what state this lock is currently in.  If it has no readers/writers
00188                 // then transition it to a read lock.
00189                 if (InterlockedCompareExchange(&lockState_, STATE_READLOCK, STATE_NONE) != STATE_NONE)
00190                 {
00191                         // Is currently in a read or write state.  See if we are already a reader for
00192                         // this lock.
00193                         if (pLock->count > 0)
00194                         {
00195                         JTI_ASSERT(lockState_ & CURRENT_READERS);
00196                                 ++pLock->count;
00197                                 return true;
00198                         }
00199 
00200                         // The lock has at least one client already; go through the various cases.
00201                         long startState = lockState_, currState;
00202                         unsigned int spinCount = 0;
00203                         do 
00204                         {
00205                                 // Capture the current state
00206                                 currState = startState;
00207 
00208                                 // Collect some numbers
00209                                 bool isReadSignaled = ((currState & STATE_READSIGNAL) > 0);
00210                                 int currentReaderCount = (currState & CURRENT_READERS);
00211                                 int currentReadWaiterCount = ((currState & WAITREADERS_MASK) >> WAITREADERS_SHIFT);
00212                                 int currentWriteWaiterCount = ((currState & WAITWRITERS_MASK) >> WAITWRITERS_SHIFT);
00213                                 DWORD elapsedTime = ELAPSED_TIME(writerTime_);
00214 
00215                                 // If we haven't maxed out our reader count and the read event is signaled,
00216                                 // then allow this thread to go through the lock. 
00217                                 if ((currState < CURRENT_READERS || 
00218                                          (isReadSignaled &&
00219                                           ((currentReaderCount + currentReadWaiterCount) < CURRENT_READERS))) &&
00220                                          (currentWriteWaiterCount == 0 || elapsedTime < MAX_WRITER_WAITTIME))
00221                                 {
00222                                         // Bump the reader count
00223                                         JTI_ASSERT((currState & STATE_WRITELOCK)==0);
00224                                         startState = InterlockedCompareExchange(&lockState_, (currState+STATE_READLOCK), currState);
00225                                         if (startState == currState)
00226                                                 break;
00227                                 }
00228 
00229                                 // Check for too many readers or waiting readers or signaling in progress
00230                                 else if ((currentReaderCount == CURRENT_READERS) ||
00231                                                  ((currState & WAITREADERS_MASK) == WAITREADERS_MASK))
00232                                 {
00233                                         // Put this thread to sleep
00234                                         Sleep(0);
00235                                         spinCount = 0;
00236                                 }
00237 
00238                                 // If the read signal is set, and we have writing waiters, then loop until
00239                                 // it is unset (i.e. all reader threads are released) and then wait on the read
00240                                 // signal.  We want to shift priority to the writer threads in this case.
00241                                 else if (isReadSignaled && currentWriteWaiterCount > 0)
00242                                 {
00243                                         Sleep(0);
00244                                         spinCount = 0;
00245                                 }
00246 
00247                                 // Otherwise add to the number of waiting readers. Contention is a bad thing; 
00248                                 // we will attempt to use a spin counter rather than wait on our event unless 
00249                                 // we spin for too long.  We wait longer on MP boxes as testing shows an 
00250                                 // extremely high contention rate on MP boxes; this alleviates some of the need 
00251                                 // to enter a kernel object.
00252                                 else if (++spinCount > static_cast<unsigned int>(gmaxSpinCount_)) 
00253                                 {
00254                                         // Try to set the state
00255                                         currState = InterlockedCompareExchange(&lockState_, (startState + STATE_READWAIT), startState);
00256                                         if (currState == startState)
00257                                         {
00258                                                 // Wait on the reader.
00259 #ifdef LOCK_STATS
00260                                                 InterlockedIncrement(&readerContentionCount_);
00261 #endif
00262                                                 long modifyState = 0;
00263                                                 DWORD dwStatus = readerEvent_.Wait(nMillisecondTimeout);
00264                                                 if (dwStatus == WAIT_OBJECT_0)
00265                                                 {
00266                                                         JTI_ASSERT((lockState_ & STATE_WRITELOCK) == 0);
00267                                                         JTI_ASSERT(lockState_ & STATE_READSIGNAL);
00268                                                         JTI_ASSERT((lockState_ & CURRENT_READERS) < CURRENT_READERS);
00269                                                         JTI_ASSERT(lockState_ & WAITREADERS_MASK);
00270                                                         modifyState = STATE_READLOCK - STATE_READWAIT;
00271                                                 }
00272                                                 else
00273                                                 {
00274                                                         // Remove us as a read waiter
00275                                                         modifyState = (DWORD) -STATE_READWAIT;
00276                                                         if (dwStatus == WAIT_TIMEOUT)
00277                                                                 dwStatus = ERROR_TIMEOUT;
00278                                                         else if (dwStatus == WAIT_IO_COMPLETION)
00279                                                                 dwStatus = ERROR_IO_INCOMPLETE;
00280                                                         else
00281                                                                 dwStatus = GetLastError();
00282                                                 }
00283 
00284                                                 // One less waiting reader and he may have become a reader
00285                                                 startState = InterlockedExchangeAdd(&lockState_, modifyState);
00286 
00287                                                 // If this thread was signaled, then potentially reset the
00288                                                 // lock to stop future threads from re-entering.  This allows writer
00289                                                 // threads to get through the lock after all existing readers.
00290                                                 if (dwStatus == WAIT_OBJECT_0)
00291                                                 {
00292                                                         JTI_ASSERT((startState & CURRENT_READERS) < CURRENT_READERS);
00293                                                         JTI_ASSERT(startState & STATE_READSIGNAL);
00294 
00295                                                         // If there is only one reading thread left then lower the
00296                                                         // read signal flag and reset the reader event.
00297                                                         if ((startState & WAITREADERS_MASK) == STATE_READWAIT)
00298                                                         {
00299                                                                 // Reset the event and lower reader signaled flag
00300                                                                 readerEvent_.ResetEvent();
00301                                                                 InterlockedExchangeAdd(&lockState_, -STATE_READSIGNAL);
00302                                                         }
00303                                                 }
00304                                                 // Any other status is a failure.
00305                                                 else
00306                                                 {
00307                                                         if (((startState & WAITREADERS_MASK) == STATE_READWAIT) &&
00308                                                             (startState & STATE_READSIGNAL))
00309                                                         {
00310                                                                 // Ensure the event is signaled before resetting it.
00311                                                                 DWORD tResult = readerEvent_.Wait(INFINITE);
00312                                                                 UNREFERENCED_PARAMETER(tResult);
00313                                     JTI_ASSERT(tResult == WAIT_OBJECT_0);
00314                                                                 JTI_ASSERT((lockState_ & CURRENT_READERS) < CURRENT_READERS);
00315                                     
00316                                                                 // Reset the event and lower reader signaled flag
00317                                                                 readerEvent_.ResetEvent();
00318                                                                 InterlockedExchangeAdd(&lockState_, (STATE_READLOCK - STATE_READSIGNAL));
00319 
00320                                                                 // Force the lock to be removed.
00321                                                                 pLock->count = 1;
00322                                                                 ReleaseReadLock();
00323                                                         }
00324                                                         else
00325                                                                 RemoveLockCounter(pLock);
00326 #ifdef _DEBUG
00327                                                         // Failed to acquire lock
00328                                                         DebugBreak();
00329 #endif
00330                                                         return false;
00331                                                 }
00332                                                 JTI_ASSERT(dwStatus == WAIT_OBJECT_0);
00333                                                 break;                        
00334                                         }
00335                                 }
00336                                 CpuPause();
00337                                 startState = lockState_;
00338                         } 
00339                         while(TRUE);
00340                 }
00341                 else
00342                 {
00343                         JTI_ASSERT(pLock->count == 0);
00344                 }
00345 
00346                 // Should not have write lock here.
00347                 JTI_ASSERT(writerID_ == 0 && (lockState_ & STATE_WRITELOCK)==0);
00348 
00349                 // Success
00350 #ifdef LOCK_STATS
00351                 InterlockedIncrement(&readerEntryCount_);
00352 #endif
00353                 pLock->count = 1;
00354                 return true;
00355         }

00358         // WriteLock:
00359         //
00360         // This method acquires the write lock - a write lock can be acquired more than
00361         // one time by a single thread - you must release the lock for each time it is 
00362         // acquired.
00363         //
00364         // In addition, a write lock can be acquired by a thread which currently owns
00365         // a read lock; i.e. we allow conversions from a read to a write.
00366         //
00367     bool WriteLock(DWORD nMillisecondTimeout)
00368         {
00369                 const DWORD nThreadId = GetCurrentThreadId();
00370 
00371                 // Check if the current thread already has writer lock.
00372                 if (writerID_ == nThreadId)
00373                 {
00374                         ++writerCount_;
00375                         return true;
00376                 }
00377 
00378                 // If this thread already has a read lock, then we must release it in order to gain a
00379                 // write lock.  We don't want to lose that information however.
00380                 int readCount = 0;
00381                 LockCounter* pLock = FindLockCounter(nThreadId, false);
00382                 if (pLock != NULL)
00383                 {
00384                         // Remove the lock information from the queue.
00385                         readCount = pLock->count;
00386                         pLock->count = 1;
00387                         ReleaseReadLock();
00388                 }
00389 
00390                 // If we have readers or writers then begin our wait loop.
00391                 if (InterlockedCompareExchange(&lockState_, STATE_WRITELOCK, STATE_NONE) != STATE_NONE)
00392                 {
00393                         // Initialize
00394                         long startState = lockState_, currState;
00395                         int spinCount = 0;
00396                         do
00397                         {
00398                                 currState = startState;
00399 
00400                                 // We don't need to wait if there are no readers/writers.  We don't care if
00401                                 // there are waiting readers in this case.
00402                                 if ((currState == STATE_NONE) || (currState & ~(STATE_WRITESIGNAL | WAITREADERS_MASK)) == 0)
00403                                 {
00404                                         // Can be a writer
00405                                         startState = InterlockedCompareExchange(&lockState_, (currState + STATE_WRITELOCK), currState);
00406                                         if (startState == currState)
00407                                                 break;
00408                                 }
00409 
00410                                 // Check for too many waiting writers
00411                                 else if (((currState & WAITWRITERS_MASK) == WAITWRITERS_MASK))
00412                                 {
00413                                         Sleep(0);
00414                                         spinCount = 0;
00415                                 }
00416 
00417                                 // Contention is a bad thing; we will attempt to use a spin counter rather than wait on 
00418                                 // our event unless we spin for too long.  We wait longer on MP boxes as testing shows an 
00419                                 // extremely high contention rate on MP boxes; this alleviates some of the need to enter a 
00420                                 // kernel object.
00421                                 else if (++spinCount > gmaxSpinCount_)
00422                                 {
00423                                         // We need to wait for the write event.  Add to waiting writers
00424                                         startState = InterlockedCompareExchange(&lockState_, (currState + STATE_WRITEWAIT), currState);
00425                                         if (startState == currState)
00426                                         {
00427                                                 // Save off the time the first thread started waiting; we will begin 
00428                                                 // blocking reader threads if a threshold is exceeded.
00429                                                 if (((startState & WAITWRITERS_MASK) >> WAITWRITERS_SHIFT) == 1)
00430                                                         writerTime_ = GetTickCount();
00431 
00432                                                 // We are now officially waiting - block on the wait event.
00433 #ifdef LOCK_STATS
00434                                                 InterlockedIncrement(&writerContentionCount_);
00435 #endif
00436                                                 long modifyState = 0;
00437                                                 DWORD dwStatus = writerEvent_.Wait(nMillisecondTimeout);
00438                                                 if (dwStatus == WAIT_OBJECT_0) {
00439 #ifdef LOCK_STATS
00440                                                         DWORD elapsed = ELAPSED_TIME(writerTime_);
00441                                                         if (elapsed > maxWriterWaitTime_)
00442                                                                 maxWriterWaitTime_ = elapsed;
00443 #endif
00444                                                         writerTime_ = GetTickCount();
00445                                                         modifyState = STATE_WRITELOCK - STATE_WRITEWAIT - STATE_WRITESIGNAL;
00446                                                 }
00447                                                 else
00448                                                 {
00449                                                         modifyState = -STATE_WRITEWAIT;
00450                                                         if (dwStatus == WAIT_TIMEOUT)
00451                                                                 dwStatus = ERROR_TIMEOUT;
00452                                                         else if (dwStatus == WAIT_IO_COMPLETION)
00453                                                                 dwStatus = ERROR_IO_INCOMPLETE;
00454                                                         else
00455                                                                 dwStatus = GetLastError();
00456                                                 }
00457 
00458                                                 // Remove this thread from the waiting list.
00459                                                 startState = InterlockedExchangeAdd(&lockState_, modifyState);
00460 
00461                                                 // If the write event did not get signaled, then perform error cleanup.
00462                                                 if (dwStatus != WAIT_OBJECT_0)
00463                                                 {
00464                                                         // If we have maxed out our waiting threads for the write then
00465                                                         // loop waiting for all the existing writers to clear out.
00466                                                         if ((startState & STATE_WRITESIGNAL) && 
00467                                                                 ((startState & WAITWRITERS_MASK) == STATE_WRITEWAIT))
00468                                                         {
00469                                                                 do
00470                                                                 {
00471                                                                         // If the write event is signaled and we have no waiters
00472                                                                         // then wait on the event until we get it.
00473                                                                         startState = lockState_;
00474                                                                         if ((startState & STATE_WRITESIGNAL) &&
00475                                                                                 ((startState & WAITWRITERS_MASK) == 0))
00476                                                                         {
00477                                                                                 DWORD dwTemp = writerEvent_.Wait(100);
00478                                                                                 if (dwTemp == WAIT_OBJECT_0)
00479                                                                                 {
00480                                                                                         // Remove the signaling bit
00481                                                                                         startState = InterlockedExchangeAdd(&lockState_, (STATE_WRITELOCK - STATE_WRITESIGNAL));
00482 
00483                                                                                         // Reset to the orginal status
00484                                                                                         JTI_ASSERT(writerID_ == 0);
00485                                                                                         JTI_ASSERT(writerCount_ == 0);
00486                                                                                         writerID_ = nThreadId;
00487                                                                                         writerCount_ = 1;
00488 
00489                                                                                         // Release the lock
00490                                                                                         ReleaseWriteLock();
00491                                                                                         break;
00492                                                                                 }
00493                                                                         }
00494                                                                         else
00495                                                                                 break;
00496                                                                 }
00497                                                                 while(TRUE);
00498                                                         }
00499 #ifdef _DEBUG
00500                                                         DebugBreak();
00501 #endif                                  
00502                                                         return false;
00503                                                 }
00504                                                 break;
00505                                         }
00506                                 }
00507                                 CpuPause();
00508                                 startState = lockState_;
00509                         } 
00510                         while(TRUE);
00511                 }
00512 
00513                 // Save new writer information
00514                 JTI_ASSERT(writerID_ == 0);
00515                 JTI_ASSERT(writerCount_ == 0);
00516                 writerID_ = nThreadId;
00517                 writerCount_ = 1;
00518 #ifdef LOCK_STATS
00519                 InterlockedIncrement(&writerEntryCount_);
00520 #endif
00521                 // If we originally held a read lock, then re-add the counter to the list.
00522                 if (readCount > 0)
00523                 {
00524                         pLock = FindLockCounter(nThreadId, true);
00525                         pLock->count = readCount;
00526                 }
00527 
00528                 return true;
00529         }

00532         // ReleaseReadLock:
00533         //
00534         // This method releases a lock for the given thread.
00535         //
00536     void ReleaseReadLock()
00537         {
00538                 const DWORD nThreadId = GetCurrentThreadId();
00539 
00540                 // Check if the thread has writer lock
00541                 if (writerID_ == nThreadId)
00542                 {
00543                         ReleaseWriteLock();
00544                         return;
00545                 }
00546 
00547                 // Find the read count
00548                 LockCounter* pLock = FindLockCounter(nThreadId);
00549                 if (pLock != NULL)
00550                 {
00551                         // If this thread no longer holds a lock then see if we 
00552                         // need to release writers.
00553                         if (--pLock->count == 0)
00554                         {
00555                 JTI_ASSERT((lockState_ & STATE_WRITELOCK) == 0);
00556                 JTI_ASSERT(lockState_ & CURRENT_READERS);
00557 
00558                                 bool fLastReader = false;
00559                                 long startState = lockState_, currState, modifyState;
00560                                 do
00561                                 {
00562                                         currState = startState;
00563                                         modifyState = -STATE_READLOCK;
00564                                         fLastReader = ((currState & (STATE_READSIGNAL | CURRENT_READERS)) == STATE_READLOCK);
00565                                         if (fLastReader)
00566                                         {
00567                                                 if (currState & WAITWRITERS_MASK)
00568                                                         modifyState += STATE_WRITESIGNAL;
00569                                                 else if (currState & WAITREADERS_MASK)
00570                                                         modifyState += STATE_READSIGNAL;
00571                                         }
00572 
00573                     JTI_ASSERT((currState & STATE_WRITELOCK) == 0);
00574                     JTI_ASSERT(currState & CURRENT_READERS);
00575 
00576                                         startState = InterlockedCompareExchange(&lockState_, (currState + modifyState), currState);
00577                                 } 
00578                                 while(currState != startState);
00579 
00580                                 // Check for last reader
00581                                 if (fLastReader)
00582                                 {
00583                                         // Check for waiting writers
00584                                         if (currState & WAITWRITERS_MASK)
00585                                         {
00586                         JTI_ASSERT((lockState_ & (STATE_READSIGNAL|STATE_WRITESIGNAL)) == STATE_WRITESIGNAL);
00587                                                 writerEvent_.SetEvent();
00588                                         }
00589 
00590                                         // Check for waiting readers
00591                                         else if (currState & WAITREADERS_MASK)
00592                                         {
00593                         JTI_ASSERT((lockState_ & (STATE_READSIGNAL|STATE_WRITESIGNAL)) == STATE_READSIGNAL);
00594                                                 readerEvent_.SetEvent();
00595                                         }
00596                                 }
00597 
00598                                 // Delete our entry
00599                                 RemoveLockCounter(pLock);
00600                         }
00601                 }
00602                 else
00603                 {
00604 #ifdef _DEBUG
00605                         DebugBreak();
00606 #endif
00607                         throw std::logic_error("Failed to locate lockCounter structure");
00608                 }
00609         }

00612         // ReleaseWriteLock
00613         //
00614         // This method releases the write lock for the owning thread.
00615         //
00616     void ReleaseWriteLock()
00617         {
00618                 const DWORD nThreadId = GetCurrentThreadId();
00619 
00620                 // Check validity of caller
00621                 if (writerID_ == nThreadId)
00622                 {
00623                         // Check for nested release
00624                         if (--writerCount_ == 0)
00625                         {
00626                                 long currState, startState, modifyState;
00627 
00628                                 // Not a writer any more
00629                                 writerID_ = 0;
00630                                 startState = lockState_;
00631                                 do
00632                                 {
00633                                         currState = startState;
00634                                         modifyState = -STATE_WRITELOCK;
00635                                         if (currState & WAITREADERS_MASK)
00636                                                 modifyState += STATE_READSIGNAL;
00637                                         else if (currState & WAITWRITERS_MASK)
00638                                                 modifyState += STATE_WRITESIGNAL;
00639                                         startState = InterlockedCompareExchange(&lockState_, (currState + modifyState), currState);
00640                                 } 
00641                                 while (currState != startState);
00642 
00643                                 // Check for waiting writers/readers
00644                                 if (modifyState & STATE_WRITESIGNAL)
00645                                 {
00646                     JTI_ASSERT((lockState_ & (STATE_READSIGNAL|STATE_WRITESIGNAL)) == STATE_WRITESIGNAL);
00647                                         writerEvent_.SetEvent();
00648                                 }
00649                                 else if (modifyState & STATE_READSIGNAL)
00650                                 {
00651                     JTI_ASSERT((lockState_ & (STATE_READSIGNAL|STATE_WRITESIGNAL)) == STATE_READSIGNAL);
00652                                         readerEvent_.SetEvent();
00653                                 }
00654 
00655                                 // If this thread originally held a read-lock, then reacquire the read lock.
00656                                 LockCounter* pLock = FindLockCounter(nThreadId);
00657                                 if (pLock != NULL)
00658                                 {
00659                                         // Retrieve the original lock counter
00660                                         int readCount = pLock->count;
00661 
00662                                         // Remove it
00663                                         pLock->count = 0;
00664                                         RemoveLockCounter(pLock);
00665                                         
00666                                         // Reacquire the read lock - we wait forever.
00667                                         if (ReadLock(INFINITE))
00668                                         {
00669                                                 // Reset the lock counter
00670                                                 pLock = FindLockCounter(nThreadId);
00671                                                 pLock->count = readCount;
00672                                         }
00673                                 }
00674                         }
00675                 }
00676                 else
00677                 {
00678 #ifdef _DEBUG
00679                         DebugBreak();
00680 #endif
00681                 }
00682         }
00683

00685         inline void ResetStatistics() { readerEntryCount_ = readerContentionCount_ = writerEntryCount_ = writerContentionCount_ = 0; maxWriterWaitTime_ = 0; }
00686     inline DWORD GetReaderEntryCount() const { return readerEntryCount_; }

00688     inline DWORD GetWriterEntryCount() const { return writerEntryCount_; }

00696         // AllocLockCounter
00697         // Allocates a new lock counter from our heap
00698         inline LockCounter* AllocLockCounter(DWORD nThreadId=0)
00699         {
00700                 JTI_ASSERT(heapMem_ != 0);
00701                 LockCounter* pLock = reinterpret_cast<LockCounter*>(HeapAlloc(heapMem_, HEAP_NO_SERIALIZE | HEAP_ZERO_MEMORY, sizeof(LockCounter)));
00702                 if (pLock == NULL)
00703                         throw jtiexception(E_OUTOFMEMORY, "Out of memory.");
00704                 pLock->threadId = nThreadId;
00705                 return pLock;
00706         }
00707 
00709         // FreeLockCounter
00710         // Returns a lock counter to our heap
00711         inline void FreeLockCounter(LockCounter* pLock)
00712         {
00713                 BOOL rc = HeapFree(heapMem_, HEAP_NO_SERIALIZE, pLock);
00714                 JTI_ASSERT(rc == TRUE);
00715                 UNREFERENCED_PARAMETER(rc);
00716         }
00717 
00719         // CreateCacheInstances
00720         // Pre-fills our free list with cache instances
00721         inline void CreateCacheInstances(int initialCache)
00722         {
00723                 JTI_ASSERT(heapMem_ == 0);
00724                 if (initialCache > MAX_CACHE_ENTRIES)
00725                         initialCache = MAX_CACHE_ENTRIES;
00726                 heapMem_ = HeapCreate(HEAP_NO_SERIALIZE, max(initialCache,2) * sizeof(LockCounter), 0);
00727                 if (heapMem_ == NULL)
00728                         throw jtiexception(E_OUTOFMEMORY, "Out of memory.");
00729                 for (freeCount_ = 0; freeCount_ < initialCache; ++freeCount_)
00730                 {
00731                         LockCounter* p = AllocLockCounter();
00732                         p->pNext = currFree_;
00733                         if (currFree_ != NULL)
00734                                 currFree_->pPrev = p;
00735                         currFree_ = p;
00736                 }
00737         }
00738 
00740         // CreateNewLockCounter
00741         // This method creates or retrieves a cached lock counter
00742         inline LockCounter* CreateNewLockCounter(DWORD nThreadId)
00743         {
00744                 if (currFree_ == NULL)
00745                         return AllocLockCounter(nThreadId);
00746 
00747                 --freeCount_;
00748                 JTI_ASSERT(freeCount_ >= 0);
00749 
00750                 LockCounter* p = currFree_;
00751                 currFree_ = p->pNext;
00752                 if (currFree_)
00753                         currFree_->pPrev = NULL;
00754 
00755                 p->count = 0;
00756                 p->pNext = p->pPrev = NULL;
00757                 p->threadId = nThreadId;
00758                 return p;
00759         }
00760 
00762         // FindLockCounter
00763         // This method locates an existing LockCounter object for a given threadID 
00764         // from our maintained list.
00765         inline LockCounter* FindLockCounter(DWORD nThreadId, bool add = false)
00766         {
00767                 CCSLock<MRSWLock> _lockGuard(this);
00768                 LockCounter* pCurr = currReaders_;
00769                 while (pCurr) {
00770                         if (pCurr->threadId == nThreadId)
00771                                 break;
00772                         pCurr = pCurr->pNext;
00773                 }
00774                 if (add && pCurr == NULL) {
00775                         pCurr = CreateNewLockCounter(nThreadId);
00776                         pCurr->pNext = currReaders_;
00777                         if (currReaders_ != NULL)
00778                                 currReaders_->pPrev = pCurr;
00779                         currReaders_ = pCurr;
00780                 }
00781                 return pCurr;
00782         }
00783 
00785         // RemoveLockCounter
00786         // This method removes the specified lock counter from our maintained list.
00787         inline void RemoveLockCounter(LockCounter* p)
00788         {
00789                 CCSLock<MRSWLock> _lockGuard(this);
00790                 if (p->pNext != NULL)
00791                 {
00792                         JTI_ASSERT(p->pNext->pPrev == p);
00793                         p->pNext->pPrev = p->pPrev;
00794                 }
00795                 if (p->pPrev != NULL)
00796                 {
00797                         JTI_ASSERT(p->pPrev->pNext == p);
00798                         p->pPrev->pNext = p->pNext;
00799                 }
00800                 if (p == currReaders_)
00801                 {
00802                         JTI_ASSERT(p->pPrev == NULL);
00803                         currReaders_ = p->pNext;
00804                 }
00805                 if (freeCount_ >= MAX_CACHE_ENTRIES)
00806                         FreeLockCounter(p);
00807                 else
00808                 {
00809                         ++freeCount_;
00810                         p->pPrev = NULL;
00811                         p->pNext = currFree_;
00812                         if (currFree_ != NULL)
00813                                 currFree_->pPrev = p;
00814                         currFree_ = p;
00815                 }
00816         }

00819         // CpuPause
00820         // This method implements the PAUSE instruction on the Intel platform.
00821         // This causes the CPU to drop down to the bus speed for this cycle and
00822         // therefore reduces power consumption without releasing the timeslice
00823         // from the OS perspective.
00824         inline void CpuPause()
00825         {
00826                 // Single processor; give up our timeslice.
00827                 if (gmaxSpinCount_ == 1)
00828                         Sleep(0);
00829                 else
00830 #ifdef _M_IX86
00831                 __asm 
00832                 {
00833                         REP NOP
00834                 }
00835 #endif
00836         }
