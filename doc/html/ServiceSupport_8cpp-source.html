00150 /**************************************************************************************
00151 ** Procedure: ServiceBase::Install
00152 **
00153 ** Arguments: void
00154 **
00155 ** Returns: TRUE/FALSE whether service is installed in the SCM.
00156 **
00157 ** Description: This installs the service into the SCM.
00158 **
00159 **************************************************************************************/
00160 DWORD ServiceBase::Install(const TCHAR* pszUserID, const TCHAR* pszPassword, bool fAutoStart, const TCHAR* pszDepends, LPCTSTR pszDescription) const
00161 {
00162         DWORD lastError = 0;
00163 
00164         // Open the Service Control Manager
00165         bool fInstalled = false;
00166         SC_HANDLE hSCM = OpenSCManager(NULL, NULL, SC_MANAGER_ALL_ACCESS);
00167     if (hSCM != NULL)
00168         {
00169                 // Get the executable file path
00170                 TCHAR szFilePath[_MAX_PATH];
00171                 GetModuleFileName(NULL, szFilePath, _MAX_PATH); //lint !e534
00172 
00173                 // If the service is already installed, then just change it's configuration
00174                 SC_HANDLE hService = NULL;
00175 
00176                 if (IsInstalled())
00177                 {
00178                         hService = OpenService(hSCM, ServiceName().c_str(), SERVICE_ALL_ACCESS);
00179                         if (!ChangeServiceConfig(hService, SERVICE_WIN32_OWN_PROCESS,
00180                                 (fAutoStart) ? SERVICE_AUTO_START :     SERVICE_DEMAND_START, 
00181                                 SERVICE_ERROR_NORMAL, szFilePath,
00182                                 NULL, NULL,     pszDepends, pszUserID, pszPassword, DisplayName().c_str()))
00183                         {
00184                                 lastError = GetLastError();
00185                                 CloseServiceHandle(hService); //lint !e534
00186                                 hService = NULL;
00187                         }
00188                 }
00189                 else
00190                 {
00191                         hService = CreateService(hSCM, ServiceName().c_str(), DisplayName().c_str(), 
00192                                 SERVICE_ALL_ACCESS, SERVICE_WIN32_OWN_PROCESS,
00193                                 (fAutoStart) ? SERVICE_AUTO_START :     SERVICE_DEMAND_START, SERVICE_ERROR_NORMAL,     szFilePath, 
00194                                 NULL, NULL,     pszDepends, pszUserID, pszPassword);
00195                 }
00196 
00197                 if (hService != NULL)
00198                 {
00199                         // If a description is given, then set that in place.
00200                         if (pszDescription != NULL)
00201                         {
00202                                 HMODULE hAdvApi = LoadLibraryA("ADVAPI32.DLL");
00203                                 if (hAdvApi != NULL)
00204                                 {
00205 #ifdef _UNICODE
00206                                         CHANGESERVICECONFIG2 pcsc = (CHANGESERVICECONFIG2)GetProcAddress(hAdvApi, "ChangeServiceConfig2W");
00207 #else
00208                                         CHANGESERVICECONFIG2 pcsc = (CHANGESERVICECONFIG2)GetProcAddress(hAdvApi, "ChangeServiceConfig2A");
00209 #endif
00210                                         if (pcsc != NULL)
00211                                         {
00212                                                 SERVICE_DESCRIPTION sdBuf;
00213                                                 sdBuf.lpDescription = const_cast<LPTSTR>(pszDescription);
00214                                                 if (!pcsc(hService, SERVICE_CONFIG_DESCRIPTION, &sdBuf))
00215                                                         lastError = GetLastError();
00216                                         }
00217                                         FreeLibrary(hAdvApi);
00218                                 }
00219                         }
00220                         CloseServiceHandle(hService); //lint !e534
00221                         fInstalled = true;
00222                 }
00223                 else if (lastError == 0)
00224                         lastError = GetLastError();
00225                 CloseServiceHandle(hSCM); //lint !e534
00226         }
00227 
00228         if (fInstalled)
00229         {
00230                 if (!AddEventLogSupport())
00231                         lastError = GetLastError();
00232                 else
00233                         lastError = 0;
00234         }
00235 
00236         return lastError;
00237 
00238 }// ServiceBase::Install

00286 /**************************************************************************************
00287 ** Procedure: ServiceBase::Uninstall
00288 **
00289 ** Arguments: void
00290 **
00291 ** Returns: TRUE/FALSE whether service is uninstalled from the SCM.
00292 **
00293 ** Description: This removes the service from the SCM list.
00294 **
00295 **************************************************************************************/
00296 DWORD ServiceBase::Uninstall() const
00297 {
00298         DWORD lastError = 0;
00299 
00300         // If it is not currently installed, then we are done.
00301         if (!IsInstalled())
00302                 return lastError;
00303 
00304         // Open the service and ask it to shutdown if it is running.
00305         bool fDeleted = false;
00306         SC_HANDLE hSCM = OpenSCManager(NULL, NULL, SC_MANAGER_ALL_ACCESS);
00307     if (hSCM != NULL)
00308         {
00309                 SC_HANDLE hService = OpenService(hSCM, ServiceName().c_str(), SERVICE_STOP | DELETE);
00310                 if (hService != NULL)
00311                 {
00312                         // Ask the service to stop running.
00313                         SERVICE_STATUS status;
00314                         if (!ControlService(hService, SERVICE_CONTROL_STOP, &status))
00315                                 lastError = GetLastError();
00316 
00317                         // Now delete the service.
00318                         fDeleted = (DeleteService(hService) == TRUE);
00319                         if (fDeleted)
00320                                 RemoveApplicationEventLog();
00321                         else
00322                                 lastError = GetLastError();
00323 
00324                         CloseServiceHandle(hService); //lint !e534
00325                 }
00326                 CloseServiceHandle(hSCM); //lint !e534
00327         }
00328         
00329         if (fDeleted)
00330                 lastError = 0;
00331 
00332         return lastError;
00333 
00334 }// ServiceBase::Uninstall

00816 **
00817 ** Arguments: pszServiceName - name of the service to start
00818 **
00819 ** Returns: Win32 error code
00820 **
00821 ** Description: This method starts the named service
00822 **
00823 **************************************************************************************/
00824 DWORD ServiceBase::StartService(LPCTSTR pszServiceName, DWORD dwMaxWaitTime)
00825 {
00826         bool fStarted = false;
00827         DWORD lastErr = 0;
00828 
00829         SC_HANDLE hSCM = OpenSCManager(NULL, NULL, SC_MANAGER_ALL_ACCESS);
00830     if (hSCM != NULL)
00831         {
00832                 SC_HANDLE hService = OpenService(hSCM, pszServiceName, SERVICE_ALL_ACCESS);
00833                 if (hService == NULL)
00834                         lastErr = GetLastError();
00835                 else
00836                 {
00837                         if (::StartService(hService, 0, NULL))
00838                         {
00839                                 SERVICE_STATUS sStatus;
00840                                 if (QueryServiceStatus(hService, &sStatus))
00841                                 {
00842                                         DWORD dwStartTickCount = GetTickCount();
00843                                         DWORD dwOldCheckPoint = sStatus.dwCheckPoint;
00844 
00845                                         while (sStatus.dwCurrentState == SERVICE_START_PENDING)
00846                                         {
00847                                                 DWORD dwWaitTime = sStatus.dwWaitHint / 10;
00848                                                 if (dwWaitTime < 1000)
00849                                                         dwWaitTime = 1000;
00850                                                 else if (dwWaitTime > dwMaxWaitTime)
00851                                                         dwWaitTime = dwMaxWaitTime;
00852 
00853                                                 Sleep(dwWaitTime);
00854 
00855                                                 // Check the status again. 
00856                                                 if (!QueryServiceStatus(hService, &sStatus))
00857                                                         break;
00858 
00859                                                 if (sStatus.dwCheckPoint > dwOldCheckPoint)
00860                                                 {
00861                                                         dwStartTickCount = GetTickCount();
00862                                                         dwOldCheckPoint = sStatus.dwCheckPoint;
00863                                                 }
00864                                                 else if (ELAPSED_TIME(dwStartTickCount) > sStatus.dwWaitHint)
00865                                                         break;
00866                                         }
00867                                         fStarted = (sStatus.dwCurrentState == SERVICE_RUNNING);
00868                                 }
00869                                 else
00870                                         lastErr = GetLastError();
00871                         }
00872                         else
00873                                 lastErr = GetLastError();
00874 
00875                         CloseServiceHandle(hService);   //lint !e534
00876                 }
00877                 CloseServiceHandle(hSCM); //lint !e534
00878         }
00879         else
00880                 lastErr = GetLastError();
00881 
00882         if (fStarted == true)
00883                 lastErr = 0;
00884         else if (lastErr == 0)
