00030 // Implementation class for the binary stream.
00031 //
00032 /******************************************************************************/
00033 class memstream : public binstream
00034 {
00035 // Class data
00036 private:
00037         static const unsigned int SIZE_INC = 4096;
00038         BYTE *dataBuff_, *currRead_, *currWrite_;
00039         unsigned int buffSize_;
00040 
00041 // Constructor/Destructor
00042 public:
00043         memstream(const void* pData, unsigned int nSize) : buffSize_(nSize), dataBuff_(0), currRead_(0), currWrite_(0) {
00044                 currRead_ = currWrite_ = dataBuff_ = reinterpret_cast<BYTE*>(realloc(dataBuff_,nSize));
00045                 memcpy(dataBuff_, pData, nSize);
00046         }
00047         memstream() : buffSize_(SIZE_INC), currRead_(0), currWrite_(0), dataBuff_(0) { 
00048                 currRead_ = currWrite_ = dataBuff_ = reinterpret_cast<BYTE*>(realloc(dataBuff_,buffSize_));
00049                 setbit(binstream::eofbit);
00050         }
00051         virtual ~memstream() { realloc(dataBuff_,0); }
00052 
00053 // Overridable operations required in the derived classes
00054 public:
00055         virtual bool skip(int sz) { 
00056                 currRead_ += sz;
00057                 clrbit(binstream::eofbit);
00058                 if (currRead_ < dataBuff_)
00059                         currRead_ = dataBuff_;
00060                 else if (currRead_ >= (dataBuff_ + size()))
00061                 {
00062                         currRead_ = (dataBuff_ + size());
00063                         setbit(binstream::eofbit);
00064                 }
00065                 return true;
00066         }
00067 
00068         virtual bool peek(void* pBuff, unsigned int sz) const {
00069                 if (!currRead_) 
00070                         return false;
00071                 if ( (currRead_ + sz) > (dataBuff_ + size()) )
00072                         return false;
00073                 memcpy(pBuff,currRead_,sz);
00074                 return true;
00075         }
00076 
00077         virtual bool read(void* pBuff, unsigned int sz) {
00078                 if (peek(pBuff,sz)) {
00079                         currRead_ += sz;
00080                         if (currRead_ >= (dataBuff_+size()))
00081                                 setbit(binstream::eofbit);
00082                         return true;
00083                 }
00084                 return false;
00085         }
00086 
00087         virtual bool write(const void* pBuff, unsigned int size) {
00088                 if ( (currWrite_ + size) > (dataBuff_ + buffSize_) )
00089                 {
00090                         unsigned int nDiffR = static_cast<unsigned int>(currRead_ - dataBuff_);
00091                         unsigned int nDiffW = static_cast<unsigned int>(currWrite_ - dataBuff_);
00092                         dataBuff_ = reinterpret_cast<BYTE*>(realloc(dataBuff_, buffSize_+SIZE_INC));
00093                         buffSize_ += SIZE_INC;
00094                         currWrite_ = dataBuff_ + nDiffW;
00095                         currRead_ = dataBuff_ + nDiffR;
00096                 }
00097                 memcpy(currWrite_, pBuff, size);
00098                 currWrite_ += size;
00099                 clrbit(binstream::eofbit);
00100                 return true;
00101         }
00102 
00103 // Access methods
00104 public:
00105         void* get() const { return dataBuff_; }
00106         unsigned long size() const { 
00107                 if (currWrite_ > dataBuff_)
00108                         return static_cast<unsigned long>(currWrite_ - dataBuff_);
00109                 return buffSize_;
00110         }
00111 };
