00031 /****************************************************************************/
00032 // ParamValue
00033 //
00034 // This class describes a single parameter on the command line.
00035 //
00036 /****************************************************************************/
00037 class ParamValue
00038 {
00039 // Class data
00040 private:
00041         const TCHAR* param_;
00042         const TCHAR* value_;
00043 
00044 // Constructor
00045 public:
00046         ParamValue() : param_(0), value_(0) {/* */}
00047         ParamValue(const ParamValue &p) : param_(p.param_), value_(p.value_) {/* */}
00048         ParamValue(const TCHAR* param) : param_(param), value_(0) {/* */}
00049         ParamValue(const TCHAR* param, const TCHAR* value) : param_(param), value_(value) {/* */}
00050 
00051 // Properties
00052 public:
00053         __declspec(property(get=get_Name)) const TCHAR* Name;
00054         __declspec(property(get=get_Value)) const TCHAR* Value;
00055         __declspec(property(get=get_HasValue)) bool HasValue;
00056 
00057 // Accessors
00058 public:
00059         const TCHAR* get_Name() const { return param_; }
00060         bool get_HasValue() const { return (value_ != NULL); }
00061         const TCHAR* get_Value() const { return (value_) ? value_ : _T(""); }
00062 
00063 // Converters
00064 public:
00065         float ToFloat()
00066         {
00067                 TCHAR* err;
00068                 float x = static_cast<float>(_tcstod(value_, &err));
00069                 return x;
00070         }
00071 
00072         int ToInt32()
00073         {
00074                 TCHAR* err;
00075                 int x = static_cast<int>(_tcstol(value_, &err, 10));
00076                 return x;
00077         }
00078 
00079         DWORD GetHashCode() const { return ParamValue::GetHashCode(param_); }
00080 
00081 // Hash function for faster lookups (switch)
00082 public:
00083         static DWORD GetHashCode(const TCHAR* pszText_)
00084         {
00085                 char* pszText;
00086 #ifdef UNICODE
00087                 int size = WideCharToMultiByte(CP_ACP, 0, pszText_, -1, 0, 0, NULL, NULL);
00088                 if (size == 0)
00089                         return 0;
00090                 pszText = reinterpret_cast<char*>(_alloca(size+1));
00091                 WideCharToMultiByte(CP_ACP,0,pszText_,-1,pszText,size,NULL,NULL);
00092 #else
00093                 pszText = reinterpret_cast<char*>(_alloca(lstrlen(pszText_)+1));
00094                 lstrcpy(pszText,pszText_);
00095 #endif
00096                 strlwr(pszText); // Force lowercase
00097 
00098                 const BYTE* b = reinterpret_cast<const BYTE*>(pszText);
00099                 int len = lstrlenA(pszText);
00100                 if (len >= 4)
00101                         return b[0]+(b[1] << 12) + (b[2] << 6) + (b[3] << 18) + (b[len-1] << 3) + len;
00102                 else if (len >= 1)
00103                         return b[0]+(b[len-1] << 4)+len;
00104                 return 0;
00105         }
00106 };

00108 /****************************************************************************/
00109 // CommandLineParser
00110 //
00111 // This class implements a simple command line parser
00112 //
00113 /****************************************************************************/
00114 class CommandLineParser
00115 {
00116 // Class data
00117 public:
00118         typedef std::list<ParamValue> ParamList;
00119         typedef ParamList::iterator iterator;
00120 private:
00121         const TCHAR* m_Program;
00122         ParamList m_lstParams;
00123 
00124 // Constructor
00125 public:
00126     CommandLineParser(int argc, TCHAR* argv[])
00127         {
00128                 m_lstParams.clear();
00129                 AddArgs(argc, argv);
00130         }
00131 
00132 // Search function
00133 public:
00134         iterator find(const TCHAR *param, bool caseSensitive=false)
00135         {
00136                 ParamList::iterator it;
00137                 for (it = m_lstParams.begin(); it != m_lstParams.end(); ++it)
00138                 {
00139                         if ( (!caseSensitive && !_tcsicmp(it->Name, param)) ||
00140                                  (caseSensitive && !_tcscmp(it->Name, param)) )
00141                                  return it;
00142                 }
00143                 return m_lstParams.end();
00144         }
00145 
00146 // Properties
00147 public:
00148         __declspec(property(get=get_Program)) const TCHAR* ProgramName;
00149 
00150 // Accessors
00151 public:
00152         const TCHAR* get_Program() const { return m_Program; }
00153         int size() { return static_cast<int>(m_lstParams.size()); }
00154         bool empty() { return m_lstParams.empty(); }
00155         void clear() { m_lstParams.clear();     }
00156         iterator pop_front() { return m_lstParams.erase(m_lstParams.begin()); }
00157         iterator pop_back() { return m_lstParams.erase(--m_lstParams.end()); }
00158         iterator begin() { return m_lstParams.begin(); }
00159         iterator end() { return m_lstParams.end(); }
00160         iterator erase(ParamList::iterator it) { return m_lstParams.erase(it); }
00161 
00162 // Internal functions
00163 private:
00164         int AddArgs(int argc, TCHAR* argv[]) 
00165         {
00166                 if (argc > 0)
00167                 {
00168                         m_Program = argv[0];
00169 
00170                         TCHAR* pMatch;
00171                         for (int i = 1; i < argc; ++i)
00172                         {
00173                                 TCHAR* pArg = argv[i];
00174                                 if (*pArg == _T('-') || *pArg == _T('/'))
00175                                         ++pArg;
00176 
00177                                 if ( (pMatch=_tcschr(pArg,_T('=')) ) != NULL )
00178                                 {
00179                                         if (*(pMatch+1)==_T('\0'))
00180                                                 m_lstParams.push_back(ParamValue(pArg));
00181                                         else
00182                                         {
00183                                                 *pMatch = _T('\0');
00184                                                 m_lstParams.push_back(ParamValue(pArg, pMatch+1));
00185                                         }
00186                                 }
00187                                 else
00188                                         m_lstParams.push_back(ParamValue(pArg));
00189                         }
00190                 }
00191                 return static_cast<int>(m_lstParams.size());
00192         }
00193 };
