00038 /****************************************************************************/
00039 // ObserverList
00040 //
00041 // This class implements the class observer pattern.  It holds a list of 
00042 // objects which can be "invoked" by the client in a thread-safe fashion.
00043 //
00044 // Usage:
00045 //
00046 // class Observer {};
00047 // class Ob_Event2 {};
00048 // static void StaticFunc(Observer* p);
00049 
00050 // ObserverList<Observer*> obList;
00051 //
00052 // obList.add(new Observer1);
00053 // obList.add(new Observer2);
00054 //
00055 // obList.invoke(std::mem_fun(&Observer::Event));
00056 // obList.invoke(std::bind2nd(std::mem_fun(&Observer::Event1), 10));
00057 // obList.invoke(Ob_Event2("Test"));
00058 // obList.invoke(std::ptr_fun(&StaticFunc));
00059 //
00060 /****************************************************************************/
00061 template <class _Observer, bool mustDelete = false, class _LockType = JTI_OBSERVERLST_LOCK_TYPE>
00062 class ObserverList : 
00063         public LockableObject<_LockType>
00064 {
00065 // Constructor
00066 public:
00067         ObserverList() {/* */}
00068         virtual ~ObserverList() { clear(); }
00069 
00070 // Methods
00071 public:
00072         void add(const _Observer& ob)
00073         {
00074                 CCSLock<ObserverList> lockGuard(this);
00075                 lstObservers_.push_front(ob);
00076         }
00077 
00078         bool remove(const _Observer& ob)
00079         {
00080                 CCSLock<ObserverList> lockGuard(this);
00081                 if (std::find(lstObservers_.begin(), lstObservers_.end(), ob) != lstObservers_.end()) {
00082                         lstObservers_.remove(ob);
00083                         return true;
00084                 }
00085                 return false; // does not exists
00086         }
00087 
00088         void clear()
00089         {
00090                 CCSLock<ObserverList> lockGuard(this);
00091                 DeleteObservers(Int2Type<mustDelete>());
00092                 lstObservers_.clear();
00093         }
00094 
00095         // Invoke function
00096         template <class _Func>
00097         void invoke(_Func func)
00098         {
00099                 CCSLock<ObserverList> lockGuard(this);
00100                 if (!lstObservers_.empty()) {
00101                         std::list<_Observer> lst;
00102                         std::copy(lstObservers_.begin(), lstObservers_.end(), std::front_inserter(lst));
00103                         lockGuard.Unlock();
00104                         std::for_each(lst.begin(), lst.end(), func);
00105                 }
00106         }
00107 
00108 // Internal functions
00109 private:
00110         void DeleteObservers(Int2Type<true>)
00111         {
00112                 std::for_each(lstObservers_.begin(), lstObservers_.end(), 
00113                         stdx::delptr<_Observer>());
00114         }
00115 
00116         void DeleteObservers(Int2Type<false>)
00117         {
00118                 /* Do nothing */
00119         }
00120 
00121 // Class data
00122 private:
00123         std::list<_Observer> lstObservers_;
00124 };
