00030         // This structure manages the linked list for the lifetime tracking array
00031         struct LinkedListEntry
00032         {
00033                 LifetimeTracker_Base* pt;
00034                 LinkedListEntry* pn;
00035                 LinkedListEntry() : pt(0), pn(0) {/* */}
00036         };
00037         static LinkedListEntry* _head = 0;
00038 }

00051 static CriticalSectionLock* _getLock()

00058 /*****************************************************************************
00059 ** Procedure:  _AtExitFn
00060 ** 
00061 ** Arguments:  void
00062 ** 
00063 ** Returns: void
00064 ** 
00065 ** Description: This function is set through an atexit() hook to destroy
00066 **              objects in the Lifetime tracker array.  It is called once
00067 **              for every object added to the lifetime array.
00068 **
00069 /****************************************************************************/
00070 static void _AtExitFn()
00071 {
00072         CCSLock<CriticalSectionLock> lockGuard(_getLock());
00073         if (_head != NULL)
00074         {
00075                 // Get the current head and then assign to the next entry.
00076                 LinkedListEntry* top = _head;
00077                 _head = top->pn;
00078                 
00079                 // Delete the entry and the link itself.  The deletion of the 
00080                 // entry will invoke the proper Deleter policy function to correctly
00081                 // destroy the matching object.
00082                 delete top->pt;
00083                 delete top;
00084         }
00085 
00086 }// _AtExitFn

00088 /*****************************************************************************
00089 ** Procedure:  LifetimeTracker_Base::AddTrackedItem
00090 ** 
00091 ** Arguments:  p - New object to track
00092 ** 
00093 ** Returns: true/false success
00094 ** 
00095 ** Description: This function adds a new item to the tracked list.
00096 **
00097 /****************************************************************************/
00098 bool LifetimeTracker_Base::AddTrackedItem(LifetimeTracker_Base* p)
00099 {
00100         // Create our entry
00101         LinkedListEntry* ple = new LinkedListEntry;
00102         ple->pt = p;
00103 
00104         // The chain is kept in lowest longevity order; insert the
00105         // new element into the proper position.
00106         CCSLock<CriticalSectionLock> lockGuard(_getLock());
00107         LinkedListEntry* curr = _head, *prev = NULL;
00108         while (curr != NULL)
00109         {
00110                 // Found our insertion position
00111                 if (curr->pt->longevity_ > p->longevity_)
00112                 {
00113                         // At the head of the chain
00114                         if (prev == NULL)
00115                         {
00116                                 ple->pn = _head;
00117                                 _head = ple;
00118                         }
00119                         else
00120                         {
00121                                 ple->pn = curr;
00122                                 prev->pn = ple;
00123                         }
00124                         break;
00125                 }
00126                 else
00127                 {
00128                         prev = curr;
00129                         curr = curr->pn;
00130                 }
00131 
00132         }
00133 
00134         // If we did not find a position, then insert at the end of the chain
00135         if (curr == NULL)
00136         {
00137                 if (prev != NULL)
00138                         prev->pn = ple;
00139                 else
00140                 {
00141                         ple->pn = _head;
00142                         _head = ple;
00143                 }
00144         }
00145 
00146         // Add an exit function to delete this element.
00147         atexit(_AtExitFn); //lint !e534
00148         return true;
00149 
00150 }// LifetimeTracker_Base::AddTrackedItem
