00039 namespace JTI_Internal

00041         template <class _Container, class _Ty>

00044         template <class _Ty>
00045         inline void internal_push_front(std::vector<_Ty>& container, const _Ty& data) { 
00046                 container.push_back(data);
00047                 std::rotate(container.begin(), container.end()-1, container.end());
00048         }

00052 /**************************************************************************
00053 // ThreadSafeContainer
00054 //
00055 // This container holds the some value type objects; this class has 
00056 // built-in thread safety support but when the iterators are used directly, 
00057 // the user must perform the lock/unlock.
00058 //
00059 **************************************************************************/
00060 template <class _Ty, template<class,class> class _Container = std::vector, 
00061                   class _Alloc = std::allocator<_Ty> >
00062 class ThreadSafeContainer : public LockableObject<JTI_TSCONTAINER_LOCK_TYPE>
00063 {
00064 // Public types
00065 public:
00066         typedef ThreadSafeContainer<_Ty, _Container, _Alloc> ThisType;
00067         typedef _Container<_Ty, _Alloc> TContainer;
00068         typedef typename TContainer::value_type value_type;
00069         typedef typename TContainer::const_iterator const_iterator;
00070         typedef typename TContainer::iterator iterator;
00071         typedef typename TContainer::size_type size_type;
00072 
00073 // Constructor
00074 public:
00075         ThreadSafeContainer() : container_() {/* */}
00076         explicit ThreadSafeContainer(const ThisType& rhs) : container_(rhs.container_) {/* */}
00077         explicit ThreadSafeContainer(const TContainer& rhs) : container_(rhs) {/* */}
00078 
00079 // Accessors
00080 public:
00081         size_type size() const { CCSLock<ThisType> lockGuard(this); return container_.size(); }
00082         bool empty() const { CCSLock<ThisType> lockGuard(this); return container_.empty(); }
00083         const_iterator begin() const { return container_.begin(); }
00084         const_iterator end() const { return container_.end(); }
00085         iterator begin() { return container_.begin(); }
00086         iterator end() { return container_.end(); }
00087         void clear() { CCSLock<ThisType> lockGuard(this); container_.clear(); }
00088 
00089 // Methods
00090 public:
00091         template <class _OutIt, class _Pred>
00092         _OutIt locked_copy_if(_OutIt it, _Pred pred) {
00093                 CCSLock<ThisType> lockGuard(this);
00094                 // No copy_if algorithm in STL; so use Not(remove_copy_if)
00095                 return std::remove_copy_if(begin(), end(), it, std::not1(pred));
00096         }
00097         template <class _OutIt>
00098         _OutIt locked_copy(_OutIt it) {
00099                 CCSLock<ThisType> lockGuard(this);
00100                 return std::copy(begin(), end(), it);
00101         }
00102 
00103         template <class _Pred>
00104         const_iterator find_if(_Pred predicate) const {
00105                 CCSLock<ThisType> lockGuard(this);
00106                 return std::find_if(container_.begin(), container_.end(), predicate);
00107         }
00108         template <class _Pred>
00109         iterator find_if(_Pred predicate) {
00110                 CCSLock<ThisType> lockGuard(this);
00111                 return std::find_if(container_.begin(), container_.end(), predicate);
00112         }
00113 
00114         bool exists(const _Ty& data) const {
00115                 CCSLock<ThisType> lockGuard(this); 
00116                 return (std::find(container_.begin(), container_.end(), data) != container_.end());
00117         }
00118 
00119         void push_back(const _Ty& data) { 
00120                 CCSLock<ThisType> lockGuard(this); 
00121                 container_.push_back(data); 
00122                 lockGuard.Unlock();
00123         }
00124 
00125         void push_front(const _Ty& data) {
00126                 CCSLock<ThisType> lockGuard(this); 
00127                 JTI_Internal::internal_push_front(container_, data); 
00128                 lockGuard.Unlock();
00129         }
00130 
00131         _Ty pop()
00132         {
00133                 CCSLock<ThisType> lockGuard(this); 
00134                 iterator it = container_.begin();
00135                 _Ty t = (*it);
00136                 container_.erase(it);
00137                 return t;
00138         }
00139 
00140         void remove(const _Ty& data) { 
00141                 CCSLock<ThisType> lockGuard(this); 
00142                 iterator it = std::find(container_.begin(), container_.end(), data);
00143                 if (it != container_.end())
00144                         container_.erase(it);
00145         }
00146 
00147 // Class data
00148 private:
00149         TContainer container_;
00150 };
00151
