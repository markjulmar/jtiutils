00048 /****************************************************************************/
00049 // LifetimeTracker_Base
00050 //
00051 // The LifetimeTracker is the basic object used to track an object's lifetime.
00052 // It is a simple numeric identity where the higher the numeric value, the
00053 // longer that object will exist.
00054 //
00055 /****************************************************************************/
00056 class LifetimeTracker_Base
00057 {
00058 public:
00059         virtual ~LifetimeTracker_Base() {}
00060         static bool AddTrackedItem(LifetimeTracker_Base* p);
00061 protected:
00062         LifetimeTracker_Base(unsigned int x) : longevity_(x) {/* */}
00063 private:
00064         unsigned int longevity_;
00065 };
00066 
00067 /****************************************************************************/
00068 // CRTDeleter
00069 //
00070 // This provides a "normal" C++ CRT delete function for the LifetimeTracker
00071 // wrapper to use for deleting it's contained object.
00072 //
00073 /****************************************************************************/
00074 template <typename T>
00075 struct CRTDeleter
00076 {
00077         void operator()(T* p)
00078         {
00079                 delete p;
00080         }
00081 };
00082 
00083 /****************************************************************************/
00084 // CallbackDeleter
00085 //
00086 // This provides a callback deleter function to delete a given type.
00087 //
00088 /****************************************************************************/
00089 template <typename T>
00090 struct CallbackDeleter
00091 {
00092         void operator()(T* p) 
00093         { 
00094                 pf_(p); 
00095         }
00096         
00097         typedef void (*VFUNCPTR)(T*);
00098         VFUNCPTR pf_;
00099         CallbackDeleter(VFUNCPTR pf) : pf_(pf) {/* */}
00100 };
00101 
00102 /****************************************************************************/
00103 // LifetimeTracker
00104 //
00105 // This is the concrete implementation of the lifetime tracker object.
00106 // It provides for overriding the method of destruction (the default is the
00107 // object deleter above).
00108 //
00109 // Usage:
00110 //
00111 // class A {}
00112 // A* pa = new A(), *pb = new A();
00113 //
00114 // LifetimeTracker<A>::SetLongevity(pa, 100);
00115 // LifetimeTracker<A>::SetLongevity(pb, 50);
00116 //
00117 // Destruction order will be "pb" followed by "pa".
00118 //
00119 // Note that you cannot unregister a deletion, so make sure to not delete
00120 // any objects registered with the LifetimeTracker<> template directly.
00121 //
00122 /****************************************************************************/
00123 template <typename T, typename D = CRTDeleter<T> >
00124 class LifetimeTracker : public LifetimeTracker_Base
00125 {
00126 public:
00127         virtual ~LifetimeTracker() { _destroyer(_p); }
00128 
00129         // This function sets the longevity for a tracked object and provides the
00130         // mechanism to destroy the object when it is finally reclaimed.
00131         static void SetLongevity(T* p, unsigned int longevity, D d = CRTDeleter<T>())
00132         {
00133                 if (!LifetimeTracker_Base::AddTrackedItem(JTI_NEW LifetimeTracker<T,D>(p,longevity,d)))
00134                         throw std::bad_alloc();
00135         }
00136 
00137 // Constructor
00138 private:
00139         LifetimeTracker(T* p, unsigned int longevity, D d) : 
00140                 LifetimeTracker_Base(longevity), _p(p), _destroyer(d) 
00141         {
00142         }
00143 
00144 // Class data
00145 private:
00146         T* _p;
00147         D _destroyer;
00148 };
