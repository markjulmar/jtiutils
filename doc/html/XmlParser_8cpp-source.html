00238 /*****************************************************************************
00239 ** Procedure:  InternalParser CollectToWhitespace
00240 ** 
00241 ** Arguments: 'pszData' - Current position in buffer (adjusted)
00242 ** 
00243 ** Returns: String read
00244 ** 
00245 ** Description: This reads up to the next whitespace character
00246 **
00247 /****************************************************************************/
00248 std::string InternalParser::CollectToWhitespaceOrTag(LPCSTR& pszCurrent, XmlTagType tagToEnd) const
00249 {
00250         std::string text;
00251         while (*pszCurrent && !isspace(*pszCurrent))
00252         {
00253                 XmlTagType tagThis = IdentifyToken(pszCurrent);
00254                 if (tagThis == tagToEnd || (tagToEnd == TT_ANY && tagThis != TT_TEXT))
00255                         break;
00256                 text += *pszCurrent++;
00257         }
00258 
00259         if (*pszCurrent == '\0' && tagToEnd != TT_ANY)
00260                 Throw("Hit end of stream while searching for tag %s", GetTagText(tagToEnd));
00261 
00262         return text;
00263 
00264 }// InternalParser CollectToWhitespace

00266 /*****************************************************************************
00267 ** Procedure:  InternalParser::CollectToTag
00268 ** 
00269 ** Arguments: 'pszData' - Current position in buffer (adjusted)
00270 ** 
00271 ** Returns: String read
00272 ** 
00273 ** Description: This reads up to the given tag.  This is similar to the
00274 **              above function but ignores whitespace.

00276 /****************************************************************************/
00277 std::string InternalParser::CollectToTag(LPCSTR& pszCurrent, XmlTagType tagToEnd) const
00278 {
00279         std::string text;
00280         while (*pszCurrent)
00281         {

00288         if (*pszCurrent == '\0')

00290 
00291         return text;

00295 /*****************************************************************************
00296 ** Procedure:  InternalParser::ParseAttribute
00297 ** 
00298 ** Arguments: 'node' - Node to fill in 
00299 **            'pszCurrent' - Current position in the buffer.
00300 ** 
00301 ** Returns: True/False success
00302 ** 
00303 ** Description: This parses out a single attribute from the buffer and
00304 **              adds it to the given node.
00305 **
00306 /****************************************************************************/
00307 bool InternalParser::ParseAttribute(XmlNode& node, LPCSTR& pszCurrent) const
00308 {
00309         // Current position is at beginning of text.  Read until we hit a space
00310         // or an equal sign indicating the start of the value.
00311         std::string attValue, attName = CollectToWhitespaceOrTag(pszCurrent, TT_EQUAL_SIGN);
00312 
00313         // Make sure the next token is an equal sign!
00314         int size = 0;
00315         if (NextToken(pszCurrent, size) != TT_EQUAL_SIGN)
00316                 Throw("Missing '=' on attribute for element %s.", node.Name.c_str());
00317 
00318         // Now, move until we hit a text tag.  Note whether we hit a quote/doublequote
00319         // tag so we can search for the end marker.
00320         XmlTagType tagNext = NextToken(pszCurrent, size);
00321         if (tagNext == TT_QUOTE || tagNext == TT_SINGLEQUOTE)
00322         {
00323                 attValue = CollectToTag(pszCurrent, tagNext);
00324                 if (NextToken(pszCurrent, size) != tagNext)
00325                         Throw("Missing closing quote on attribute %s for element %s.", attName.c_str(), node.Name.c_str());
00326         }
00327         else if (tagNext == TT_TEXT)
00328                 attValue = CollectToWhitespaceOrTag(pszCurrent, TT_ANY);
00329         else
00330         {
00331                 // Invalid tag hit
00332                 return false;
00333         }
00334 
00335         node.Attributes.add(attName.c_str(), attValue.c_str());
00336         return true;
00337 
00338 }// InternalParser::ParseAttribute

00340 /*****************************************************************************
00341 ** Procedure:  InternalParser::ParseElementData
00342 ** 
00343 ** Arguments: 'node' - Node to fill in 
00344 **            'pszCurrent' - Current position in the buffer.
00345 ** 
00346 ** Returns: Node which was read in.
00347 ** 
00348 ** Description: This parses out a single element from the XML file.
00349 **
00350 /****************************************************************************/
00351 bool InternalParser::ParseElementData(XmlNode& node, LPCSTR& pszCurrent) const
00352 {
00353         // Current position is at beginning of text.  Read until we hit a space
00354         // or any known tag.  Element names are not allowed to contain spaces or
00355         // reserved keyword (tag) characters.
00356         node.Name = CollectToWhitespaceOrTag(pszCurrent).c_str();
00357 
00358         // Now parse the remaining tokens until we hit an end marker.
00359         while (*pszCurrent)
00360         {
00361                 int size = 0;
00362                 XmlTagType tag = NextToken(pszCurrent, size);
00363                 switch (tag)
00364                 {
00365                         // Hit more text?  must be an attribute.
00366                         case TT_TEXT:
00367                                 ParseAttribute(node, pszCurrent);
00368                                 break;
00369 
00370                         // Got the end-tag or empty tag, backup and end.
00371                         case TT_OPEN_ENDTAG:
00372                         case TT_CLOSE_ENDTAG:
00373                                 pszCurrent -= size;
00374                                 return true;
00375 
00376                         // Some unknown tag.
00377                         default:
00378                                 return false;
00379                 }
00380         }
00381         return false;
00382 
00383 }// InternalParser::ParseElementData

00385 /*****************************************************************************
00386 ** Procedure:  InternalParser::ParseText
00387 ** 
00388 ** Arguments: 'node' - Node to fill in 
00389 **            'pszCurrent' - Current position in the buffer.
00390 **            'lastTag' - Last tag we hit in the stream before the text.
00391 ** 
00392 ** Returns: void
00393 ** 
00394 ** Description: This parses out text from the XML stream.
00395 **
00396 /****************************************************************************/
00397 bool InternalParser::ParseText(XmlNode& node, LPCSTR& pszCurrent, XmlTagType lastTag) const
00398 {
00399         if (lastTag == TT_OPEN_ENDTAG)
00400         {
00401                 // Collect to the next close tag.
00402                 node.Value = CollectToTag(pszCurrent, TT_CLOSE_TAG).c_str();
00403         }
00404         else if (lastTag == TT_CLOSE_TAG)
00405         {
00406                 std::string name = CollectToWhitespaceOrTag(pszCurrent, TT_OPEN_ENDTAG);
00407                 if (name != node.Name)
00408                         Throw("Name on end tag (%s) does not match start tag (%s).", name.c_str(), node.Name.c_str());
00409                 return true;
00410         }
00411         else
00412         {
00413                 Throw("Invalid text found (%s...), malformed document.", std::string(pszCurrent, 20).c_str());
00414         }
00415 
00416         return false;
00417 
00418 }// InternalParser::ParseText

00420 /*****************************************************************************
00421 ** Procedure:  InternalParser::ReadElement
00422 ** 
00423 ** Arguments:  'pszCurrent' - Current position in the buffer.
00424 **             'level' - Current element level
00425 ** 
00426 ** Returns: Node which was read in.
00427 ** 
00428 ** Description: This parses out a single element from the XML file.
00429 **
00430 /****************************************************************************/
00431 XmlNode InternalParser::ReadElement(LPCSTR& pszCurrent, int level) const
00432 {
00433         XmlNode node;
00434         XmlTagType lastTag = TT_ANY;
00435 
00436         // If not a valid buffer..
00437         if (pszCurrent == NULL || *pszCurrent == '\0')
00438                 Throw("Hit end of stream reading an incomplete element.");
00439 
00440         // Keep walking until we hit the end.
00441         while (*pszCurrent)
00442         {
00443                 int size = 0;
00444                 XmlTagType tag = NextToken(pszCurrent, size);
00445                 switch (tag)
00446                 {
00447                         case TT_OPEN_TAG:
00448                                 if (!node.IsValid)
00449                                 {
00450                                         if (!ParseElementData(node, pszCurrent))
00451                                                 Throw("Error parsing element %s", std::string(pszCurrent,20).c_str());
00452                                 }
00453                                 else
00454                                 {
00455                                         pszCurrent -= size;
00456                                         node.Children.add(ReadElement(pszCurrent, level+1));
00457                                 }
00458                                 break;
00459 
00460                         case TT_OPEN_ENDTAG:
00461                                 if (!node.IsValid)
00462                                         Throw("Hit end-tag without fully formed element (%s...)", std::string(pszCurrent, 20).c_str());
00463                                 break;
00464 
00465                         case TT_CLOSE_TAG:
00466                                 if (!node.IsValid)
00467                                         Throw("Hit end-tag without fully formed element (%s...)", std::string(pszCurrent, 20).c_str());
00468                                 break;
00469 
00470                         
00471                         case TT_QUOTE:
00472                         case TT_SINGLEQUOTE:
00473                                 if (!node.IsValid) {
00474                                         Throw("Hit end-tag without fully formed element (%s...)", std::string(pszCurrent, 20).c_str());
00475                                         break;
00476                                 }
00477                                 pszCurrent -= size;
00478 
00479                                 // Fall through intentional.
00480 
00481                         case TT_TEXT:
00482                                 if (ParseText(node, pszCurrent, lastTag))
00483                                 {
00484                                         // Parse indicated end tag found for this element; make sure
00485                                         // and then exit.
00486                                         if (NextToken(pszCurrent, size) != TT_OPEN_ENDTAG)
00487                                                 Throw("Missing end-tag on element %s", node.Name.c_str());
00488                                         return node;
00489                                 }
00490                                 break;
00491 
00492                         case TT_CLOSE_ENDTAG:
00493                                 return node;
00494 
00495                         case TT_PI_TAG:
00496                                 CollectToTag(pszCurrent, TT_PI_ENDTAG);
00497                                 break;
00498                         case TT_COMMENT_TAG:
00499                                 node.Comments.add(CollectToTag(pszCurrent, TT_COMMENT_ENDTAG).c_str());
00500                                 break;
00501                         case TT_MARKUP_TAG:
00502                                 CollectToTag(pszCurrent, TT_OPEN_ENDTAG);
00503                                 break;
00504 
00505                         // Ignore
00506                         case TT_PI_ENDTAG:
00507                         case TT_COMMENT_ENDTAG:
00508                                 break;
00509 
00510                         default:
00511                                 return node;
00512                 }
00513 
00514                 // Save off the last tag we hit.
00515                 lastTag = tag;
00516         }
00517 
00518         return node;
00519 
00520 }// InternalParser::ReadElement

00591                 }

00593 
00594         // Add the opening tag
00595         stm << std::string(level,'\t') << InternalParser::GetTagText(TT_OPEN_TAG) << name_;

00697                 return XmlNode();

00704         XmlNodeImpl* pcurrNode = pImpl_;
00705         for (std::vector<std::string>::iterator it = arrPath.begin(); it != arrPath.end(); ++it)
00706         {
00707                 // If the current node has no children then we are done.
00708                 if (pcurrNode->children_.empty())
00709                         return XmlNode();
00710 
00711                 // Search the array.
00712                 std::vector<XmlNodeImpl*>::iterator itNode = pcurrNode->children_.begin();
00713                 for (; itNode != pcurrNode->children_.end(); ++itNode)
00714                 {
00715                         if ((*itNode)->name_ == (*it)) {
00716                                 pcurrNode = (*itNode);
00717                                 break;
00718                         }
00719                 }
00720 
00721                 if (itNode == pcurrNode->children_.end())
00722                         return XmlNode();
00723         }
00724 
00725         // Return the found node
00726         return XmlNode(pcurrNode);
00727 
00728 }// XmlNode::find

00754 XmlNode XmlDocument::find(const char* pszPath) const
00755 {
00756         if (pszPath == NULL || *pszPath == '\0')
00757                 return XmlNode();
00758 
00759         // Skip any leading seperator.  Standard XML syntax allows for
00760         // root node to be identified this way.
00761         if (*pszPath == '/' || *pszPath == '\\')
00762                 ++pszPath;
00763 
00764         std::vector<std::string> arrPath;
00765         stdx::split(pszPath, "/\\", std::back_inserter(arrPath));
00766 
00767         if (arrPath.empty() || arrPath[0] != root_.Name)
00768                 return XmlNode();
00769 
00770         XmlNode currNode = root_;
00771         if (arrPath.size() > 1)

00774                 {
00775                         if (it->empty())
00776                                 continue;
00777                         if (!currNode.IsValid)
00778                                 break;
00779                         if (!currNode.HasChildren)
00780                                 return XmlNode();
00781                         currNode = currNode.Children.find(it->c_str());
00782                 }
00783         }
00784         return currNode;

00788 /*****************************************************************************
00789 ** Procedure:  XmlDocument::load
00790 ** 
00791 ** Arguments:  'pszFile' - Filename to load
00792 ** 
00793 ** Returns: true/false whether load was successful.
00794 ** 
00795 ** Description: This loads an XML file into memory.
00796 **
00797 /****************************************************************************/
00798 bool XmlDocument::load(const char* pszFile)
00799 {
00800         const int PATTERN_COUNT = 4;
00801         static unsigned char gszPatterns[PATTERN_COUNT][5] = {
00802                 { 3, 0xEF, 0xBB, 0xBF },                        // UTF-8
00803                 { 2, 0xFE, 0xFF },                                      // UTF-16 UCS-2 little endian
00804                 { 4, 0xFF, 0xFE, 0x00, 0x00 },          // UTF-16/32 UCS-2/4 
00805                 { 4, 0x00, 0x00, 0xFE, 0xFF }           // UTF-32 UCS-4 big endian
00806         };
00807 
00808         // Read the file into a string.
00809         std::string xmlBuffer;
00810         std::ifstream ifs(pszFile);
00811         std::istreambuf_iterator<char> itBegin(ifs), itEnd;
00812 
00813         // Skip any XML byte ordering marker
00814         if (itBegin != itEnd)
00815         {
00816                 unsigned char ch = static_cast<unsigned char>(*itBegin);
00817                 for (int i = 0; i < PATTERN_COUNT; ++i)
00818                 {
00819                         if (ch == gszPatterns[i][1])
00820                         {
00821                                 for (int x = 1; x <= gszPatterns[i][0]; ++x && itBegin != itEnd, ++itBegin)
00822                                 {
00823                                         ch = static_cast<unsigned char>(*itBegin);
00824                                         if (ch != gszPatterns[i][x])
00825                                                 break;
00826                                 }
00827                                 break;
00828                         }
00829                 }
00830         }
00831 
00832         // Load the full XML text.
00833         xmlBuffer = std::string(itBegin, itEnd);
00834 
00835         // Now parse out the buffer
00836         parse(xmlBuffer);
00837         return true;
00838 
00839 }// XmlDocument::load
