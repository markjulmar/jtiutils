00035 /******************************************************************************/
00036 // TimerEntry
00037 //
00038 // This describes a single timer entry with delegate.
00039 //
00040 /******************************************************************************/
00041 class TimerEntry
00042 {
00043 // Constructor
00044 public:
00045         TimerEntry(int timerID, DWORD msecInterval, DelegateInvoke_1<int>* Func) : 
00046           timerID_(timerID), lastFired_(GetTickCount()), msecInterval_(msecInterval), invokeFunc_(Func) {/* */}
00047     ~TimerEntry() { delete invokeFunc_; }
00048 
00049 // Operators
00050 public:
00051         // Comparison
00052         bool operator==(const TimerEntry& rhs) const
00053         {
00054                 return (timerID_ == rhs.timerID_ && invokeFunc_ == rhs.invokeFunc_);
00055         }
00056 
00057         // Sort by next timer
00058         bool operator<(const TimerEntry& rhs) const
00059         {
00060                 return (get_NextFireTime() < rhs.get_NextFireTime());
00061         }
00062 
00063 // Accessors
00064 public:
00065         int get_ID() const { return timerID_; }
00066         // Calculate the next tick count when this timer should fire
00067         DWORD get_NextFireTime() const { return lastFired_ + msecInterval_; }
00068         // Return the last fire time
00069         DWORD get_LastFireTime() const { return lastFired_; }
00070         // Return the timer interval
00071         DWORD get_Interval() const { return msecInterval_; }
00072         // Execute the timer
00073         bool FireTimer();
00074 
00075 // Class data
00076 private:
00077         int timerID_;
00078         DWORD lastFired_;
00079         DWORD msecInterval_;
00080         DelegateInvoke_1<int>* invokeFunc_;
00081 
00082 // Unavailable methods
00083 private:
00084         TimerEntry(const TimerEntry& rhs);
00085         TimerEntry operator=(const TimerEntry& rhs);
00086 };

00088 /******************************************************************************/
00089 // TimerManager
00090 //
00091 // This describes the timer manager class.  This class controls the invocation
00092 // of timer events.
00093 //
00094 /******************************************************************************/
00095 class TimerManager : public LockableObject<MultiThreadModel>
00096 {
00097 // Constructor
00098 public:
00099         TimerManager() : thread_(0), evtStop_(false, true), evtNewTimer_(false, true) {/* */}
00100 public:
00101         ~TimerManager();
00102 
00103 // Accessors
00104 public:
00105         // Clear all timers
00106         void Clear();
00107 
00108         // Add a new timer
00109         template<class _Object, class _Class>
00110         void AddTimer(int nTimer, DWORD msecTimeout, const _Object& object, void (_Class::* fnc)(int))
00111         {
00112                 AddTimer(nTimer, msecTimeout, JTI_NEW DelegateInvokeMethod_1<_Class,int>(*(_Object*)&object, fnc));
00113         }
00114         void AddTimer(int nTimer, DWORD msecTimeout, void (*fnc)(int))
00115         {
00116                 AddTimer(nTimer, msecTimeout, JTI_NEW DelegateInvokeFunc_1<int>(fnc));
00117         }
00118 
00119         // Remove an existing timer.  This removes the 1st timer in the list matching the id.
00120         bool KillTimer(int nTimer);
00121 
00122 // Internal functions
00123 private:
00124         void AddTimer(int nTimer, DWORD msecTimeout, DelegateInvoke_1<int>* Func);
00125 
00126         // Timer Worker thread
00127         void TimerWorker();
00128         static unsigned __stdcall TimerWorkerThread(void* p)
00129         {
00130                 reinterpret_cast<TimerManager*>(p)->TimerWorker();
00131                 return 0;
00132         }
00133 
00134 // Unavailable functions
00135 private:
00136         TimerManager(const TimerManager&);
00137         TimerManager& operator==(const TimerManager&);
00138 
00139 // Class data
00140 private:
00141         std::vector<TimerEntry*> qTimers_;
00142         HANDLE thread_;
00143         EventSynch evtStop_;
00144         EventSynch evtNewTimer_;
00145 };
