00028 namespace JTI_Util

00030 /*****************************************************************************
00031 ** seh_exception
00032 ** 
00033 ** This exception class can catch Structured Exceptions (SEH) on the 
00034 ** Windows platform.
00035 **
00036 /****************************************************************************/
00037 class seh_exception : public std::runtime_error
00038 {
00039         // Constructors
00040 public:
00041         seh_exception() : _type(0), std::runtime_error("") {}
00042         seh_exception(unsigned int type) : _type(type), std::runtime_error("") {}
00043         seh_exception(const seh_exception& rhs) : _type(rhs._type), std::runtime_error(rhs) {}
00044 public:
00045         int GetExceptionType() const { return _type; }
00046         static void Install() { _set_se_translator(&_SEHHandler); }
00047 private:
00048         static void _SEHHandler(unsigned int uException, _EXCEPTION_POINTERS*) {
00049                 throw seh_exception(uException);
00050         }
00051         // Class data
00052 private:
00053         unsigned int _type;
00054 };
00055 }// namespace JTI_Util
