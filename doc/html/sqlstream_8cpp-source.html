00051 /**************************************************************************************
00052 ** Procedure: CDlBstr::ValidatedValue
00053 **
00054 ** Arguments: void
00055 **
00056 ** Returns:     Bool/BSTR pair for the given value.
00057 **
00058 ** Description: This is used during the column creation process to validate the
00059 **              internal boolean value.
00060 **
00061 **************************************************************************************/
00062 std::pair<bool,std::wstring> CDlBstr::ValidatedValue() const 
00063 {
00064         // Bracket entire string in single quotes and escape each embedded quote with 
00065         // by an additional single quote. Zero length string generates destination 
00066         // string of two consecutive single quotes
00067         const wchar_t *pSrc = value_.c_str();
00068         const wchar_t QUOTE= L'\'';
00069 
00070         // Begin building the destination string; it starts with a quote
00071         std::wstring strValue; strValue.reserve(value_.length() * 2);
00072         strValue = QUOTE;
00073             
00074         // If the string is not null --
00075         if (!value_.empty())
00076         {
00077         while (*pSrc)
00078         {       
00079                 // double-copy sequence of quotes, if any
00080                 while (*pSrc && (*pSrc == QUOTE))
00081                 {       
00082                         strValue += *pSrc;      
00083                         strValue += *pSrc++;    
00084                 }
00085 
00086                 // pSrc is pointing at non-quote or reached NULL
00087                 size_t lFoundAtIdx= wcscspn(pSrc, L"'");        
00088                 if (!lFoundAtIdx) 
00089                                 continue; //reached NULL        
00090 
00091                         // Copy the new section over and increment our buffer.
00092                         strValue += std::wstring(pSrc,lFoundAtIdx);
00093                 pSrc += lFoundAtIdx; // points to quote or terminating NULL     
00094         }
00095         }       
00096             
00097         // .. and ends with a quote
00098         strValue += QUOTE;
00099 
00100         return (value_.empty()) ? std::make_pair(isNullValid_, strValue) : std::make_pair(true, strValue);
00101 
00102 }// CDlBstr::ValidatedValue

00104 /**************************************************************************************
00105 ** Procedure: CDlCmdSQL::BuildCommand
00106 **
00107 ** Arguments: void
00108 **
00109 ** Returns:     String with full command to execute
00110 **
00111 ** Description: This takes all the parameters and builds a string to send to the 
00112 **              data driver.
00113 **
00114 **************************************************************************************/
00115 std::wstring CDlCmdSQL::BuildCommand() const
00116 {
00117         std::wstring strExpression(sqlText_); 
00118         strExpression.reserve(lstrlenW(sqlText_)+argLength_);
00119 
00120         // Walk the expressions, changing out the question marks which 
00121         // indicate parameters for the real string in our array.
00122         long nCurr = 0;
00123         std::wstring::size_type idx = strExpression.find_first_of(L'?', 0);
00124         while (idx != std::wstring::npos)
00125         {
00126                 JTI_ASSERT(static_cast<long>(argsArray_.size()) > nCurr);
00127                 if (static_cast<long>(argsArray_.size()) > nCurr)
00128                         strExpression.replace(idx,1,argsArray_[nCurr].c_str());
00129                 ++nCurr; idx = strExpression.find_first_of(L'?', idx+1);
00130         }
00131 
00132         return strExpression;
00133 
00134 }// CDlCmdSQL::BuildCommand

00147 std::wstring CDlCmdExec::BuildCommand() const
00148 {
00149         const wchar_t* gpszExec = L"Exec ";
00150         std::wstring strExpression;      
00151         strExpression.reserve(lstrlenW(gpszExec)+lstrlenW(sqlText_)+1+argLength_+static_cast<int>(argsPairArray_.size()));
00152         strExpression = gpszExec;
00153 
00154         strExpression += sqlText_;
00155         strExpression += L' ';
00156 
00157         // Append all paramater expressions with a trailing comma.
00158         for(std::vector<std::pair<std::wstring,std::wstring> >::const_iterator it = argsPairArray_.begin(); it != argsPairArray_.end(); ++it)
00159         {
00160                 strExpression += L'@';
00161                 strExpression += it->first;
00162                 strExpression += L'=';
00163                 strExpression += it->second;
00164                 strExpression += L',';
00165         }
00166 
00167         // Remove the trailing comma or blank.
00168         strExpression.erase(strExpression.length()-1);
00169         return strExpression;
00170 
00171 }// CDlCmdExec::BuildCommand

00180 ** Description: This builds the shape command to send to the driver.

00187 **        ({ ... })
00188 **
00189 **************************************************************************************/

00191 {
00192         static const wchar_t * const gSyntax_[] = {
00193                 L"SHAPE { ",            // 0
00194                 L" } APPEND ",          // 1
00195                 L" ({ ",                        // 2
00196                 L" } ",                         // 3
00197                 L", "                           // 4
00198         };
00199 
00200         // Build the shape command
00201         std::wstring sResult = gSyntax_[0];
00202         sResult += textHeader_;
00203         sResult += gSyntax_[1];
00204 
00205         for (std::vector<std::pair<std::wstring,std::wstring> >::const_iterator it = shapeDetails_.begin(); it != shapeDetails_.end(); ++it)
00206         {
00207                 if (it != shapeDetails_.begin())
00208                         sResult += gSyntax_[4];
00209                 sResult += gSyntax_[2];
00210                 sResult += it->first;
00211                 sResult += gSyntax_[3];
00212                 sResult += it->second;
00213         }
00214         return sResult;
