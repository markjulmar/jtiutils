00239 /*****************************************************************************
00240 ** Procedure:  QueryImplementation
00241 ** 
00242 ** Arguments:  'punk' - IUnknown interface
00243 **             'clsid' - CLSID to retrieve
00244 **             'ppobj' - Returning pointer to implementation
00245 ** 
00246 ** Returns: OLE2 result code
00247 ** 
00248 ** Description: This template function can be used on a class which has
00249 **              the COM_INTERFACE_ENTRY_THIS macro in it's ATL interface
00250 **              map.  This function will then allow the caller to retrieve
00251 **              the underlying implementation pointer (i.e. Cxxx object).
00252 **
00253 /****************************************************************************/
00254 template <typename T>
00255 HRESULT QueryImplementation(IUnknown* punk, REFCLSID clsid, T** ppobj)
00256 {
00257         COINITPARAM(ppobj);
00258     if (!punk) return E_INVALIDARG;
00259     HRESULT hr = punk->QueryInterface(clsid, (void**)ppobj);
00260     if( FAILED(hr) ) return hr;
00261 #ifdef _ATL_DEBUG_INTERFACES
00262     // If _ATL_DEBUG_INTERFACES, then ATL has changed the
00263     // pointer computed by QI to be a _QIThunk for the pointer
00264     _QIThunk* thunk = reinterpret_cast<_QIThunk*>(*ppobj);
00265     ATLASSERT(thunk);
00266 #if(_ATL_VER < 0x0700)
00267     // The pUnk held by the thunk is the pv returned by _This.
00268     *ppobj = reinterpret_cast<T*>(thunk->pUnk);
00269     // Delete this thunk as it will never be released.
00270     ATLASSERT(thunk->m_dwRef == 1);
00271         if( !thunk->bNonAddRefThunk ) _pModule->DeleteThunk(thunk);
00272 #else
00273     // The pUnk held by the thunk is the pv returned by _This.
00274     *ppobj = reinterpret_cast<T*>(thunk->m_pUnk);
00275     // Delete this thunk as it will never be released.
00276     ATLASSERT(thunk->m_dwRef == 1);
00277         if( !thunk->m_bNonAddRefThunk ) _pModule->DeleteThunk(thunk);
00278 #endif // _ATL_VER < 0x700
00279 #endif // _ATL_DEBUG_INTERFACES
00280     return S_OK;
00281 }
00282 
00283 template <typename T>
00284 HRESULT QueryImplementation(IUnknown* punk, T** ppobj) {
00285     return QueryImplementation(punk, T::GetObjectCLSID(), ppobj);
00286 }

00416         VARIANT varArray;
00417         VariantInit(&varArray);

00664 //

00666 //   HRESULT hr = pe->CreateInstance(&pe);

00670 //      CComObject<CComContainerCopy<std::vector<VARIANT> > >* pCopy = 0;

00674 //         pCopy->AddRef();
00675 //         hr = pCopy->Copy(m_arr);

00679 //             if (SUCCEEDED(hr))
