00146 class ManagementObjectImpl : public JTI_Util::RefCountedObject<JTI_Util::MultiThreadModel>
00147 {
00148 // Class data
00149 private:
00150         TString machine_;
00151         TString scope_;
00152         TString objectPath_;
00153         ManagementNameValueCollection coll_;
00154         ManagementNameValueCollection syscoll_;
00155         ManagementOptions options_;
00156 
00157 // Constructor
00158 public:
00159         ManagementObjectImpl() : scope_(_DEFAULT_WMI_SCOPE_) {/* */}
00160         ManagementObjectImpl(const TCHAR* wmiPath) : 
00161                 scope_(_DEFAULT_WMI_SCOPE_), objectPath_((wmiPath)?wmiPath:"") {/* */}
00162         ManagementObjectImpl(const TCHAR* scope, const TCHAR* wmiPath) :
00163                 scope_((scope)?scope:_DEFAULT_WMI_SCOPE_), objectPath_((wmiPath)?wmiPath:"") {/* */}
00164         ManagementObjectImpl(const TCHAR* machine, const TCHAR* scope, const TCHAR* wmiPath) :
00165                 machine_((machine)?machine:_T("")), scope_((scope)?scope:_DEFAULT_WMI_SCOPE_), objectPath_((wmiPath)?wmiPath:"") {/* */}
00166 
00167 // Operators
00168 private:
00169         ManagementObjectImpl(const ManagementObjectImpl& rhs);
00170         ManagementObjectImpl& operator=(const ManagementObjectImpl& rhs);
00171 
00172 // Methods
00173 public:
00174         bool Get() {
00175                 // Clear our collection
00176                 coll_.Clear();
00177                 syscoll_.Clear();
00178 
00179                 // Split out the path from the object
00180                 TString typeName; bool foundMatch=false;
00181                 size_t nPos = objectPath_.find(_T("."));
00182                 if (nPos != TString::npos)
00183                         typeName = objectPath_.substr(0,nPos);
00184                 else 
00185                         return false;
00186 
00187                 // Load up our WMI locater.
00188                 CComPtr<IWbemLocator> piLocator = 0;
00189                 HRESULT hr = piLocator.CoCreateInstance(CLSID_WbemLocator, NULL, CLSCTX_INPROC_SERVER);
00190                 if (FAILED(hr))
00191                         return false;
00192 
00193                 // Build the connection string; this is the machine name + scope (namespace).
00194                 TString connString;
00195                 if (!machine_.empty()) {
00196                         if (machine_[0] != _T('\\'))
00197                                 connString = _T("\\\\");
00198                         connString += machine_;
00199                         connString += _T("\\");
00200                 }
00201                 connString += scope_;
00202 
00203                 // Connect to the appropriate server
00204                 CComPtr<IWbemServices> piNamespace = 0;
00205                 hr = piLocator->ConnectServer(CComBSTR(connString.c_str()), NULL, NULL, 0L, 0L, NULL, NULL, &piNamespace);
00206                 if (FAILED(hr))
00207                         return false;
00208 
00209                 // Get an instance enum for this type.
00210                 CComPtr<IEnumWbemClassObject> piEnum = 0;
00211                 hr = piNamespace->CreateInstanceEnum(CComBSTR(typeName.c_str()), 
00212                         WBEM_FLAG_SHALLOW | WBEM_FLAG_FORWARD_ONLY | WBEM_RETURN_IMMEDIATELY, NULL, &piEnum);
00213                 if (FAILED(hr))
00214                         return false;
00215 
00216                 // Now walk all objects which match this class type.  There may be more
00217                 // than one - Win32Disk for example has an entry for each logical drive.
00218                 USES_CONVERSION;
00219                 CComPtr<IWbemClassObject> piObject = 0; ULONG celt = 1;
00220                 while (!foundMatch && (hr=piEnum->Next(options_.Timeout, 1, &piObject, &celt)) == S_OK && celt == 1)
00221                 {
00222                         // Search out all the properties to retrieve
00223                         CComVariant val; CIMTYPE valType; CComBSTR bstrName;
00224                         piObject->BeginEnumeration(0);
00225                         while (piObject->Next(0, &bstrName, &val, &valType, NULL) == S_OK)
00226                         {
00227                                 if (bstrName.Length() > 0)
00228                                 {
00229                                         TString sName = W2T(bstrName), sValue;
00230                                         if (val.vt != VT_NULL && val.vt != VT_EMPTY)
00231                                         {
00232                                                 if (val.vt != VT_BSTR && 
00233                                                         (valType != CIM_OBJECT && valType != CIM_REFERENCE))
00234                                                         val.ChangeType(VT_BSTR);
00235                                                 if (val.vt == VT_BSTR)
00236                                                         sValue = W2T(val.bstrVal);
00237                                         }
00238 
00239                                         if (!foundMatch)
00240                                         {
00241                                                 if (!lstrcmpi(sName.c_str(),_T("__PATH")))
00242                                                 {
00243                                                         TString sPath = sValue;
00244                                                         nPos = sPath.find(scope_.c_str());
00245                                                         if (nPos != TString::npos)
00246                                                                 sPath = sPath.substr(nPos+scope_.length()+1);
00247                                                         if (lstrcmpi(sPath.c_str(), objectPath_.c_str()) != 0)
00248                                                         {
00249                                                                 coll_.Clear();
00250                                                                 syscoll_.Clear();
00251                                                                 break;
00252                                                         }
00253                                                         else foundMatch = true;
00254                                                 }
00255                                         }
00256 
00257                                         if (sName.length() > 2 && sName.substr(0,2) == _T("__"))
00258                                                 syscoll_.Add(sName, sValue);
00259                                         else
00260                                                 coll_.Add(sName, sValue);
00261                                 }
00262                                 val.Clear();
00263                         }
00264                         piObject->EndEnumeration();
00265                         piObject.Release();
00266                 }
00267                 return true;
00268         }
00269 
00270 // Property helpers
00271 public:
00272         TString get_Machine() const { return machine_; }
00273         void set_Machine(const TString& machine) { machine_ = machine; }
00274         TString get_Path() const { return objectPath_; }
00275         void set_Path(const TString& Path) { objectPath_ = Path; }
00276         TString get_Scope() const { return scope_; }
00277         void set_Scope(const TString& Scope) { scope_ = Scope; }
00278         ManagementOptions& get_Options() { return options_; }
00279         const ManagementNameValueCollection& get_Properties() { if (coll_.IsEmpty) Get(); return coll_; }
00280         const ManagementNameValueCollection& get_SysProperties() { if (syscoll_.IsEmpty) Get(); return syscoll_; }
00281 };

00430         bool Get()
00431         {
00432                 // Clear our collection
00433                 coll_.Clear();

00435                 // Determine if this is a query or a single object type.
00436                 bool isQuery = false;
00437                 if (query_.empty())
00438                         return false;
00439 
00440                 TString query = query_; _tcsupr(&query[0]);
00441                 if (query.find(_T("SELECT")) != TString::npos)
00442                         isQuery = true;
00443 
00444                 // Load up our WMI locator.
00445                 CComPtr<IWbemLocator> piLocator = 0;
00446                 HRESULT hr = piLocator.CoCreateInstance(CLSID_WbemLocator, NULL, CLSCTX_INPROC_SERVER);

00451                 TString connString;
00452                 if (!machine_.empty()) {
00453                         if (machine_[0] != _T('\\'))
00454                                 connString = _T("\\\\");
00455                         connString += machine_;
00456                         connString += _T("\\");
00457                 }
00458                 connString += scope_;

00460                 // Connect to the appropriate server
00461                 CComPtr<IWbemServices> piNamespace = 0;
00462                 hr = piLocator->ConnectServer(CComBSTR(connString.c_str()), NULL, NULL, 0L, 0L, NULL, NULL, &piNamespace);
00463                 if (FAILED(hr))
00464                         return false;
00465 
00466                 // Get an instance enum for this type.
00467                 CComPtr<IEnumWbemClassObject> piEnum = 0;

00469                 // If this is a query, then execute the query using WQL.
00470                 if (isQuery)
00471                 {
00472                         hr = piNamespace->ExecQuery(CComBSTR(L"WQL"), CComBSTR(query_.c_str()), 
00473                                 WBEM_FLAG_FORWARD_ONLY | WBEM_RETURN_IMMEDIATELY, NULL, &piEnum);
00474                 }
00475                 else
00476                 {
00477                         hr = piNamespace->CreateInstanceEnum(CComBSTR(query_.c_str()), 
00478                                 WBEM_FLAG_SHALLOW | WBEM_FLAG_FORWARD_ONLY | WBEM_RETURN_IMMEDIATELY, NULL, &piEnum);
00479                 }
00480 
00481                 if (FAILED(hr))
00482                         return false;

00504                                                         sValue = W2T(val.bstrVal);
00505                                         }

00507                                         // Found a object; create an instance
