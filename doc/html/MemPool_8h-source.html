00035 /****************************************************************************/
00036 // MemPool
00037 //
00038 // Class which encapsulates a small-alloc memory pool
00039 //
00040 /****************************************************************************/
00041 template <class T, int EXPANSION_SIZE = 4096, class LockType = JTI_MEMPOOL_LOCK_TYPE>
00042 class MemPool : public LockableObject<LockType>
00043 {
00044 // Class data
00045 private:
00046         typedef MemPool<T, EXPANSION_SIZE, LockType> _MemPtr;
00047         _MemPtr* pNext_;
00048 
00049 // Constructor
00050 public:
00051         MemPool(size_t size = EXPANSION_SIZE) { ExpandFreeList(size); }
00052         ~MemPool() {
00053                 _MemPtr* pNext = pNext_;
00054                 for (; pNext_ != NULL; pNext_ = pNext) {
00055                         pNext = pNext->pNext_; free(pNext_);
00056                 }
00057         }
00058 
00059         void* Alloc(size_t size) {
00060                 if (size != sizeof(T)) return ::JTI_NEW char[size];
00061                 CCSLock<MemPool> lock(this);
00062                 if (!pNext_) {
00063                         if (!ExpandFreeList())
00064                                 return NULL;
00065                 }
00066                 _MemPtr* pHead = pNext_;
00067                 pNext_ = pHead->pNext_;
00068                 return pHead;
00069         }
00070 
00071         void Free(void* pElem, size_t size) {
00072                 if (size != sizeof(T)) {
00073                         ::delete [] ((char*) pElem);
00074                         return;
00075                 }
00076                 CCSLock<MemPool> lock(this);
00077                 _MemPtr* pHead = static_cast<_MemPtr*>(pElem);
00078                 pHead->pNext_ = pNext_;
00079                 pNext_ = pHead;
00080         }
00081 
00082 // Internal methods
00083 private:
00084         bool ExpandFreeList(size_t count = EXPANSION_SIZE) {
00085         size_t size = std::max(sizeof(T),sizeof(_MemPtr*));
00086                 _MemPtr* pList = reinterpret_cast<_MemPtr*>(malloc(size));
00087                 if (!pList) throw std::bad_alloc();
00088                 pNext_ = pList;
00089                 for (size_t i = 0; i < count; ++i) {
00090                         pList->pNext_ = reinterpret_cast<_MemPtr*>(malloc(size));
00091                         pList = pList->pNext_;
00092                 }
00093                 pList->pNext_ = 0;
00094                 return true;
00095         }
00096 };
