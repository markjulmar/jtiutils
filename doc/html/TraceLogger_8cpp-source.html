00174 /*****************************************************************************
00175 ** Procedure:  TraceLogger_Base::Runner
00176 ** 
00177 ** Arguments:  void
00178 ** 
00179 ** Returns: void
00180 ** 
00181 ** Description: This is the runner for the logging engine
00182 **
00183 /****************************************************************************/
00184 void TraceLogger_Base::Runner()
00185 {
00186         HANDLE arrHandles[] = { Stopping_.get(), HasData_.get() };
00187         CCSLock<CriticalSectionLock> lockGuard(&lockElements_, false);
00188 
00189         for (;;)
00190         {
00191                 // Wait on either the STOP event, or data arriving.
00192                 DWORD waitCode = WaitForMultipleObjects(2, arrHandles, FALSE, INFINITE);
00193                 if (waitCode == WAIT_OBJECT_0 || waitCode == WAIT_ABANDONED_0)
00194                         break;
00195 
00196                 // Lock for exclusive access to the queue.
00197                 lockGuard.Lock();
00198                 if (!listElements_.empty())
00199                 {
00200                         // Copy the elements off.
00201                         LogElementList lstElems;
00202                         lstElems.swap(listElements_);
00203 
00204                         // Unlock and dispatch
00205                         lockGuard.Unlock();
00206                         std::for_each(lstElems.begin(), lstElems.end(), 
00207                                 std::bind1st(std::mem_fun(&TraceLogger_Base::DispatchSingle),this));
00208 
00209                         // Now test the size again; if we have added one while processing, allow the thread
00210                         // to continue working on new data.
00211                         lockGuard.Lock();
00212                         if (listElements_.empty())
00213                                 HasData_.ResetEvent();  //lint !e534
00214                 }
00215                 lockGuard.Unlock();
00216         }
00217 
00218         // Dispatch all the remaining elements.
00219         lockGuard.Lock();
00220         std::for_each(listElements_.begin(), listElements_.end(), 
00221                 std::bind1st(std::mem_fun(&TraceLogger_Base::DispatchSingle),this));
00222         listElements_.clear();
00223 
00224 }// TraceLogger_Base::Runner

00295 /*****************************************************************************
00296 ** Procedure:  TraceLogger_Base::DumpFile
00297 ** 
00298 ** Arguments:  nLevel - Trace Level
00299 **             fileName - name of the file to dump in the stream.
00300 ** 
00301 ** Returns: Success indicator
00302 ** 
00303 ** Description: This adds the contents of a file to the trace buffer.
00304 **
00305 /****************************************************************************/
00306 bool TraceLogger_Base::DumpFile(unsigned long nLevel, const char* fileName, bool isBinary)
00307 {
00308         // Get the total size of the file.
00309         struct _stat results;
00310         if (_stat(fileName, &results) != 0)
00311                 return false;
00312 
00313         try
00314         {
00315                 // Open the file as a memory mapped file.
00316                 MemoryMappedFile file(fileName, 
00317                         GENERIC_READ, FILE_SHARE_READ, NULL,
00318                         OPEN_EXISTING, FILE_FLAG_SEQUENTIAL_SCAN);
00319 
00320                 // Now dump it out to the stream.
00321                 if (isBinary)
00322                         InternalHexDump(nLevel, file.Buffer, static_cast<int>(file.Size));
00323                 else
00324                 {
00325                         if (!listHandlers_.empty())
00326                         {
00327                                 LogElement* ple = JTI_NEW LogElement(nLevel, 
00328                                                 std::string(reinterpret_cast<const char*>(file.Buffer), 
00329                                                 static_cast<int>(file.Size)));
00330                                 CCSLock<CriticalSectionLock> lockGuard(&lockElements_);
00331                                 if (threadHandle_ != INVALID_HANDLE_VALUE)
00332                                 {
00333                                         listElements_.push_back(ple);
00334                                         HasData_.SetEvent(); //lint !e534
00335                                 }
00336                                 else
00337                                         DispatchSingle(ple);
00338                         }
00339                 }       
00340         }
00341         catch(const std::exception&)
00342         {
00343                 return false;
00344         }
00345 
00346         return true;
00347 
00348 }// TraceLogger_Base::DumpFile

00456 {

00458         int nCount = 0, nLine = 0;

00479                 
00480                 for (i = 0; i < 16; ++i)
00481                         if (b[i] < 0x20 || b[i] > 0x7d)
