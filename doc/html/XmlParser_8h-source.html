00445                 if (this != &rhs) {

00452 // Properties
00453 public:

00455         __declspec(property(get=size)) int Count;

00457 
00458 // Accessors
00459 public:
00460         bool empty() const { return arrNodes_.empty(); }
00461         int size() const { return  static_cast<int>(arrNodes_.size()); }
00462 
00463         void add(const XmlNode& xmlNode)
00464         {
00465                 CCSLock<XmlNodeImpl> _lockGuard(node_.pImpl_);
00466                 if (xmlNode == node_)
00467                         throw(std::runtime_error("Cannot insert self into child array!"));
00468                 XmlNodeImpl* pImpl = xmlNode.pImpl_;
00469                 NodeArray::iterator it = std::find(arrNodes_.begin(), arrNodes_.end(), pImpl);
00470                 if (it != arrNodes_.end())
00471                         throw(std::runtime_error("Multiple insertion of same node within array not allowed!"));
00472                 pImpl->AddRef();
00473                 arrNodes_.push_back(pImpl);
00474         }
00475 
00476         bool remove(const XmlNode& xmlNode)
00477         {
00478                 CCSLock<XmlNodeImpl> _lockGuard(node_.pImpl_);
00479                 XmlNodeImpl* pImpl = xmlNode.pImpl_;
00480                 NodeArray::iterator it = std::find(arrNodes_.begin(), arrNodes_.end(), pImpl);
00481                 if (it != arrNodes_.end())
00482                 {
00483                         (*it)->Release();
00484                         arrNodes_.erase(it);
00485                         return true;
00486                 }
00487                 return false;
00488         }
00489 
00490         XmlNode find(const char* pszName) const
00491         {
00492                 CCSLock<XmlNodeImpl> _lockGuard(node_.pImpl_);
00493                 for (NodeArray::iterator it = arrNodes_.begin(); it != arrNodes_.end(); ++it)
00494                 {
00495                         if ((*it)->name_ == pszName)
00496                                 return XmlNode(*it);
00497                 }
00498                 return XmlNode();
00499         }

00501         XmlNode Index(int index) const
00502         {

00504                 if (index < 0 || index >= static_cast<int>(arrNodes_.size()))

00509         XmlNode operator[](int index) const
00510         {

00512         }
