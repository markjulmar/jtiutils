00051         // EncodeFile - This function takes a filename and returns a Base64 
00052         // encoded buffer in a string.
00053         static TString EncodeFile(const TCHAR* pszFilename)
00054         {
00055                 // Open the file for read access.
00056                 HANDLE hFile = CreateFile(pszFilename, GENERIC_READ, FILE_SHARE_READ, NULL, OPEN_EXISTING, FILE_FLAG_SEQUENTIAL_SCAN, NULL);
00057                 if (hFile != INVALID_HANDLE_VALUE)
00058                 {
00059                         // Get the file size, we assume the file is smaller than 4G
00060                         DWORD dwSize = GetFileSize(hFile, NULL);
00061 
00062                         // Create a file mapping object; it increments the ref count on the original file so close
00063                         // it now to avoid resource leaks.
00064                         HANDLE hFileMap = CreateFileMapping(hFile, NULL, PAGE_READONLY, 0, dwSize, NULL);
00065                         CloseHandle(hFile);
00066                         if (hFileMap != NULL)
00067                         {
00068                                 // Map a view of the file; again close the handle we used.
00069                                 LPVOID lpvFile = MapViewOfFile(hFileMap, FILE_MAP_READ, 0, 0, 0);
00070                                 CloseHandle(hFileMap);
00071                                 if (lpvFile != NULL)
00072                                 {
00073                                         // Encode and release the view on the file
00074                                         TString s = EncodeBuffer(lpvFile, dwSize);
00075                                         UnmapViewOfFile(lpvFile);
00076                                         return s;
00077                                 }
00078                         }
00079                 }
00080                 return TString();
00081         }

00083         // EncodeBuffer - This function takes a buffer/length combination and returns
00084         // a Base64 encoded buffer in a string.
00085         static TString EncodeBuffer(const void* in, size_t length)
00086         {
00087                 const unsigned char* data = reinterpret_cast<const unsigned char*>(in);
00088 
00089                 // 3 unsigned chars encode to 4 chars.  Output is always an even
00090                 // multiple of 4 characters.
00091                 TCHAR* alphabet =       _T("ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/=");
00092                 TCHAR out[5] = {0}; 
00093                 TString s; 
00094                 
00095                 s.reserve(((length + 2) / 3) * 4);
00096                 for (size_t i = 0; i < length; i += 3)
00097                 {
00098                         bool quad = false, trip = false;
00099                         int val = (0xFF & (int) data[i]);
00100                         val <<= 8;
00101                         if ((i + 1) < length) 
00102                         {
00103                                 val |= (0xFF & (int) data[i + 1]);
00104                                 trip = true;
00105                         }
00106 
00107                         val <<= 8;
00108                         if ((i + 2) < length) 
00109                         {
00110                                 val |= (0xFF & (int) data[i + 2]);
00111                                 quad = true;
00112                         }
00113                         
00114                         out[3] = alphabet[(quad ? (val & 0x3F) : 64)];
00115                         val >>= 6;
00116                         out[2] = alphabet[(trip ? (val & 0x3F) : 64)];
00117                         val >>= 6;
00118                         out[1] = alphabet[val & 0x3F];
00119                         val >>= 6;
00120                         out[0] = alphabet[val & 0x3F];
00121                         
00122                         s.append(out);
00123                 }
00124                 return s;
00125         }

00132         static ByteArray DecodeBuffer(const void* inData, size_t length)
00133         {
00134                 const unsigned char* data = static_cast<const unsigned char*>(inData);
00135                 unsigned char base64Decode_[256]; int i;
00136                 for (i = 0; i < 256; i++)
00137                         base64Decode_[i] = 0xff;
00138                 for (i = 'A'; i <= 'Z'; i++)
00139                         base64Decode_[i] = (unsigned char)(i - 'A');
00140                 for (i = 'a'; i <= 'z'; i++)
00141                         base64Decode_[i] = (unsigned char)(26 + i - 'a');
00142                 for (i = '0'; i <= '9'; i++)
00143                         base64Decode_[i] = (unsigned char)(52 + i - '0');
00144                 base64Decode_['+'] = 62;
00145                 base64Decode_['/'] = 63;
00146 
00147                 size_t len = ((length + 3) / 4) * 3;
00148                 if (length > 0 && data[length - 1] == '=')
00149                         --len;
00150                 if (length > 1 && data[length - 2] == '=')
00151                         --len;
00152 
00153                 ByteArray out; 
00154                 out.reserve(len);
00155 
00156                 int shift = 0; // # of excess bits stored in accum
00157                 int accum = 0; // excess bits
00158 
00159                 for (size_t ix = 0; ix < length; ++ix)
00160                 {
00161                         TCHAR ch = data[ix];
00162                         int value = base64Decode_[ch];  // ignore high byte of char
00163                         if (value >= 0)                                 // skip over non-code
00164                         {
00165                                 accum <<= 6;                            // bits shift up by 6 each time through
00166                                 shift += 6;                                     // loop, with new bits being put in
00167                                 accum |= value;                         // at the bottom.
00168                                 if (shift >= 8)                         // whenever there are 8 or more shifted in,
00169                                 { 
00170                                         shift -= 8;                             // write them out (from the top, leaving any
00171                                         unsigned char ch = static_cast<unsigned char>((accum >> shift) & 0xff);
00172                                         out.push_back(ch);              // excess at the bottom for next iteration.
00173                                 }
00174                         }
00175                 }
00176                 return out;
00177         }
